 18 var sys = require('sys');
 19 var path = require('path');
 20 
 21 var sprintf = require('extern/sprintf').sprintf;
 22 
 23 var manifest = require('manifest/index');
 24 var manifest_constants = require('manifest/constants');
 25 
 26 /** Configuration options for validate-manifest subcommand */
 27 exports.config = {
 28   'short_description': 'Validates the cast.json manifest',
 29   'long_description': 'Validates that the cast.json manifest files contains all the required fields.',
 30   'required_arguments' : [['path', 'Path to the directory containing the cast.json file']],
 31   'optional_arguments': []
 32 };
 33 
 34 /**
 35  * Handler for creating a bundle.
 36  * @param {Object} args Command line arguments.
 37  * @param {String} args.version Application version number to create.
 38  * @param {String} [args.apppath=process.cwd()] Application path to operate on.
 39  */
 40 exports.handle_command = function(args)
 41 {
 42   var application_path;
 43 
 44   if (!args.path || args.path === '.') {
 45     application_path = process.cwd();
 46   }
 47   else {
 48     application_path = args.path;
 49   }
 50 
 51   var manifest_file = path.join(application_path, manifest_constants.MANIFEST_FILENAME);
 52 
 53   // Check that the manifest file exists
 54   path.exists(manifest_file, function(exists) {
 55     if (!exists) {
 56       sys.puts(sprintf('Failed to find a manifest file %s in the directory %s, quitting',
 57                        manifest_constants.MANIFEST_FILENAME, application_path));
 58 
 59       return;
 60     }
 61 
 62     manifest.validate_manifest(manifest_file, function(error, manifest_object) {
 63       if (error) {
 64         sys.puts(sprintf('Manifest file validation failed: %s', error.message));
 65 
 66         return;
 67       }
 68 
 69       sys.puts('Manifest file is valid.');
 70    });
 71  });
 72 };
