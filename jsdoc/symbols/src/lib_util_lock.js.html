 18 var fs = require('fs');
 19 var path = require('path');
 20 var constants = require('util/constants');
 21 
 22 exports.with_lock = function(lock_file_path, callback) {
 23   fs.open(lock_file_path, constants.O_EXCL | constants.O_CREAT, 644, function(error, fd) {
 24     if (error) {
 25       return callback(new Error('Error acquiring a lock'));
 26     }
 27 
 28     // Release a lock and call an optional callback
 29     function release(callback) {
 30       fs.close(fd, function(error) {
 31         fs.unlink(lock_file_path, function(error) {
 32           if (callback) {
 33             callback();
 34           }
 35         });
 36       });
 37     }
 38 
 39     // Try to remove the lock on SIGINT (if one exists)
 40     process.on('SIGINT', function()
 41     {
 42       release(function() {
 43         process.exit(0);
 44       });
 45     });
 46 
 47     callback(null, release);
 48   });
 49 };
