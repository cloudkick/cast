 45 var sys = require('sys');
 46 var spawn = require('child_process').spawn;
 47 var exec = require('child_process').exec;
 48 var path = require('path');

 50 var sprintf = require('extern/sprintf').sprintf;
 51 var async = require('extern/async');

 53 var log = require('util/log');
 54 var config = require('util/config');
 55 var fsutil = require('util/fs');

 58 exports.extract_tarball = function(tarball, target, mode, cb) {
 59   fs.mkdir(target, mode, function(err) {
 60     if (err) {
 61       return cb(err);
 62     }
 63 
 64     var completed = false;
 65 
 66     function on_error(err) {
 67       completed = true;
 68       fsutil.rmtree(target, function() {
 69         return cb(err);
 70       });
 71     }
 72 
 73     var gzip = spawn(config.get().gzip, ['--decompress', '--stdout', '-f', tarball]);
 74     var tar = spawn(config.get().tar, ['vx', '--file=-', '--strip-components=1', '-C', target]);
 75 
 76     gzip.on('exit', function(code) {
 77       if (code && !completed) {
 78         log.err('Error decompressing bundle, gzip exited with code: ' + code);
 79         try {
 80           tar.kill();
 81         }
 82         catch (err) {
 83           log.warn('Unable to kill tar: ' + err);
 84         }
 85         return on_error(new Error('Error decompressing bundle'));
 86       }
 87     });
 88 
 89     tar.on('exit', function(code) {
 90       if (!completed) {
 91         if (code) {
 92           return on_error(new Error('Error extracting tarball. tar exited with code: ' + code));
 93         }
 94         else {
 95           completed = true;
 96           cb();
 97         }
 98       }
 99     });
100 
101     sys.pump(gzip.stdout, tar.stdin);
102   });
103 };

105 /**
106  * Create a tarball from the specified path.
107  *
108  * @param {String} source_path Path to the directory or the file you want to archive.
109  * @param {String} destination_path Path to the directory where the tarball will be saved.
110  * @param {String} tarball_name The name of the tarball.
111  * @param {Object} options Options object.
112  * @param {Function} callback A callback called with a possible error.
113  *
114  */
115 exports.create_tarball = function(source_path, target_path, tarball_name, options, callback) {
116   var default_options = {
117     'delete_if_exists': false,
118     'ignore_file_changed_error': false,
119     'exclude_pattern': '',
120     'store_in_tarball_name_directory': true // If true, all the files inside the archive will be stored in a
121                                             // directory named tarball_name minus the .tar.gz extension
122   };
123 
124   options = misc.merge(default_options, options);
125   var options_array = [];
126 
127   var tarball_path = path.join(target_path, tarball_name);
128 
129   async.parallel([function(callback) {
130       path.exists(source_path, function(exists) {
131         if (!exists) {
132           return callback(new Error('Source path does not exist'));
133         }
134 
135         callback();
136       });
137     },
138 
139     function(callback) {
140        path.exists(target_path, function(exists) {
141         if (!exists) {
142           return callback(new Error('Target path does not exist'));
143         }
144 
145         callback();
146       });
147     },
148 
149     function(callback) {
150       path.exists(tarball_path, function(exists) {
151         if (exists) {
152           if (!options.delete_if_exists) {
153             return callback(new Error(sprintf('Tarball already exists: %s', tarball_path)));
154           }
155 
156           fs.unlink(tarball_path, function(error) {
157             if (error) {
158               return callback(error);
159             }
160 
161             callback();
162           });
163         }
164 
165         callback();
166       });
167     }],
168 
169     function(error, results) {
170       if (error) {
171         return callback(error);
172       }
173 
174       var base_name = tarball_name.replace('.tar.gz', '');
175 
176       function handle_error(error) {
177         fs.unlink(tarball_path, function() {
178           return callback(error);
179         });
180       }
181 
182       if (options.exclude_pattern) {
183         options_array.push(sprintf('--exclude=%s', options.exclude_pattern));
184       }
185 
186       if (options.store_in_tarball_name_directory) {
187         options_array.push(sprintf('--transform \'s,^./*,%s/,\'', base_name));
188       }
189 
190       var tarcmd = sprintf('%s czpf "%s" -C "%s" --exclude=%s %s .', config.get().tar, tarball_path, source_path, base_name, options_array.join(' '));
191       var tar = exec(tarcmd, {}, function(error, stdout, stderr) {
192         if (error && !(options.ignore_file_changed_error && error.code === 1 && stderr.match(/file changed as we read it/i))) {
193           log.err(sprintf('Creating tarball failed: (%s) %s\nCommand was: %s', error.code, stderr, tarcmd));
194           return handle_error(new Error(sprintf('Creating tarball failed: %s', stderr)));
195         }
196 
197         log.info(sprintf('Successfully created tarball: %s', tarball_name));
198         callback();
199       });
200     }
201   );
202 };

205 /**
206  * Return an array of files and directories in a tarball.
207  *
208  * @param {String} tarball_path Path to the tarball.
209  * @param {Function} callback A callback called with a possible error as the first argument and an array
210  *                            with all the files and directories in an archive as the second argument on success.
211  *
212  */
213 exports.get_tarball_file_list = function(tarball_path, callback) {
214   var file_list = [];
215 
216   var tarcmd = sprintf('%s tvf "%s"', config.get().tar, tarball_path);
217   var tar = exec(tarcmd, function(error, stdout, stderr) {
218     var line, match;
219 
220     if (error) {
221       return callback(error, null);
222     }
223 
224     var lines = stdout.split(/\n\r?/g);
225 
226     for (var i = 0; i < lines.length; i++) {
227       line = lines[i];
228       match = line.match(/^.*\s(.*?)$/);
229 
230       if (match) {
231         file_list.push(match[1].trim());
232       }
233     }
234 
235     callback(null, file_list);
236   });
237 };
