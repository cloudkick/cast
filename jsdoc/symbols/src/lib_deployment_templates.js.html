 18 var fs = require('fs');
 19 
 20 var async = require('extern/async');
 21 
 22 var templates = require('util/templates');
 23 var misc = require('util/misc');
 24 
 25 var manifest = require('manifest');
 26 
 27 /*
 28  * Realize (render and save) application templates.
 29  *
 30  * @param {Object} instance_data
 31  * @param {Object} manifest_object
 32  * @param {String} templates_path
 33  * @oaram {String} target_path
 34  * @param {Function} callback
 35  *
 36  */
 37 var realize_application_templates = function(manifest_object, instance_data, templates_path, target_path, callback) {
 38   var context;
 39 
 40   context = { 'manifest': manifest_object };
 41   if (instance_data) {
 42     context = misc.merge(context, { 'instance': instance_data });
 43   }
 44 
 45   if (manifest_object.hasOwnProperty('template_variables')) {
 46     // Assign "template_variables" property value to the property named "user_defined" on the
 47     // context object and delete it from the context.manifest
 48     context = misc.merge(context, {'user_defined': manifest_object.template_variables });
 49     delete context.manifest.template_variables;
 50   }
 51 
 52   templates.get_template_context(true, context, function(error, context) {
 53     if (error) {
 54       return callback(error);
 55     }
 56 
 57     templates.render_and_save_templates(templates_path, manifest_object.template_files, target_path, context,
 58                                         function(error) {
 59       if (error) {
 60         return callback(error);
 61       }
 62 
 63       callback();
 64     });
 65   });
 66 };
 67 
 68 exports.realize_application_templates = realize_application_templates;
