 29 /**
 30  * Validators functions for the manifest file values.
 31  *
 32  * @param {Varies} value Value which is being validated.
 33  * @param {Object} options Options objects which contains additional information about the manifest.
 34  * @param {Function} callback Callback which is called with the error as the first argument if the validation fails,
 35  *                            without argument otherwise.
 36  */
 37 var is_valid_type = function(value, options, callback) {
 38   if (!misc.in_array(value, constants.APPLICATION_TYPES)) {
 39     callback(true);
 40     return;
 41   }
 42 
 43   callback();
 44 };

 46 var is_valid_string = function(value, options, callback) {
 47   if (typeof(value) === 'string') {
 48     return callback();
 49   }
 50   else if (typeof(value) === 'object') {
 51     if (value instanceof String) {
 52       return callback();
 53     }
 54   }
 55 
 56   callback(new Error('value is not a string'));
 57 };
 58 
 59 var is_valid_number = function(value, options, callback) {
 60   if (typeof(value) === 'number') {
 61     return callback();
 62   }
 63   else if (typeof(value) === 'object') {
 64     if (value instanceof Number) {
 65       return callback();
 66     }
 67   }
 68 
 69   callback(new Error('value is not a number'));
 70 };
 71 
 72 var is_valid_port = function(value, options, callback) {
 73   is_valid_number(value, options, function(error) {
 74     if (error) {
 75       return callback(error);
 76     }
 77 
 78     if (!((value > 0) && (value <= 65535))) {
 79       return callback(new Error('Port number must be between 1 and 65535'));
 80     }
 81 
 82     callback();
 83   });
 84 };

 86 var is_valid_check = function(value, options, callback) {
 87   var check_name, check_module, check_class, index;
 88 
 89   if (typeof(value) !== 'object') {
 90     return callback(new Error('Health check item, must be an object'));
 91   }
 92 
 93   if (!value.hasOwnProperty('check')) {
 94       return callback(new Error('Missing "check" property'));
 95   }
 96 
 97   check_name = value.check;
 98 
 99   // Check for a valid check name
100   index = misc.array_find(check_name,
101                           health.available_checks,
102                           null,
103                           function(item, needle) {
104                             return item[0] === needle;
105                           });
106 
107   if (index === false) {
108     return callback(new Error(sprintf('Invalid check type: %s', check_name)));
109   }
110 
111   check_module = require(sprintf('health/checks/%s', check_name));
112   check_class = health.available_checks[index][1];
113 
114   // Create a check object which throws an exception if a required argument is missing
115   try {
116     var module = new check_module[check_class](value['arguments']);
117   }
118   catch (error) {
119     return callback(error);
120   }
121 
122   callback();
123 };

125 var is_valid_template = function(value, options, callback) {
126   if (!value) {
127     return callback(new Error('Template file is an empty string'));
128   }
129 
130   var template_path = path.join(options.manifest_path, value);
131   path.exists(template_path, function(exists) {
132     if (!exists) {
133       return callback(new Error('Template file does not exist'));
134     }
135 
136     var read_stream = fs.createReadStream(template_path);
137     var data_buffer = [];
138 
139     read_stream.on('data', function(chunk) {
140       data_buffer.push(chunk);
141     });
142 
143     read_stream.on('end', function() {
144       var data = data_buffer.join('');
145       var template = new T.Template();
146 
147       try {
148         template.parse(data);
149       }
150       catch (error) {
151         return callback(error);
152       }
153 
154       callback();
155     });
156   });
157 };

159 var is_valid_data_file = function(value, options, callback) {
160   is_valid_string(value, options, function(error) {
161     if (error) {
162       return callback(error);
163     }
164 
165     if (value.substr(0, 1) === '/' ||
166         value.substr(0, 2) === '~/' ||
167         value.substr(0, 2) === './' ||
168         value.substr(0, 2) === '..') {
169       return callback(new Error('All the paths must be relative to the application root directory'));
170     }
171 
172     callback();
173   });
174 };

176 var is_valid_template_variable = function(value, options, callback) {
177   var item;
178 
179   if (typeof(value) !== 'object' || value instanceof Array) {
180     return callback(new Error('Template variable item must be an object'));
181   }
182 
183   function validate_item(item) {
184     async.waterfall([
185       function(callback) {
186         is_valid_string(item, options, function(error) {
187           if (error) {
188             return callback(null, false);
189           }
190 
191           callback(null, true);
192         });
193       },
194 
195       function(is_valid, callback) {
196         if (is_valid) {
197           return callback();
198         }
199 
200         is_valid_number(item, options, function(error) {
201           callback(error);
202         });
203       }
204     ],
205 
206     function(error) {
207       if (error) {
208         return callback(new Error('Template variable values can only be strings or integers'));
209       }
210 
211       callback();
212     });
213   }
214 
215   for (var key in value) {
216     if (value.hasOwnProperty(key)) {
217       item = value[key];
218       validate_item(item);
219     }
220   }
221 };

223 /**
224  * Validator name to function mappings
225  */
226 var VALIDATORS = {
227   'valid_type': is_valid_type,
228   'valid_string': is_valid_string,
229   'valid_number': is_valid_number,
230   'valid_port': is_valid_port,
231   'valid_check': is_valid_check,
232   'valid_template': is_valid_template,
233   'valid_data_file': is_valid_data_file,
234   'valid_template_variable': is_valid_template_variable
235 };

237 /**
238  * Applies validator to an Array of values
239  *
240  * @param {Array} array Array of items to validate.
241  * @param {String} validator Validator to run against values.
242  * @param {?Object} options Extra options to the Validator.
243  * @param {Function} callback Invoked on completion, first parameter
244  *                            is present if errors happened.
245  */
246 exports.validate_array = function(array, validator, options, callback) {
247   async.forEach(array, function(item, callback) {
248     VALIDATORS[validator].call(null, item, options, function(error) {
249       callback(error);
250     });
251   },
252 
253   function(error) {
254     if (error) {
255       return callback(error);
256     }
257 
258     callback();
259   });
260 };
261 
262 /**
263  * Applies validator to a single value.
264  *
265  * @param {Object} value Item to validate.
266  * @param {String} validator Validator to run against values.
267  * @param {?Object} options Extra options to the Validator.
268  * @param {Function} callback Invoked on completion, first parameter
269  *                            is present if errors happened.
270  */
271 exports.validate_value = function(value, validator, options, callback) {
272   VALIDATORS[validator].call(null, value, options, function(error) {
273     callback(error);
274   });
275 };
