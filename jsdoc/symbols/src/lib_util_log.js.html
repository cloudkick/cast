 19 /**
 20  * Logging subsytem, providing log filtering and a simple set of log message
 21  * types.
 22  */
 23 
 24 var sys = require('sys');
 25 
 26 var sprintf = require('extern/sprintf').sprintf;
 27 var stacktrace = require('extern/stacktrace');
 28 
 29 var loglevels = {
 30   'nothing': 0,
 31   'crit': 1,
 32   'err': 2,
 33   'warn': 3,
 34   'info': 4,
 35   'debug': 5
 36 };
 37 
 38 var loglevel_strs = [];
 39 var loglevel = loglevels.debug;
 40 
 41 function logit(level, inargs) {
 42   if (level <= loglevel) {
 43     sys.log(loglevel_strs[level] + ': ' + sprintf.apply({}, inargs));
 44   }
 45 }
 46 
 47 exports.set_loglevel = function(level) {
 48   loglevel = loglevels[level];
 49 };
 50 
 51 exports.trace = function(label) {
 52   var err = new Error();
 53   err.name = 'Trace';
 54   err.message = label || '';
 55   Error.captureStackTrace(err, arguments.callee);
 56   return err.stack;
 57 };
 58 
 59 (function() {
 60   var localf = function(attrname) {
 61     exports[attrname] = (function(level) {
 62       return function() {
 63         return logit(level, arguments);
 64         };
 65     })(loglevels[attrname]);
 66   };
 67   for (var attrname in loglevels) {
 68     if (loglevels.hasOwnProperty(attrname)) {
 69       loglevel_strs[loglevels[attrname]] = attrname;
 70       localf(attrname);
 71     }
 72   }
 73 })();
