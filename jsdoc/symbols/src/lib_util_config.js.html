 32 var norris = require('norris');
 33 var service_management = require('service_management');

 36   'port': 49443, /* 49152 through 65535 are 'Private' Ports from IANA.
 37                   * It'd be nice to register one for Cast at somepoint... */
 38 
 39   'data_root': '/opt/cast/',
 40 
 41   'service_dir_enabled': null, // Directory in which the enabled runit services reside. This path can be relative
 42                                // or absolute. If it is relative, it is joined with the data_root path.
 43 
 44   // This directories are relative to the data root
 45   'data_dir': 'data', // Stores application data files
 46   'service_dir_available': 'services', // Directory in which the available runit services reside
 47   'bundle_dir': 'bundles', // Stores bundle tarballs
 48   'extracted_dir': 'extracted', // Stores extracted bundles
 49   'app_dir': 'applications', // Stores application instances
 50 
 51   /* TODO: Enable SSL again */
 52   'ssl_enabled': false,
 53   'ssl_cert': 'agent.crt', // Relative path to agent ssl certificate
 54   'ssl_key': 'agent.key', // Relative path to agent ssl private key
 55   'ssl_ciphers': 'RC4-SHA:AES128-SHA:AES256-SHA', // enabled ciphers for SSL
 56 
 57   'hmac_algorithm': 'sha256',
 58 
 59   /* TODO: make this a client certificate? */
 60   'secret': null,
 61 
 62   'fileread_buffer_size': (1024 * 64),
 63   'runsvdir_binary': 'runsvdir',
 64   'gzip': 'gzip',
 65   'tar': null,
 66   'norris_ttl': (60 * 10),
 67 
 68   // These are used by a client
 69   'temp_directory': '.cast-temp',

 75 // Which two settings to join together to form a new absolute path.
 76 // Both paths which are joined together are saved in the config_current object where the property name is the name
 77 // of the second variable (basically, the value of the second variable gets replaced with the new joined value).
 78 var paths_to_join = [['data_root', 'data_dir'],
 79                      ['data_root', 'service_dir_available'],
 80                      ['data_root', 'bundle_dir'],
 81                      ['data_root', 'extracted_dir'],
 82                      ['data_root', 'app_dir'],
 83                      ['data_root', 'ssl_cert'],
 84                      ['data_root', 'ssl_key']];
 85 
 86 var poffset = 0;
 87 
 88 exports.config_files = ['~/.cast/config.json', '/etc/cast.conf'];
 89 exports.config_current = {};
 90 
 91 exports.setup = function(cb) {
 92   function join_paths(callback) {
 93     // Join the paths listed in paths_to_join together
 94     var service_dir_enabled, path1_setting, path2_setting, path1, path2;
 95 
 96     // If the service_dir_enabled option is defined and it's not absolute, add it to the paths_to_join
 97     // array
 98     service_dir_enabled = exports.config_current.service_dir_enabled;
 99     if (service_dir_enabled && service_dir_enabled[0] !== '/') {
100       paths_to_join.push(['data_root', 'service_dir_enabled']);
101     }
102 
103     paths_to_join.forEach(function(path_tuple) {
104       path1_setting = path_tuple[0];
105       path2_setting = path_tuple[1];
106 
107       path1 = exports.config_current[path1_setting];
108       path2 = exports.config_current[path2_setting];
109 
110       exports.config_current[path2_setting] = path.join(path1, path2);
111     });
112 
113     callback(null);
114   }
115 
116   function merge_manager_config_defaults(callback) {
117     // Merge the service manager default config values
118     var manager_name = exports.config_current.service_manager;
119     var manager = service_management.get_manager(manager_name).config_defaults;
120     var manager_properties, property, value;
121 
122     exports.config_current[manager_name] = {};
123     manager_properties = Object.keys(manager);
124 
125     for (var i = 0; i < manager_properties.length; i++) {
126       property = manager_properties[i];
127       value = manager[property];
128 
129       exports.config_current[manager_name][property] = value;
130     }
131 
132     callback(null);
133   }
134 
135   function resolve_via_norris(callback, err) {
136     // Resolve config values which depend on the norris facts
137     if (err) {
138       cb(err);
139       return;
140     }
141 
142     norris.get(function(facts) {
143       if (exports.config_current.tar === null) {
144         exports.config_current.tar = facts.gnutar;
145       }
146 
147       if (exports.config_current.service_dir_enabled === null) {
148         exports.config_current.service_dir_enabled = facts.runit_services_path;
149       }
150 
151       if (exports.config_current.ssl_enabled && exports.config_current.secret === null) {
152         callback(new Error('Configuration is missing shared secret. Hint: Set secret in /etc/cast.conf'));
153         return;
154       }
155 
156       callback(null);
157     });
158   }
159 
160   exports.config_current = merge(exports.config_current, defaults);
161   function dostep() {
162     var p = expanduser(exports.config_files[poffset]);
163     ufs.json_file(p, function(err, obj) {
164       if (!err) {
165         exports.config_current = merge(exports.config_current, obj);
166       }
167 
168       poffset++;
169       if (exports.config_files.length === poffset) {
170         async.series([
171           resolve_via_norris,
172           join_paths,
173           merge_manager_config_defaults
174         ],
175 
176         function(err) {
177           cb(err);
178         });
179       }
180       else {
181         dostep();
182       }
183     });
184   }
185 
186   dostep();
187 };
188 
189 exports.get = function() {
190   return exports.config_current;
191 };
