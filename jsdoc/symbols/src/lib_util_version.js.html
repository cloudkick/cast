 18 /**
 19  * Provides the current version of cast.
 20  *
 21  * These are written in an easily parsable format for use by
 22  * other (non-js) scripts to easily grep out the version number.
 23  */
 24 
 25 var CAST_VERSION_MAJOR = 0;
 26 var CAST_VERSION_MINOR = 1;
 27 var CAST_VERSION_PATCH = 0;
 28 
 29 /* change this on release tags */
 30 var CAST_IS_DEV = true;
 31 
 32 /** Major version */
 33 exports.MAJOR = CAST_VERSION_MAJOR;
 34 /** Minor version */
 35 exports.MINOR = CAST_VERSION_MINOR;
 36 /** Patch level */
 37 exports.PATCH = CAST_VERSION_PATCH;
 38 /** Boolean testing if this is a development version or release version */
 39 exports.IS_DEV = CAST_IS_DEV;
 40 
 41 /**
 42  * Get the current version of Cast.
 43  * @return {String} Dot'ed format string of the version.
 44  */
 45 exports.toString = function()
 46 {
 47   var dstr = '-dev';
 48 
 49   if (exports.IS_DEV === false) {
 50     dstr = '-release';
 51   }
 52 
 53   return 'cast-' + CAST_VERSION_MAJOR + '.' + CAST_VERSION_MINOR + '.' + CAST_VERSION_PATCH + '' + dstr;
 54 };
