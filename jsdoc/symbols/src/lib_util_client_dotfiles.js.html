 23 var fs = require('fs');
 24 var path = require('path');
 25 
 26 var async = require('extern/async');
 27 
 28 var client_config = require('util/config');
 29 var misc = require('util/misc');
 30 var ufs = require('util/fs');
 31 var expanduser = require('util/misc').expanduser;
 32 var fsutil = require('util/fs');
 33 
 34 
 35 /**
 36  * The path to the dot cast directory (currently ~/.cast)
 37  */
 38 var dot_cast = exports.dot_cast_path = misc.expanduser('~/.cast');
 39 
 40 /**
 41  * The path to the remotes.json file in the dot cast directory
 42  */
 43 var dot_cast_remotes = exports.dot_cast_remotes_path = path.join(dot_cast, 'remotes.json');
 44 
 45 /**
 46  * Get the path to the cast project directory for a project
 47  *
 48  * @param {String} projectpath  The path to the root of the project.
 49  */
 50 exports.dot_cast_project_path = function(projectpath) {
 51   return path.join(projectpath, '.cast-project');
 52 };
 53 
 54 /**
 55  * Create the user's ~/.cast directory if it doesn't exist.
 56  *
 57  * @param {Function} callback A callback fired with a possible error.
 58  */
 59 exports.ensure_dot_cast = function(callback) {
 60   fsutil.ensure_directory(dot_cast, callback);
 61 };
 62 
 63 /**
 64  * Creates a dot cast project directory if it doesn't exist.
 65  *
 66  * @param {String} projectpath The path to the project.
 67  * @param {Function} callback A callback fired upon completion with (err, dot_cast_project_path).
 68  */
 69 exports.ensure_dot_cast_project = function(projectpath, callback) {
 70   var dot_cast_project_path = exports.dot_cast_project_path(projectpath);
 71   fsutil.ensure_directory(dot_cast_project_path, function(err) {
 72     return callback(err, dot_cast_project_path);
 73   });
 74 };
 75 
 76 /**
 77  * Get the remotes object which will default to an empty object if
 78  * the remotes file doesn't exist.
 79  *
 80  * @param {Function} callback A callback fired with (err, remotes).
 81  */
 82 exports.get_remotes = function(callback) {
 83   path.exists(dot_cast_remotes, function(exists) {
 84     if (!exists) {
 85       return callback(null, {});
 86     }
 87     fs.stat(dot_cast_remotes, function(err, stats) {
 88       if (err) {
 89         return callback(err);
 90       }
 91       else if (!err && !stats.isFile()) {
 92         return callback(new Error(dot_cast_remotes + ' exists but is not a file'));
 93       }
 94 
 95       ufs.json_file(dot_cast_remotes, function(err, remotes) {
 96         if (err) {
 97           callback(err);
 98         }
 99         else {
100           callback(null, remotes);
101         }
102       });
103     });
104   });
105 };
106 
107 /*
108  * Return a default remote.
109  *
110  * If a remotes file does not exist or there is a no default remote, callback will be called with null as
111  * the second argument.
112  *
113  * @param {Function} callback Callback which is called with an error as the first argument and remote object
114  *                            as the second one if a default remote is found.
115  */
116 exports.get_default_remote = function(callback) {
117   var remote, default_remote;
118   exports.get_remotes(function(err, remotes) {
119     if (err) {
120       callback(err);
121       return;
122     }
123 
124     if (Object.keys(remotes) === 0) {
125       callback(null, null);
126       return;
127     }
128 
129     function is_default(item, callback) {
130       remote = remotes[item];
131 
132       if (remote.is_default === true) {
133         callback(true);
134         return;
135       }
136 
137       callback(false);
138     }
139 
140     async.filter(Object.keys(remotes), is_default, function(results) {
141       if (!results) {
142         // No default remote found
143         callback(null, null);
144         return;
145       }
146 
147       // In case there are multi default remotes found, always return only one
148       default_remote = results[0];
149       callback(null, remotes[default_remote]);
150     });
151   });
152 };
