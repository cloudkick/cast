 18 var sys = require('sys');
 19 var path = require('path');
 20 var http = require('util/http');
 21 
 22 var sprintf = require('extern/sprintf').sprintf;
 23 
 24 var config = {
 25   'short_description': 'Print log file for a specified service',
 26   'long_description': 'Print a log file for a specified service. ' +
 27                       'If a --follow switch is provided, new data is ' +
 28                       'printed as soon as it is available.',
 29   'required_arguments' : [['name', 'Service name']],
 30   'optional_arguments': [['bytes', 'How many bytes from the end of the log file to print']],
 31   'options': [
 32     {
 33       names: ['--follow'],
 34       dest: 'follow',
 35       action: 'store_true',
 36       desc: 'Output new data as some as it is available (aka tail -F)'
 37     }
 38   ]
 39 };
 40 
 41 function handle_command(args) {
 42   var service_name, remote_path, bytes;
 43 
 44   service_name = args.name;
 45 
 46   if (args.bytes) {
 47     bytes = args.bytes;
 48   }
 49   else {
 50     // Return 500 bytes from the end of the log file by default
 51     bytes = 500;
 52   }
 53 
 54   if (args.follow) {
 55     remote_path = path.join('/services/', service_name, '/tail/', bytes, '/follow/');
 56   }
 57   else {
 58     remote_path = path.join('/services/', service_name, '/tail/', bytes, '/');
 59   }
 60 
 61   service_name = args.name;
 62 
 63   var client = http.client();
 64 
 65   var request = client.request('GET', remote_path, {
 66     'host': 'localhost'
 67   });
 68 
 69   request.on('response', function(response) {
 70     response.setEncoding('ascii');
 71 
 72     response.on('data', function(chunk) {
 73       if (response.statusCode !== 200) {
 74         var error = JSON.parse(chunk);
 75         sys.puts(sprintf('Error: %s', error.message));
 76 
 77         return;
 78       }
 79 
 80       sys.print(chunk);
 81     });
 82   });
 83 
 84   // Send the request
 85   request.end();
 86 }
 87 
 88 exports.config = config;
 89 exports.handle_command = handle_command;
