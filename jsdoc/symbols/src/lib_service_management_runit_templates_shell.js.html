 18 var path = require('path');
 19 
 20 var sprintf = require('extern/sprintf').sprintf;
 21 var config = require('util/config').get().runit;
 22 
 23 var get_template = function(service_name, service_path, entry_file, callback) {
 24   var template, entry_path;
 25 
 26   entry_path = path.join(service_path, entry_file);
 27 
 28   template = {
 29     run: sprintf('#!/bin/sh\nexec chpst -u%s /bin/sh %s 2>&1', config.service_user, entry_path),
 30     finish: sprintf("#!/bin/sh\necho 'Service %s stopped'\nsleep 1\nexit 0", service_name)
 31   };
 32 
 33   callback(null, template);
 34 };
 35 
 36 exports.get_template = get_template;
