 18 var sys = require('sys');
 19 var http = require('http');
 20 var url = require('url');
 21 
 22 var sprintf = require('extern/sprintf').sprintf;
 23 var log = require('util/log');
 24 
 25 var Check = require('health').Check;
 26 var CheckResult = require('health').CheckResult;
 27 var CheckStatus = require('health').CheckStatus;
 28 
 29 var config = {
 30   'name': 'HTTP check',
 31   'required_arguments': ['url', 'type', 'match_value'],
 32   'optional_arguments': [],
 33   'default_values': {},
 34   'types': {
 35     STATUS_CODE_MATCH: 0,
 36     BODY_REGEX_MATCH: 1
 37   }
 38 };
 39 
 40 function HTTPCheck(check_arguments) {
 41   Check.call(this, config.name, config.required_arguments,
 42              config.optional_arguments, config.default_values, check_arguments);
 43 }
 44 
 45 // Inherit from the base Check class
 46 sys.inherits(HTTPCheck, Check);
 47 
 48 HTTPCheck.prototype.run = function(callback) {
 49   log.info(sprintf('Running HTTP check (URL: %s, type: %s, match_value: %s)',
 50            this.check_arguments.url,
 51            this.check_arguments.type,
 52            this.check_arguments.match_value));
 53 
 54   var self = this;
 55   var result = new CheckResult();
 56   var timeout = false;
 57 
 58   var client = http.createClient(self.check_arguments.port,
 59                                  self.check_arguments.host,
 60                                  self.check_arguments.secure);
 61 
 62   var request = client.request('GET',
 63                                self.check_arguments.path,
 64                                {'host': self.check_arguments.host});
 65 
 66   function error_handler(exception) {
 67     result.status = CheckStatus.ERROR;
 68     result.details = sprintf('Check failed - returned exception: %s', exception);
 69 
 70     self.add_result(result, callback);
 71   }
 72 
 73   // Register generic error event handlers
 74   client.on('error', error_handler);
 75   request.on('error', error_handler);
 76 
 77   client.on('timeout', function() {
 78     timeout = true;
 79     // Close the connection, the check status is set in the client end event handler
 80     client.end();
 81   });
 82 
 83   client.on('end', function() {
 84     if (self.last_run_date !== result.date) {
 85       // If the result hasn't been added yet at this stage this means that the check has failed
 86       result.status = CheckStatus.ERROR;
 87 
 88       if (timeout === true) {
 89         result.details = sprintf('Timeout after %s seconds', self.check_arguments.timeout / 1000);
 90       }
 91       else {
 92         result.details = sprintf('Unknown error');
 93       }
 94 
 95       self.add_result(result, callback);
 96     }
 97 
 98     if (request.connection._writeQueue.length > 0) {
 99       // If for some strange reason all the buffer wasn't flushed out, manually destroy the connection
100       // (yeah, it's an ugly hack)
101       request.connection.destroy();
102     }
103   });
104 
105   // Send the request
106   request.end();
107 
108   request.on('response', function(response) {
109     if (self.check_arguments.type === config.types.STATUS_CODE_MATCH) {
110       var status_code = response.statusCode;
111 
112       if (status_code === self.check_arguments.match_value) {
113         result.status = CheckStatus.SUCCESS;
114         result.details = sprintf('Returned status code: %s', status_code);
115       }
116       else {
117         result.status = CheckStatus.ERROR;
118         result.details = sprintf('Returned status code:  %s', status_code);
119       }
120 
121       self.add_result(result, callback);
122     }
123     else if (self.check_arguments.type === config.types.BODY_REGEX_MATCH) {
124       // We only care about the content if it's a body regex match check
125       var data_buffer = [];
126 
127       response.on('data', function(chunk) {
128         data_buffer.push(chunk);
129       });
130 
131       response.on('end', function() {
132         var body = data_buffer.join('');
133 
134         if (body.match(self.check_arguments.match_value)) {
135           result.status = CheckStatus.SUCCESS;
136           result.details = sprintf('The response body matched the regular expression: %s',
137                                   self.check_arguments.match_value.toString());
138         }
139         else {
140           result.status = CheckStatus.ERROR;
141           result.details = sprintf('The response body didn\'t match the regular expression: %s',
142                                    self.check_arguments.match_value.toString());
143         }
144 
145         self.add_result(result, callback);
146       });
147     }
148   });
149 };
150 
151 HTTPCheck.prototype.format_arguments = function(required_arguments, optional_arguments,
152                                                 default_values, check_arguments) {
153   var formatted_arguments = Check.prototype.format_arguments.call(this, required_arguments,
154                                           optional_arguments, default_values,
155                                           check_arguments);
156 
157   var url_object = url.parse(formatted_arguments.url);
158 
159   formatted_arguments.host = url_object.hostname;
160   formatted_arguments.path = url_object.pathname || '/';
161 
162   formatted_arguments.secure = (url_object.protocol === 'https:') || false;
163   formatted_arguments.port = url_object.port || (formatted_arguments.secure === true ? 443 : 80);
164 
165   if (formatted_arguments.type === config.types.STATUS_CODE_MATCH) {
166     formatted_arguments.match_value = parseInt(formatted_arguments.match_value, 10);
167   }
168   else if (check_arguments.type === config.types.BODY_REGEX_MATCH) {
169     formatted_arguments.match_value = new RegExp(formatted_arguments.match_value);
170   }
171   else {
172     throw new Error(sprintf('Invalid check type: %s', formatted_arguments.type));
173   }
174 
175   return formatted_arguments;
176 };
177 
178 exports.config = config;
179 exports.HTTPCheck = HTTPCheck;
