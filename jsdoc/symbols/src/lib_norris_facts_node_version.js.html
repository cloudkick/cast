 18 exports.get = function(done) {
 19   var major, minor, patch;
 20   var node_version = process.version;
 21 
 22   node_version = node_version.replace('v', '');
 23   node_version = node_version.split('.');
 24 
 25   major = parseInt(node_version[0], 0);
 26   minor = parseInt(node_version[1], 0);
 27   patch = parseInt(node_version[2], 0);
 28 
 29   done({'node_version': {'major': major, 'minor': minor, 'patch': patch}});
 30 };
