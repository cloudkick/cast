 91 function make_path_relative(file, root, require_file, cb)
 92 {
 93   // Make the path absolute and normal
 94   if (file[0] !== '/') {
 95     file = path.normalize(path.join(process.cwd(), file));
 96   }
 97   else {
 98     file = path.normalize(file);
 99   }
100 
101   // This will be called shortly to make the path relative to the root
102   function relativise() {
103     if (file.indexOf(root) !== 0) {
104       return cb(new Error(file + " is not within " + root));
105     }
106 
107     file = file.slice(root.length);
108     if (file[0] === '/') {
109       file = file.slice(1);
110     }
111 
112     return cb(null, file);
113   }
114 
115   // Make sure the file exists, if required
116   if (require_file) {
117     fs.stat(file, function(err, stats) {
118       if (err) {
119         return cb(new Error("Unable to read " + file));
120       }
121       else if(!stats.isFile()) {
122         return cb(new Error(file + " is not a file"));
123       }
124       relativise();
125     });
126   }
127   else {
128     relativise();
129   }
130 }

132 /**
133  * Handler for creating a manifest.
134  * @param {Object} args Command line options and arguments.
135  */
136 exports.handle_command = function(args)
137 {
138   var apppath = args.apppath || process.cwd();
139   if (apppath[0] !== '/') {
140     apppath = path.normalize(path.join(process.cwd(), apppath));
141   }
142 
143   var manifest_path, tmppath, entry_path, templates, data_files;
144   var manifest = {};
145 
146   manifest_path = path.join(apppath, manifest_constants.MANIFEST_FILENAME);
147   tmppath = manifest_path + ".tmp";
148 
149   async.waterfall([
150     // See if a manifest already exists
151     function(callback) {
152       fs.stat(manifest_path, function(err, stats) {
153         // Errors other than non-existant manifest
154         if (err && err.errno !== constants.ENOENT) {
155           return callback(new Error("Error at manifest path: " + err.message));
156         }
157         // The manifest exists but is not a file
158         else if (!err && !stats.isFile()) {
159           return callback(new Error("Manifest path exists and is not a file"));
160         }
161         // The manifest exists and no --force
162         else if (!err && !args.force) {
163           return callback(new Error("Manifest already exists, use --force to overwrite"));
164         }
165         return callback();
166       });
167     },
168 
169     // Validate the application type
170     function(callback) {
171       if (args.type && apptypes.indexOf(args.type) < 0) {
172         return callback(new Error("Unknown application type specified"));
173       }
174       else {
175         return callback();
176       }
177     },
178 
179     // Validate the application root
180     function(callback) {
181       fs.stat(apppath, function(err, stats) {
182         if (!err && !stats.isDirectory()) {
183           err = new Error("Specified application root is not a directory");
184         }
185         return callback(err);
186       });
187     },
188 
189     // Validate the entry file
190     function(callback) {
191       make_path_relative(args.entry_file, apppath, true, function(err, file) {
192         entry_path = file;
193         return callback(err);
194       });
195     },
196 
197     // Validate the template files
198     function(callback) {
199       templates = [];
200       if (!args.templates) {
201         return callback();
202       }
203       async.forEach(args.templates, function(template, callback) {
204         // Template files must exist
205         make_path_relative(template, apppath, true, function(err, file) {
206           templates.push(file);
207           return callback(err);
208         });
209       },
210       function(err) {
211         return callback(err);
212       });
213     },
214 
215     // Validate the data files
216     function(callback) {
217       data_files = [];
218       if (!args.data_files) {
219         return callback();
220       }
221       async.forEach(args.data_files, function(dfile, callback) {
222         // Data files don't necessarily exist
223         make_path_relative(dfile, apppath, false, function(err, file) {
224           data_files.push(file);
225           return callback(err);
226         });
227       },
228       function(err) {
229         return callback(err);
230       });
231     },
232 
233     // Get the name of the application, if it wasn't provided
234     function(callback) {
235       if (args.name) {
236         manifest.name = args.name;
237         return callback();
238       }
239       term.prompt('Application Name:', false, false, function(name) {
240         manifest.name = name;
241         return callback();
242       });
243     },
244 
245     // Get a description of the application
246     function(callback) {
247       term.prompt('Application Description:', false, false, function(desc) {
248         manifest.description = desc;
249         return callback();
250       });
251     },
252 
253     // Get the application type
254     function(callback) {
255       if (args.type) {
256         manifest.type = args.type;
257         return callback();
258       }
259       term.prompt('Application Type:', apptypes, apptypes[0], function(type) {
260         manifest.type = type;
261         return callback();
262       });
263     },
264 
265     // Set the entry_file
266     function(callback) {
267       manifest.entry_file = entry_path;
268       return callback();
269     },
270 
271     // Store template and data file paths
272     function(callback) {
273       manifest.template_files = templates;
274       manifest.data_files = data_files;
275       return callback();
276     },
277 
278     // Ask about default checks
279     function(callback) {
280       if (manifest_constants.HEALTH_CHECKS[manifest.type].length !== 0) {
281         // TODO: It would be nice to give a more user-friendly check description
282         var query = "The '" + manifest.type + "' application type specifies " +
283                     " following default health checks:\n" +
284                     JSON.stringify(manifest_constants.HEALTH_CHECKS[manifest.type], null, 4) +
285                     "\nWould you like to enable these checks?";
286         term.prompt(query, ['y', 'n'], 'y', function(resp) {
287           if (resp === 'y') {
288             manifest.health_checks = manifest_constants.HEALTH_CHECKS[manifest.type];
289           }
290           return callback();
291         });
292       }
293       else {
294         return callback();
295       }
296     },
297 
298     // Store the manifest file
299     function(callback) {
300       var fstream = fs.createWriteStream(tmppath);
301 
302       fstream.write(JSON.stringify(manifest, null, 4));
303       fstream.end();
304 
305       fstream.on('close', function() {
306         fs.rename(tmppath, manifest_path, function(err) {
307           if (err) {
308             err = new Error("Error moving manifest file to permanent location: " + err);
309           }
310           return callback(err);
311         });
312       });
313 
314       fstream.on('error', function(err) {
315         fstream.removeAllListeners('end');
316         return callback(new Error('Error writing manifest file: ' + err));
317       });
318     }
319   ],
320   function(err) {
321     if (err) {
322       sys.puts('Error: ' + err.message);
323 
324       // Clean up temporary file if it exists
325       path.exists(tmppath, function(exists) {
326         if (exists) {
327           fs.unlink(tmppath, function(err) {
328             if (err) {
329               sys.puts('Unable to remove temporary file at ' + tmppath);
330             }
331           });
332         }
333       });
334     }
335     else {
336       sys.puts('Manifest created');
337     }
338   });
339 };
