 33 exports._base_client = function(port, host)
 34 {
 35   var conf = config.get();
 36 
 37   var client = null;
 38   if (conf.ssl_enabled) {
 39     var sslcontext = crypto.createCredentials({method: 'TLSv1_client_method'});
 40     sslcontext.context.setCiphers(conf.ssl_ciphers);
 41     client = http.createClient(port, host, true, sslcontext);
 42   }
 43   else {
 44     client = http.createClient(port, host);
 45   }
 46 
 47   client.headers = {'host': host, 'user-agent': version.toString()};
 48 
 49   // TOOD: SSL verify fingerprint
 50 
 51   var old_request = client.request;
 52   client.request = function(method, url, headers) {
 53     if (typeof(url) !== 'string') {
 54       headers = url;
 55       url = method;
 56       method = 'GET';
 57     }
 58 
 59     if (!headers) {
 60       headers = {};
 61     }
 62 
 63     hmac.create_http_hmac(method, url, headers);
 64 
 65     return old_request.call(this, method, url, headers);
 66   };
 67 
 68   return client;
 69 };

 83 exports.get_server_ssl_cert_info = function(remote_url, callback) {
 84   var conf = config.get();
 85   var callback_called = false;
 86   var url_object = url.parse(remote_url);
 87 
 88   if (url_object.protocol !== 'https:') {
 89     callback(new Error('Protocol must be https'));
 90     return;
 91   }
 92 
 93   var port = url_object.port || '443';
 94   var hostname = url_object.hostname;
 95 
 96   var sslcontext = crypto.createCredentials({method: 'TLSv1_client_method'});
 97   sslcontext.context.setCiphers(conf.ssl_ciphers);
 98 
 99   var client = http.createClient(port, hostname, true, sslcontext);
100   var request = client.request('HEAD', '/');
101 
102   request.on('response', function(response) {
103     var peer_cert = client.getPeerCertificate();
104 
105     if (!peer_cert) {
106       callback(new Error('Unable to fetch certificate information'));
107       return;
108     }
109 
110     if (callback_called) {
111         return;
112     }
113 
114     callback(null, peer_cert);
115     callback_called = true;
116   });
117 
118   function on_error(err) {
119     if (callback_called) {
120       return;
121     }
122 
123     callback(err);
124     callback_called = true;
125   }
126 
127   client.on('error', on_error);
128   request.on('error', on_error);
129 
130   request.end();
131 };

133 /**
134  * Write JSON to an HTTP response.
135  *
136  * @param {http.ServerResponse} res The HTTP Response to which to write the data.
137  * @param {Integer} code  The HTTP response code to use.
138  * @param {Object} data The data to write to the response.
139  */
140 exports.return_json = function(res, code, data) {
141   res.writeHead(code, { 'Content-Type': 'application/json' });
142   res.end(JSON.stringify(data));
143 };
144 
145 /**
146  * Return a JSON formatted error with a response code an an optinoal message.
147  *
148  * @param {http.ServerResponse} res The HTTP Response to which to write the
149  *                                    error.
150  * @param {Integer} code  The HTTP response code to use.
151  * @param {String} message  An optional error message.
152  */
153 exports.return_error = function(res, code, message) {
154   var response = {'code': code};
155   if (message !== undefined) {
156     response.message = message;
157   }
158   exports.return_json(res, code, response);
159 };

169 exports.put_file = function(source, url, options)
170 {
171   var stream = fs.createReadStream(source, {'bufferSize': options.bufferSize ? options.bufferSize : 1024 * 64});
172   options.data_stream = stream;
173   /* TODO: refactor into a UI thing in restler */
174   if (options.spinner) {
175     fs.stat(source, function(err, stats) {
176       if (err) {
177         throw err;
178       }
179       var spin = require('util/spinner').percent('Uploading ' + source, stats.size);
180       spin.start();
181       var l = 0;
182       stream.addListener('data', function(chunk) {
183         l += chunk.length;
184         spin.tick(l);
185       });
186       stream.addListener('close', function() {
187         spin.end();
188       });
189     });
190   }
191   return exports.put(url, options);
192 };

194 /**
195  * Issue a GET or POST request to the specified URL.
196  *
197  * @param {Object} client Client object (in most cases, one returned by exports.client function).
198  * @param {String} path Path to append to the remote prefix.
199  * @param {String} method What kind of request to perform (GET / POST / PUT).
200  * @param {Boolean} parse_json true to parse the response as JSON.
201  * @return {Object} Object with the following properties: headers, status_code and body
202  *                  If the parse_json is true, the body property is a response parsed as JSON,
203  *                  otherwise it is a string.
204  */
205 exports.get_response_with_client = function(client, path, method, parse_json, callback) {
206   var data_buffer, data, result_object;
207   var callback_called = false;
208 
209   if (!misc.in_array(method, ['GET', 'PUT', 'POST'])) {
210     return callback(new Errorf('Invalid method: %s', method));
211   }
212 
213   var request = client.request(method, path, client.headers);
214 
215   client.on('error', function(err) {
216     if (callback_called) {
217       return;
218     }
219 
220     callback(err, null);
221     callback_called = true;
222   });
223 
224   result_object = {'headers': null, 'status_code': null, 'body': null};
225 
226   request.on('response', function(response) {
227     data_buffer = [];
228     response.setEncoding('utf-8');
229 
230     response.on('error', callback);
231 
232     response.on('data', function(chunk) {
233       data_buffer.push(chunk);
234     });
235 
236     response.on('end', function() {
237       data = data_buffer.join('');
238 
239       if (parse_json && data.length > 0) {
240         data = JSON.parse(data);
241       }
242 
243       if (callback_called) {
244         return;
245       }
246 
247       result_object.headers = response.headers;
248       result_object.status_code = response.statusCode;
249       result_object.body = data;
250 
251       callback(null, result_object);
252       callback_called = true;
253     });
254   });
255 
256   request.end();
257 };

260 /**
261  * Issue a GET or POST request to the specified URL.
262  *
263  * @param {String} remote name of the remote.
264  * @param {String} path to append to the remote prefix.
265  * @param {String} method What kind of request to perform (GET / POST / PUT).
266  * @param {Boolean} parse_json true to parse the response as JSON.
267  * @return {*} Response String if parse_json is false, Object otherwise.
268  */
269 exports.get_response = function(remote, path, method, parse_json, callback) {
270   var client = exports.client(remote);
271   return exports.get_response_with_client(client, path, method, parse_json, callback);
272 };

303 // Modifies req and res to call logger with a log line on each res.end
304 // Think of it as "middleware"
305 function logify(req, res, logger) {
306   if (req.already_have_logify) {
307     return;
308   }
309   req.already_have_logify = true;
310   var end = res.end;
311   res.end = function() {
312     // Common Log Format (mostly)
313     logger((req.socket && req.socket.remoteAddress) + ' - - [' + (new Date()).toUTCString() + ']' +
314      ' \"' + req.method + ' ' + req.url +
315      ' HTTP/' + req.httpVersionMajor + '.' + req.httpVersionMinor + '\" ' +
316      res.statusCode + ' - \"' +
317      (req.headers.referer || '') + '\" \"' + (req.headers['user-agent'] ? req.headers['user-agent'].split(' ')[0] : '') + '\"');
318     return end.apply(this, arguments);
319   };
320 
321   var writeHead = res.writeHead;
322   res.writeHead = function(code) {
323     res.statusCode = code;
324     return writeHead.apply(this, arguments);
325   };
326 }
