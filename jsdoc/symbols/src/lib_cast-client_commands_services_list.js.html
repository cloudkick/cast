 18 var sys = require('sys');
 19 
 20 var sprintf = require('extern/sprintf').sprintf;
 21 
 22 var http = require('util/http');
 23 var terminal = require('util/terminal');
 24 
 25 var config = {
 26   'short_description': 'Print a list of available services',
 27   'long_description': 'Print a list of available services.',
 28   'required_arguments' : [],
 29   'optional_arguments': []
 30 };
 31 
 32 var format_status = function(value) {
 33   if (!value) {
 34     return 'service is disabled';
 35   }
 36 
 37   return sprintf('pid: %s, state: %s', value.pid, value.state);
 38 };
 39 
 40 function handle_command(args) {
 41   http.get_response(undefined, '/services/', 'GET', true, function(error, response) {
 42     if (error) {
 43       error = error;
 44       return sys.puts(sprintf('Error: %s\n%s', error.message, error.stack));
 45     }
 46 
 47     if (response.status_code !== 200) {
 48       return sys.puts(sprintf('HTTP Error, status code: %d, returned body: %s', response.status_code,
 49                                                                                 response.body.message));
 50     }
 51 
 52     var services = response.body;
 53 
 54     sys.puts('Available services: \n');
 55     terminal.print_table([{'title': 'Name', 'value_property': 'name', 'padding_right': 30},
 56                           {'title': 'Enabled', 'value_property': 'enabled', 'padding_right': 20},
 57                           {'title': 'Status', 'value_property': 'status', 'format_function': format_status}],
 58                           services,
 59                           'No services available');
 60   });
 61 }
 62 
 63 exports.config = config;
 64 exports.handle_command = handle_command;
