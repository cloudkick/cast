 18 var sys = require('sys');
 19 var uuid = require('extern/uuid');
 20 
 21 var async = require('extern/async');
 22 var sprintf = require('extern/sprintf').sprintf;
 23 
 24 var ps = require('util/pubsub');
 25 var log = require('util/log');
 26 
 27 var Service = require('services').Service;
 28 var check = require('health');
 29 var http_check = require('health/checks/http');
 30 
 31 var SERVICE_NAME = 'Health';
 32 
 33 /**
 34  * Class representing a single scheduled check.
 35  *
 36  * @param {Check} check Check object.
 37  * @param {Integer} interval Run interval in miliseconds.
 38  *
 39  * @constructor
 40  */
 41 function ScheduledCheck(check, interval) {
 42   this.id = uuid.uuid();
 43   this.check = check;
 44   this.interval = interval || 90000;
 45 
 46   this.timeout_id = null;
 47   this.is_paused = false;
 48   this.is_scheduled = false;
 49   this.last_run_date = null;
 50 }
 51 
 52 /**
 53  * Health service class.
 54 
 55  * @constructor
 56  */
 57 function Health() {
 58   Service.call(this, SERVICE_NAME);
 59 
 60   this.active_checks = {};
 61 }
 62 
 63 sys.inherits(Health, Service);
 64 
 65 /**
 66  * Start the service and re-schedule all the unscheduled (not paused) checks.
 67  */
 68 Health.prototype.start = function() {
 69   Service.prototype.start.call(this);
 70 
 71   this.schedule_checks();
 72 };
 73 
 74 /**
 75  * Stop the service and un-schedule all the non-paused scheduled checks.
 76  */
 77 Health.prototype.stop = function() {
 78   Service.prototype.stop.call(this);
 79 
 80   for (var check_id in this.active_checks) {
 81     if (this.active_checks.hasOwnProperty(check_id)) {
 82       this.unschedule_check(this.active_checks[check_id].id);
 83     }
 84   }
 85 };
 86 
 87 /**
 88  * Add a new check.
 89  *
 90  * @param {ScheduledCheck} scheduled_check Sccheduled check.
 91  */
 92 Health.prototype.add_check = function(scheduled_check) {
 93   var self = this;
 94   var timeout_id;
 95 
 96   this.active_checks[scheduled_check.id] = scheduled_check;
 97   this._set_interval(scheduled_check.id);
 98   this.active_checks[scheduled_check.id].is_scheduled = true;
 99 
100   log.info(sprintf('Check %s added and scheduled to run every %s seconds',
101                    scheduled_check.id, scheduled_check.interval / 1000));
102 };
103 
104 /**
105  * Remove and unschedule a check
106  *
107  * @param {Integer} id Check ID.
108  * @return {Boolean} If check was removed.
109  */
110 Health.prototype.remove_check = function(id) {
111   if (!this._check_exists(id)) {
112     throw new Error(sprintf('Check with id %s not found', id));
113   }
114 
115   if (this.active_checks[id].is_scheduled && this.active_checks[id].timeout_id) {
116     this._clear_interval(id);
117     this.active_checks[id].is_scheduled = false;
118   }
119 
120   if (delete this.active_checks[id]) {
121     log.info(sprintf('Removed check with id %s', id));
122 
123     return true;
124   }
125 
126   return false;
127 };
128 
129 /**
130  * Run a check
131  *
132  * @param {Integer} id Check ID.
133  */
134 Health.prototype.run_check = function(id) {
135   this.active_checks[id].check.run();
136   this.active_checks[id].last_run_date = new Date();
137 };
138 
139 /**
140  *  Schedule all the unscheduled checks.
141  *
142  * @param {Boolean} include_paused True to include the paused checks.
143  */
144 Health.prototype.schedule_checks = function(include_paused) {
145   for (var check_id in this.active_checks) {
146     if (this.active_checks.hasOwnProperty(check_id)) {
147       if (!include_paused && this.active_checks[check_id].is_paused) {
148         continue;
149       }
150 
151       this.schedule_check(check_id);
152     }
153   }
154 };
155 
156 /**
157  *  Unschedule all the scheduled checks.
158  */
159 Health.prototype.unschedule_checks = function() {
160   for (var check_id in this.active_checks) {
161     if (this.active_checks.hasOwnProperty(check_id)) {
162       this.unschedule_check(check_id);
163     }
164   }
165 };
166 
167 /**
168  * Schedule a check.
169  *
170  * @param {Integer} id Check ID.
171  */
172 Health.prototype.schedule_check = function(id) {
173   if (!this.active_checks[id].is_scheduled && !this.active_checks[id].timeout_id) {
174     this._set_interval(id);
175     this.active_checks[id].is_scheduled = true;
176   }
177 };
178 
179 /**
180  * Unschedule a check.
181  *
182  * @param {Integer} id Check ID.
183  */
184 Health.prototype.unschedule_check = function(id) {
185   if (this.active_checks[id].is_scheduled && this.active_checks[id].timeout_id) {
186     this._clear_interval(id);
187     this.active_checks[id].is_scheduled = false;
188   }
189 };
190 
191 /**
192  * Pause a check.
193  *
194  * @param {Integer} id Check ID.
195  */
196 Health.prototype.pause_check = function(id) {
197   if (!this._check_exists(id)) {
198     throw new Error(sprintf('Check with id %s not found', id));
199   }
200 
201   if (this.active_checks[id].is_scheduled && this.active_checks[id].timeout_id && !this.is_paused) {
202     this._clear_interval(id);
203     this.active_checks[id].is_scheduled = false;
204     this.active_checks[id].is_paused = true;
205   }
206 };
207 
208 /**
209  * Resume a check.
210  *
211  * @param {Integer} id Check ID.
212  */
213 Health.prototype.resume_check = function(id) {
214   if (!this._check_exists(id)) {
215     throw new Error(sprintf('Check with id %s not found', id));
216   }
217 
218   if (!this.active_checks[id].is_scheduled && !this.active_checks[id].timeout_id && this.is_paused) {
219     this._set_interval(id);
220     this.active_checks[id].is_scheduled = true;
221     this.active_checks[id].is_paused = false;
222   }
223 };
224 
225 /**
226  * Return all the active checks.
227  *
228  * @return {Array} Array of ScheduledCheck objects.
229  */
230 Health.prototype.get_checks_array = function() {
231   var result = [];
232   for (var check_id in this.active_checks) {
233     if (this.active_checks.hasOwnProperty(check_id)) {
234       result.push(this.active_checks[check_id]);
235     }
236   }
237 
238   return result;
239 };
240 
241 Health.prototype._check_exists = function(id) {
242   if (!this.active_checks.hasOwnProperty(id)) {
243     return false;
244   }
245 
246   return true;
247 };
248 
249 Health.prototype._set_interval = function(id) {
250   var timeout_id;
251   var self = this;
252 
253   timeout_id = setInterval(function() { self.run_check(self.active_checks[id].id); },
254                                          self.active_checks[id].interval);
255   this.active_checks[id].timeout_id = timeout_id;
256 };
257 
258 Health.prototype._clear_interval = function(id) {
259   clearTimeout(this.active_checks[id].timeout_id);
260   this.active_checks[id].timeout_id = null;
261 };

263 var health = new Health();
264 
265 var load = function() {
266   function start_health() {
267     health.start();
268 
269     ps.emit('cast.agent.services.health.started');
270     log.info('health service started');
271   }
272 
273    function stop_health() {
274      health.stop();
275 
276      ps.emit('cast.agent.services.health.stopped');
277      log.info('health service stopped');
278    }
279 
280    ps.on(ps.AGENT_STATE_START, function() {
281      if (!health.is_running) {
282        start_health();
283      }
284    });
285 
286    ps.on(ps.AGENT_STATE_STOP, function() {
287     if (health.is_running) {
288       stop_health();
289     }
290    });
291 
292 };
293 
294 exports.load = load;
295 exports.health = health;
