 18 /*
 19  * Calls a function with the provided arguments and ignores the error passed to called function callback.
 20  *
 21  * @param {Function} func Function which is called.
 22  * @param {Object|null} scope Scope in which the function is called or null.
 23  * @param {...} arg1, arg2, ... argn Optional argument which are passed to the called function.
 24  * @param {Function} callback  A callback which is called with null as the first argument (error) and other
 25  *                             arguments if the called function passes any other argument besides error to
 26  *                             its callback.
 27  */
 28 exports.call_ignoring_error = function(func, scope) {
 29   var arg, call_func, func_callback;
 30   var args = [];
 31   var callback = (arguments.length > 2) ? arguments[arguments.length - 1] : null;
 32 
 33   if (typeof(func) !== 'function') {
 34     throw new Error('First argument must be a function');
 35   }
 36 
 37   if (typeof(scope) !== 'object' && typeof(scope) !== null) {
 38     throw new Error('Second argument (scope) must be an object or null');
 39   }
 40 
 41   if (callback && (typeof(callback) !== 'function')) {
 42     throw new Error('Last argument (callback) must a function');
 43   }
 44 
 45   if (arguments.length > 3) {
 46     for (var i = 2; i < (arguments.length - 1); i++) {
 47       arg = arguments[i];
 48       args.push(arg);
 49     }
 50   }
 51 
 52   func_callback = function() {
 53     if (!callback) {
 54       return;
 55     }
 56 
 57     var args = [null];
 58 
 59     // We assume that the first argument passed to this function is error and ignore it
 60     if (arguments.length > 1) {
 61       for (var i = 1; i < arguments.length; i++) {
 62         arg = arguments[i];
 63         args.push(arg);
 64       }
 65     }
 66 
 67     callback.apply({}, args);
 68   };
 69 
 70   args.push(func_callback);
 71 
 72   func.apply(scope, args);
 73 };
