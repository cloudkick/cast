 18 var sys = require('sys');
 19 var path = require('path');
 20 
 21 var ps = require('util/pubsub');
 22 var config = require('util/config');
 23 var log = require('util/log');
 24 var version = require('util/version');
 25 var req = require('util/requirements');
 26 var CommandParser = require('util/command_parser').CommandParser;
 27 
 28 var requirements = {
 29   'node_version': ['0.3.0', req.compare_versions, 'Minimum node version must be 0.3.0']
 30 };
 31 
 32 /**
 33  * Entry point to the Cast Client Application
 34  */
 35 exports.run = function() {
 36   // Don't show log messages
 37   log.set_loglevel('nothing');
 38 
 39   var p = new CommandParser(path.join(__dirname, 'commands'));
 40 
 41   p.binary = 'cast';
 42   p.banner = 'Usage: cast command [sub-command]';
 43   p.add_commands(['version']);
 44   p.add_commands(['init']);
 45   p.add_commands(['bundles/list', 'bundles/create', 'bundles/validate-manifest', 'bundles/upload']);
 46   p.add_commands(['instances/list', 'instances/create', 'instances/enable']);
 47   p.add_commands(['services/list', 'services/tail']);
 48   p.add_commands(['remotes/add', 'remotes/list', 'remotes/set-default']);
 49 
 50   ps.once(ps.CLIENT_STATE_EXIT, function(args) {
 51     if (args && (args.why !== undefined && args.value !== undefined)) {
 52       sys.puts(args.value);
 53     }
 54   });
 55 
 56   ps.ensure(ps.CLIENT_CONFIG_DONE, function() {
 57     req.meets_requirements(requirements, function(err, meet_requirements) {
 58       if (err || !meet_requirements) {
 59         ps.emit(ps.CLIENT_STATE_EXIT, {'why': 'requirements', 'value': err});
 60         return;
 61       }
 62 
 63       try {
 64         p.parse(process.ARGV);
 65       }
 66       catch (error) {
 67         sys.puts(error.toString());
 68       }
 69     });
 70   });
 71 
 72   config.setup(function(err) {
 73     if (err) {
 74       ps.emit(ps.CLIENT_STATE_EXIT, {'why': 'config', 'value': err});
 75     }
 76     else {
 77       ps.emit(ps.CLIENT_CONFIG_DONE);
 78     }
 79   });
 80 };
