 18 var events = require('events');
 19 
 20 function is_char_boundary(byte) {
 21   return ((byte & 0xC0) !== 0x80);
 22 }
 23 
 24 /**
 25  * A UTF-8 safe ring buffer implemented on top of a Node.js buffer.
 26  *
 27  * @param {Number} size How many bytes to store in the buffer.
 28  * @constructor
 29  */
 30 function RingBuffer(size) {
 31   this.buffer = new Buffer(size);
 32   this.offset = 0;
 33   this.wrapped = false;
 34 }
 35 
 36 RingBuffer.prototype = new events.EventEmitter();
 37 
 38 /**
 39  * Append data to the ring buffer. This (obviously) may overwite existing data
 40  * and if the supplied buffer is larger than the ring buffer itself, earlier
 41  * bytes of the supplied buffer will not be stored.
 42  *
 43  * @param {Buffer} buf  The buffer to store data from.
 44  */
 45 RingBuffer.prototype.append = function(buf) {
 46   process.nextTick(function() {
 47     this.emit('data', buf);
 48   });
 49   var diff;
 50 
 51   if (buf.length >= this.buffer.length) {
 52     diff = buf.length - this.buffer.length;
 53     buf.copy(this.buffer, 0, diff);
 54     this.offset = 0;
 55     this.wrapped = true;
 56     return;
 57   }
 58 
 59   else if (buf.length > (this.buffer.length - this.offset)) {
 60     diff = this.buffer.length - this.offset;
 61     buf.copy(this.buffer, this.offset, 0, diff);
 62     buf.copy(this.buffer, 0, diff);
 63     this.offset = buf.length - diff;
 64     this.wrapped = true;
 65     return;
 66   }
 67 
 68   else {
 69     buf.copy(this.buffer, this.offset, 0);
 70     this.offset += buf.length;
 71     return;
 72   }
 73 };
 74 
 75 /**
 76  * Read the given number of bytes from the end of the ring buffer. This will
 77  * return a buffer containing no more than the requested number of bytes. It
 78  * may contain less than the requested number if either the ring buffer does
 79  * not currently hold that many or if the 'utf8' argument is true in which
 80  * case it will attempt to split along character boundaries. In other words,
 81  * if 'utf8' is set and only valid UTF-8 has been written to the buffer the returned buffer
 82  * will contain a trailing subset of the characters that have been written, but
 83  * not possibly less than the number of bytes requested.
 84  *
 85  * @param {Number} bytes  The maximum number of bytes to retrieve.
 86  * @param {Boolean} utf8  Whether to ensure that the returned buffer contains
 87  *                        only valid UTF-8.
 88  */
 89 RingBuffer.prototype.tail = function(bytes, utf8) {
 90   var _utf8 = utf8 || false;
 91   var toread = (bytes < this.buffer.length) ? bytes : this.buffer.length;
 92   var begin = this.offset - bytes;
 93 
 94   if (begin < 0) {
 95     if (this.wrapped) {
 96       begin = this.buffer.length + begin;
 97     }
 98     if (begin < 0) {
 99       toread += begin;
100       begin = 0;
101     }
102   }
103 
104   if (_utf8) {
105     while (!is_char_boundary(this.buffer[begin]) && toread > 0) {
106       begin++;
107       toread--;
108     }
109   }
110 
111   var buf = new Buffer(toread);
112   var copied = 0;
113 
114   if ((begin + toread) > this.buffer.length) {
115     copied = this.buffer.copy(buf, 0, begin);
116     begin = 0;
117   }
118 
119   this.buffer.copy(buf, copied, begin, begin + (toread - copied));
120 
121   return buf;
122 };
