 34 /**
 35  * Given a bundle name and a file name, return a validated and normalized path
 36  * to a (not necessarily existing) file. Return false on invalid paths.
 37  *
 38  * @param {String} bundle The name of the bundle.
 39  * @param {String} file The name of the file.
 40  *
 41  * @return {String|Boolean} The normalized file path, or false on invalid paths.
 42  */
 43 function file_path(bundle, file) {
 44   if (file.indexOf(bundle) !== 0 || !file.match(/\.tar\.gz$/)) {
 45     return false;
 46   }
 47   var bundleroot = path.join(config.get().bundle_dir);
 48   var bpath = path.normalize(path.join(bundleroot, bundle));
 49   var fpath = path.normalize(path.join(bundleroot, bundle, file));
 50   if ((path.dirname(bpath) === bundleroot) && (path.dirname(fpath) === bpath)) {
 51     return fpath;
 52   }
 53   return false;
 54 }

 56 /**
 57  * Given a bundle name return a validated and normalized path to the (not
 58  * necessarily existing) bundle directory. Return false on invalid paths.
 59  *
 60  * @param {String} bundle The name of the bundle.
 61  *
 62  * @return {String|Boolean} The normalized bundle path, or false on invalid paths.
 63  */
 64 function bundle_path(bundle) {
 65   var bundleroot = path.join(config.get().bundle_dir);
 66   var bpath = path.normalize(path.join(bundleroot, bundle));
 67   if (path.dirname(bpath) === bundleroot) {
 68     return bpath;
 69   }
 70   return false;
 71 }

 73 /**
 74  * Given a bundle tarball, find the path to which it should be extracted.
 75  *
 76  * @param {String} bundle The name of the bundle.
 77  * @param {String} file The name of the file.
 78  * @return {String} Path to extract into.
 79  */
 80 function extract_path(bundle, file) {
 81   var exroot = path.join(config.get().extracted_dir);
 82   var extidx = file.lastIndexOf('.tar.gz');
 83   return path.join(exroot, bundle, file.slice(0, extidx));
 84 }

 86 /**
 87  * Get a temporary file name within the same directory as the provided filename
 88  *
 89  * @param {String} filename The name of the file to create a temporary name for.
 90  * @return {String} A path to a temporary file in the same directory as filename.
 91  */
 92 function tempfileInDirectory(filename) {
 93   var randname;
 94   var dname = path.dirname(filename);
 95   randname = '.cast_tmp_' + misc.randstr(8) + path.extname(filename);
 96   return path.join(dname, randname);
 97 }

 99 /**
100  * Stat a path and verify that it exists. Return a 404 if it doesn't. If it
101  * does, fire the given callback with the stat object.
102  *
103  * @param {String} p  The path to verify.
104  * @param {http.ServerResponse} res The response to potentially send the 404 to.
105  * @param {Function} cb An optional callback which takes the stat object as
106  *                      its only argument.
107  */
108 function path_or_404(p, res, cb) {
109   if (!p) {
110     http.return_error(res, 404, 'File not found');
111     return;
112   }
113 
114   fs.stat(p, function(err, stats) {
115     if (err) {
116       http.return_error(res, 404, 'File not found');
117       return;
118     }
119     cb(stats);
120   });
121 }
122 
123 /**
124  * Stat a path and verify that it is a file which exists. Return a 404 if it
125  * doesn't. If it does, fire the given callback with the stat object.
126  *
127  * @param {String} p  The path to verify.
128  * @param {http.ServerResponse} res The response to potentially send the 404 to.
129  * @param {Function} cb An optional callback which takes the stat object as
130  *                      its only argument.
131  */
132 function file_or_404(p, res, cb) {
133   path_or_404(p, res, function(stats) {
134     if (!stats.isFile()) {
135       http.return_error(res, 404, 'File not found');
136       return;
137     }
138     cb(stats);
139     return;
140   });
141 }

143 /**
144  * Receive an uploaded bundle file and extract it. This attempts to be as asfe
145  * as possible by performing all validation before any destructive actions are
146  * taken against existing data, however failed rmtrees (which aren't all that
147  * robust) can still cause data loss.
148  *
149  * @param {http.ServerRequest} req  The HTTP request to read from.
150  * @param {http.ServerResponse} res The HTTP response to respond on.
151  * @param {String} bundle The name of the bundle to store.
152  * @param {String} file   The name of the file to store.
153  */
154 function upload(req, res, bundle, file) {
155   // We must pause the request until we decide what to do with it
156   req.pause();
157 
158   // Some data will still arrive, so buffer it
159   var received = [];
160   var completed = false;
161 
162   function on_data(data) {
163     received.push(data);
164   }
165   req.on('data', on_data);
166 
167   function on_end() {
168     completed = true;
169   }
170   req.on('end', on_end);
171 
172   // Validate the path
173   var p = file_path(bundle, file);
174   if (!p) {
175     http.return_error(res, '404', 'File not found');
176     return;
177   }
178 
179   // Figure out a bunch of path names
180   var d = path.dirname(p);
181   var extpath = extract_path(bundle, file);
182   var tempf = tempfileInDirectory(p);
183   var tempd = tempfileInDirectory(extpath);
184 
185   async.series([
186     // Verify and if necessary create the bundle directory
187     async.apply(fsutil.ensure_directory, d),
188 
189     // Store the tarball to a temporary file
190     function(callback) {
191       req.removeListener('data', on_data);
192       req.removeListener('end', on_end);
193       req.resume();
194 
195       if (req.headers.expect === '100-continue') {
196         res.writeHead(100);
197       }
198 
199       pumpfile.pumpfilein(tempf, req, received, completed, function(err, sha1) {
200         if (!err && req.headers['x-content-sha1']) {
201           if (req.headers['x-content-sha1'] !== sha1.digest('base64')) {
202             err = new Error('SHA1 Mismatch');
203             err.code = 400;
204           }
205         }
206         return callback(err);
207       });
208     },
209 
210     // Verify and if necessary create the bundle's extract directory
211     async.apply(fsutil.ensure_directory, path.dirname(extpath)),
212 
213     // Extract the tarball to a temporary directory
214     async.apply(extract_tarball, tempf, tempd, 0755),
215 
216     // Validate the manifest
217     function(callback) {
218       manifest.validate_manifest(path.join(tempd, 'cast.json'), function(err, manifest_object) {
219         if (err) {
220           err.code = 400;
221         }
222         callback(err);
223         return;
224       });
225     },
226 
227     // If a tarball already existed, remove it
228     function(callback) {
229       path.exists(p, function(exists) {
230         if (exists) {
231           return fs.unlink(p, callback);
232         }
233         else {
234           callback();
235           return;
236         }
237       });
238     },
239 
240     // Swap the new one into its place
241     async.apply(fs.rename, tempf, p),
242 
243     // If an extracted tree already existed, remove it
244     function(callback) {
245       path.exists(extpath, function(exists) {
246         if (exists) {
247           return fsutil.rmtree(extpath, callback);
248         }
249         else {
250           callback();
251           return;
252         }
253       });
254     },
255 
256     // Swap the new tree into its place
257     async.apply(fs.rename, tempd, extpath)
258   ],
259 
260   // Respond
261   function(err) {
262     if (err) {
263       log.err('Bundle upload error: ' + err.message);
264       var code = err.code || 500;
265       http.return_error(res, code, err.message);
266 
267       // Attempt to clean up after ourselves
268       async.forEach([tempf, tempd], function(itempath, callback) {
269         path.exists(itempath, function(exists) {
270           if (!exists) {
271             return callback();
272           }
273           fsutil.rmtree(itempath, function(err) {
274             if (err) {
275               log.err('Upload cleanup error: ' + err.message);
276             }
277             callback();
278           });
279         });
280       },
281       function(err) {
282         log.info('Completed upload cleanup');
283       });
284     }
285     else {
286       log.info('Successful upload of ' + file);
287       res.writeHead(204, {});
288       res.end();
289     }
290   });
291 }

293 /**
294  * Send a requested bundle file
295  *
296  * @param {http.ServerRequest} req  The HTTP request to respond to.
297  * @param {http.ServerResponse} res The HTTP response to write to.
298  * @param {String} bundle The name of the bundle to read.
299  * @param {String} file   The name of the file to read.
300  */
301 function download(req, res, bundle, file) {
302   var fpath = file_path(bundle, file);
303   file_or_404(fpath, res, function(stats) {
304     res.writeHead(200, {
305       'Content-Type': 'application/octet-stream',
306       'Content-Length': stats.size
307     });
308    pumpfile.pumpfileout(fpath, res, function(err) {
309       if (err) {
310         // The header is already written, there isn't much we can do
311         log.err('Error streaming file: ' + err);
312       }
313       // End the response in any case
314       res.end();
315     });
316   });
317 }

319 /**
320  * Remove a bundle file
321  *
322  * @param {http.ServerRequest} req  The HTTP request to respond to.
323  * @param {http.ServerResponse} res The HTTP response to respond to.
324  * @param {String} bundle The name of the bundle to delete from.
325  * @param {String} file   The name of the file to delete.
326  */
327 function remove(req, res, bundle, file) {
328   var fpath = file_path(bundle, file);
329   file_or_404(fpath, res, function() {
330     fs.unlink(fpath, function(err) {
331       if (err) {
332         http.return_error(res, 500, 'Error removing bundle file: ' + err.message);
333         return;
334       }
335       res.writeHead(204, {});
336       res.end();
337     });
338   });
339 }

341 /**
342  * List files within a bundle
343  *
344  * @param {http.ServerRequest} req  The HTTP request to respond to.
345  * @param {http.ServerResponse} res The HTTP response to respond to.
346  * @param {String} bundle The name of the bundle to delete from.
347  */
348 function list_files(req, res, bundle) {
349   var bpath = bundle_path(bundle);
350   if (!bpath) {
351     http.return_error(res, 404, 'File not found');
352     return;
353   }
354   fs.readdir(bpath, function(err, files) {
355     if (err) {
356       http.return_error(res, 404, 'File not found');
357       return;
358     }
359 
360     var filedata = [];
361 
362     // Build a list of files in this directory
363     async.forEach(files, function(file, callback) {
364       var fpath = file_path(bundle, file);
365 
366       if (!fpath) {
367         callback();
368         return;
369       }
370 
371       fs.stat(fpath, function(err, stats) {
372         if (err) {
373           log.warning('Error stat-ing file: ' + err.message);
374         }
375 
376         else if (stats.isFile()) {
377           // TODO: Do we want anything besides the name of the file?
378           filedata.push(file);
379         }
380         callback();
381         return;
382       });
383     },
384 
385     // When all of the files have been checked, return the list
386     function(err) {
387       http.return_json(res, 200, filedata);
388     });
389   });
390 }

392 /**
393  * List available bundles
394  *
395  * @param {http.ServerRequest} req  The HTTP request to respond to.
396  * @param {http.ServerResponse} res The HTTP response to respond to.
397  */
398 function list_bundles(req, res) {
399   var path_extracted = config.get().extracted_dir;
400 
401   var get_bundle_versions = function(bundle_name, callback) {
402     var bundle_path, splitted;
403     var bundle_versions = [];
404 
405     bundle_path = path.join(path_extracted, bundle_name);
406 
407     fs.readdir(bundle_path, function(err, files) {
408       if (err) {
409         callback(err);
410         return;
411       }
412 
413 
414       async.forEach(files, function(file, callback) {
415         bundle_path = path.join(path_extracted, bundle_name);
416 
417         fs.stat(bundle_path, function(err, stats) {
418           if (stats.isDirectory()) {
419             // @TODO: Don't hard-code the version delimiter and store it in util/misc.js?
420             splitted = file.split('@');
421 
422             if (splitted.length !== 2) {
423               callback();
424               return;
425             }
426 
427             bundle_versions.push({
428               'name': bundle_name,
429               'version': splitted[1]
430             });
431 
432             callback();
433             return;
434           }
435 
436           callback();
437         });
438       },
439 
440       function(err) {
441         if (err) {
442           callback(err);
443           return;
444         }
445 
446         callback(null, bundle_versions);
447       });
448     });
449   };
450 
451   fs.readdir(path_extracted, function(err, files) {
452     if (err) {
453       // This should NOT happen in the course of normal operations
454       log.err('Error reading bundle directory: ' + err.message);
455       http.return_error(res, 500, 'Error reading directory');
456       return;
457     }
458 
459     var bundles = [];
460 
461     // Build a list of directories in this directory
462     async.forEach(files, function(file, callback) {
463       var fpath = path.join(path_extracted, file);
464 
465       fs.stat(fpath, function(err, stats) {
466         if (err) {
467           log.warning('Error stat-ing file: ' + err.message);
468         }
469         else if (stats.isDirectory()) {
470           get_bundle_versions(file, function(err, bundle_versions) {
471             if (err) {
472               // This bundle has no versions so it's not included in the result array.
473               callback();
474               return;
475             }
476 
477             bundles = bundles.concat(bundle_versions);
478             callback();
479             return;
480           });
481         }
482         else {
483           callback();
484         }
485       });
486     },
487 
488     // Return the list of bundles
489     function(err) {
490       http.return_json(res, 200, bundles);
491     });
492   });
493 }
