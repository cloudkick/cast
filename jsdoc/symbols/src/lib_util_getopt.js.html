 18 var optparse = require('extern/optparse/lib/optparse');
 19 var log = require('util/log');
 20 var sys = require('sys');
 21 var version = require('util/version');
 22 var defaults = [
 23   ['-h', '--help', 'Print this help'],
 24   ['-V', '--version', 'Print the version']
 25 ];
 26 
 27 var switches = [];
 28 
 29 exports.add = function(arr) {
 30   switches = switches.concat(arr);
 31 };
 32 
 33 function halt(parser) {
 34   parser.halt();
 35   process.nextTick(function() {
 36     process.exit(0);
 37   });
 38 }
 39 
 40 exports.parser = function() {
 41   var parser = new optparse.OptionParser(switches);
 42   parser.on('help', function() {
 43     sys.puts(parser.toString());
 44     halt(parser);
 45   });
 46   parser.on('version', function() {
 47     sys.puts(version.toString());
 48     halt(parser);
 49   });
 50   parser.on(function(opt) {
 51       sys.puts('No handler was defined for option: ' + opt);
 52       halt(parser);
 53   });
 54   parser.on('*', function(opt, value) {
 55       sys.puts('wild handler for ' + opt + ', value=' + value);
 56   });
 57   return parser;
 58 };
 59 
 60 (function() {
 61   exports.add(defaults);
 62 })();
