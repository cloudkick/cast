 37 function create_instance(req, res, bundle_name_full) {
 38   var application_instance_path, instance_name, instance_number, instance_path, manifest_object, ignored_paths;
 39   var application_data_path, instance_data_path;
 40 
 41   // @TODO: Don't hard-code the version delimiter and store it in util/misc.js?
 42   var splitted = bundle_name_full.split('@');
 43 
 44   var bundle_name = splitted[0];
 45   var bundle_version = splitted[1];
 46 
 47   var extracted_bundle_path = path.join(config.get().extracted_dir, bundle_name, bundle_name_full);
 48   var manifest_path = path.join(extracted_bundle_path, manifest_constants.MANIFEST_FILENAME);
 49 
 50   application_instance_path = path.join(config.get().app_dir, bundle_name);
 51 
 52   async.series([
 53     // Make sure that the extracted bundle path exists
 54     function(callback) {
 55       path.exists(extracted_bundle_path, function(exists) {
 56         if (!exists) {
 57           return callback(new Error('Invalid bundle name'));
 58         }
 59 
 60         callback();
 61       });
 62     },
 63 
 64     // Retrieve the manifest data object
 65     function(callback) {
 66       manifest.get_manifest_data_as_object(manifest_path, true, function(error, manifest_object_) {
 67         if (error) {
 68           return callback(error);
 69         }
 70 
 71         manifest_object = manifest_object_;
 72         callback();
 73       });
 74     },
 75 
 76     // Create the application directory (if it does not already exist)
 77     function(callback) {
 78       fsutil.ensure_directory(application_instance_path, callback);
 79     },
 80 
 81     // Retrieve the instance number
 82     function(callback) {
 83       deployment.get_available_instances(bundle_name, bundle_version, function(error, instances) {
 84         if (error) {
 85           if (error.errno === process.ENOENT) {
 86             // Directory for this application does not yet exist
 87             instances = [];
 88           }
 89           else {
 90             return callback(error);
 91           }
 92         }
 93 
 94         var instances_count = instances.length;
 95         if (instances_count === 0) {
 96           // This is the first instance
 97           instance_number = 0;
 98         }
 99         else {
100           instance_number = instances[instances_count - 1][1] + 1;
101         }
102 
103         instance_name = misc.get_valid_instance_name(bundle_name_full, instance_number);
104         instance_path = path.join(application_instance_path, instance_name);
105 
106         application_data_path = path.join(config.get().data_dir, bundle_name);
107         instance_data_path = path.join(application_data_path, instance_name);
108 
109         ignored_paths = manifest_object.template_files.concat(manifest_object.data_files);
110 
111         callback();
112       });
113     },
114 
115     // Create the application data directory (if it does not already exist)
116     function(callback) {
117       fsutil.ensure_directory(application_data_path, callback);
118     },
119 
120     // Create the instance data directory (if it does not already exist)
121     function(callback) {
122       fsutil.ensure_directory(instance_data_path, callback);
123     },
124 
125     // 1. Create the instance directory
126     function(callback) {
127       fsutil.ensure_directory(instance_path, callback);
128     },
129 
130     // 2. Mirror the directory structure from the extracted bundle in the instance directory
131     function(callback) {
132       fsutil.get_directory_tree_as_template_object(extracted_bundle_path, ignored_paths, function(error, template_object) {
133         if (error) {
134           return callback(error);
135         }
136 
137         // No directories
138         if (Object.keys(template_object).length === 0) {
139           return callback();
140         }
141 
142         misc.template_to_tree(instance_path, template_object, true, function(error) {
143           if (error) {
144             return callback(error);
145           }
146 
147           callback();
148         });
149       });
150     },
151 
152     // 3. Realize (render and save) templates
153     function(callback) {
154       var instance_data = templates.get_instance_template_object(instance_name, instance_number, instance_path);
155 
156       deployment.realize_application_templates(manifest_object, instance_data, extracted_bundle_path, instance_path,
157                                               function(error) {
158         if (error) {
159           return callback(error);
160         }
161 
162         callback();
163       });
164     },
165 
166     // 4. Hard-link all the files (except template and data files)
167     function(callback) {
168       fsutil.hard_link_files(extracted_bundle_path, instance_path, ignored_paths, function(error) {
169         if (error) {
170           return callback(error);
171         }
172 
173         callback();
174       });
175     },
176 
177     // 5. Resolve the data files
178     function(callback) {
179       deployment.resolve_data_files(extracted_bundle_path, instance_data_path, instance_path,
180                                     manifest_object.data_files, callback);
181     },
182 
183     // 6. Create a runit service
184     function(callback) {
185       deployment.create_service(instance_name, instance_path, manifest_object.entry_file, manifest_object.type,
186                                 function(error) {
187         if (error) {
188           return callback(error);
189         }
190 
191         callback();
192       });
193     }
194   ],
195 
196   function(error) {
197     if (error) {
198       if (!instance_name) {
199         // Deployment failed before any non-idempotent action has been performed so no cleanup
200         // is needed
201         http.return_error(res, 500, error.message);
202         return;
203       }
204 
205       deployment.cleanup(instance_name, instance_path, instance_data_path, function(err) {
206         http.return_error(res, 500, error.message);
207         return;
208       });
209 
210       return;
211     }
212 
213     return http.return_json(res, 200, {
214       'result': 'success',
215       'instance_number': instance_number
216     });
217   });
218 }

220 /**
221  * Enable and start an instance.
222  */
223 function enable_instance(req, res, bundle_name_full, instance_number) {
224   var return_array, instance_name, instance_path;
225 
226   return_array = misc.get_instance_name_and_path(bundle_name_full, instance_number);
227   instance_name = return_array[0];
228   instance_path = return_array[1];
229 
230   path.exists(instance_path, function(exists) {
231     if (!exists) {
232       return http.return_error(res, 500, 'Invalid bundle name or instance number');
233     }
234 
235     var manager = service_management.get_default_manager().get_manager();
236 
237     async.waterfall([
238       function(callback) {
239         manager.get_service(instance_name, function(error, service) {
240           if (error) {
241             return callback(error);
242           }
243 
244           callback(null, service);
245         });
246       },
247 
248       function(service, callback) {
249         service.is_enabled(function(enabled) {
250           if (enabled) {
251             return callback(new Error('This instance is already enabled'));
252           }
253 
254           callback(null, service);
255         });
256       },
257 
258       function(service, callback) {
259         service.enable(function(error) {
260           if (error) {
261             return callback(error);
262           }
263 
264           callback(null, service);
265         });
266       },
267 
268       function(service, callback) {
269         // Remove the down file so the service is started automatically.
270         var down_path = path.join(service.path_enabled, 'down');
271 
272         fs.unlink(down_path, function(err) {
273           callback(null, service);
274         });
275       },
276 
277       function(service, callback) {
278         // Remove the down file so service is started automatically.
279 
280         // Wait 6 seconds before starting a service (runit scans directories for changes every 5 seconds so the control
281         // file is not available instantly after we enable a service)
282         setTimeout(function() {
283           service.start(function(error) {});
284         }, 6000);
285 
286         callback();
287       }],
288 
289       function(error) {
290         if (error) {
291           return http.return_error(res, 500, error.message);
292         }
293 
294         return http.return_json(res, 200, {'result': 'success'});
295       }
296     );
297   });
298 }

300 /*
301  * Disable and stop the instance (if started).
302  */
303 function disable_instance(req, res, bundle_name_full, instance_number) {
304   var return_array, instance_name, instance_path;
305 
306   return_array = misc.get_instance_name_and_path(bundle_name_full, instance_number);
307   instance_name = return_array[0];
308   instance_path = return_array[1];
309 
310   path.exists(instance_path, function(exists) {
311     if (!exists) {
312       return http.return_error(res, 500, 'Invalid bundle name or instance number');
313     }
314 
315     var manager = service_management.get_default_manager().get_manager();
316 
317     async.waterfall([
318       function(callback) {
319         manager.get_service(instance_name, function(error, service) {
320           if (error) {
321             return callback(error);
322           }
323 
324           callback(null, service);
325         });
326       },
327 
328       function(service, callback) {
329         service.is_enabled(function(enabled) {
330           if (!enabled) {
331             return callback(new Error('This instance is already disabled'));
332           }
333 
334           callback(null, service);
335         });
336       },
337 
338       function(service, callback) {
339         service.disable(function(err) {
340           callback(err);
341         });
342       }],
343 
344       function(err) {
345         if (err) {
346           return http.return_error(res, 500, err.message);
347         }
348 
349         return http.return_json(res, 200, {'result': 'success'});
350       }
351     );
352   });
353 }

355 /*
356  * Delete everything related to an instance (including data folder and service folder which means that
357  * the instance will be stopped if it's running).
358  */
359 function delete_instance(req, res, bundle_name_full, instance_number) {
360   var return_array, instance_name, instance_path;
361 
362   return_array = misc.get_instance_name_and_path(bundle_name_full, instance_number);
363   instance_name = return_array[0];
364   instance_path = return_array[1];
365 
366   instances.instance_exists(bundle_name_full, instance_number, function(err, exists) {
367     var bundle_name, application_data_path, instance_data_path;
368 
369     if (err) {
370       http.return_error(res, 500, err.message);
371       return;
372     }
373     else if (!exists) {
374       http.return_error(res, 404, 'Instance does not exist');
375       return;
376     }
377 
378     bundle_name = misc.get_bundle_name(bundle_name_full);
379 
380     application_data_path = path.join(config.get().data_dir, bundle_name);
381     instance_data_path = path.join(application_data_path, instance_name);
382 
383     deployment.cleanup(instance_name, instance_path, instance_data_path, function() {
384       http.return_json(res, 200, {'result': 'success'});
385       return;
386     });
387   });
388 }

390 /**
391  * List all the available instances for the provided bundle version.
392  *
393  * @param {String} bundle_name Bundle name as returned by the get_valid_bundle_name function.
394  * @param {String} bundle_version Bundle version.
395  */
396 function list_bundle_version_instances(req, res, bundle_name, bundle_version) {
397   deployment.get_available_instances(bundle_name, bundle_version, function(error, instances) {
398     if (error) {
399       return http.return_error(res, 500, error.message);
400     }
401 
402     var result_instances = [];
403     var item = {};
404     var splitted;
405 
406     async.forEach(instances, function(instance, callback) {
407       // @TODO: Don't hard-code the version delimiter and store it in util/misc.js?
408       splitted = instance[0].split('@');
409 
410       item = {
411         'instance_name': misc.get_valid_instance_name(instance[0], instance[1]),
412         'instance_number': instance[1],
413 
414         'bundle_name': splitted[0],
415         'bundle_name_full': misc.get_full_bundle_name(splitted[0], splitted[1]),
416         'bundle_version': splitted[1]
417       };
418 
419       result_instances.push(item);
420 
421       callback();
422     },
423 
424     function(error) {
425       if (error) {
426         return http.return_error(res, 500, error.message);
427       }
428 
429       http.return_json(res, 200, result_instances);
430     });
431   });
432 }

434 /**
435  * List all the available instances for the provided bundle.
436  *
437  * @param {String} bundle_name Bundle name as returned by the get_valid_bundle_name function.
438  */
439 function list_all_bundle_instances(req, res, bundle_name) {
