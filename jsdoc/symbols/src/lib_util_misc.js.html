 30 function Errorf(message) {
 31   this.name = 'Error';
 32   this.message = sprintf.apply(message, arguments) || '';
 33 
 34   Error.captureStackTrace(this, arguments.callee);
 35 }
 36 
 37 Errorf.prototype.toString = function() {
 38   return sprintf('%s: %s', this.name, this.message);
 39 };

 43 /**
 44  * Very simple object merging.
 45  * Merges two objects together, returning a new object containing a
 46  * superset of all attributes.  Attributes in b are prefered if both
 47  * objects have identical keys.
 48  *
 49  * @param {Object} a Object to merge.
 50  * @param {Object} b Object to merge, wins on conflict.
 51  * @return {Object} The merged object.
 52  */
 53 exports.merge = function(a, b)
 54 {
 55   var c = {};
 56   var attrname;
 57   for (attrname in a) {
 58     if (a.hasOwnProperty(attrname)) {
 59       c[attrname] = a[attrname];
 60     }
 61   }
 62   for (attrname in b) {
 63     if (b.hasOwnProperty(attrname)) {
 64       c[attrname] = b[attrname];
 65     }
 66   }
 67   return c;
 68 };

 70 /**
 71  * Expands a path with a ~user or ~ to an absolute path.
 72  * Note: This currently depends on the HOME enviroment variable,
 73  * because there are no bindings to getpwnam.
 74  *
 75  * @param {String} pstr Original path.
 76  * @return {string} Expanded path.
 77  */
 78 exports.expanduser = function(pstr)
 79 {
 80   if (pstr.indexOf('~') !== 0) {
 81     return pstr;
 82   }
 83 
 84   var splitter = pstr.indexOf('/');
 85   var user = '';
 86   if (splitter === -1) {
 87     user = pstr.substr(1);
 88     pstr = '/';
 89   }
 90   else {
 91     user = pstr.substring(1, splitter);
 92     pstr = pstr.substr(splitter + 1);
 93   }
 94 
 95   if (user.length !== 0) {
 96     /* TODO: need bindings to getpwnam to make this work for the non-current */
 97     throw new Error("no bindings to getpwnam, so you can't do get info for user: " + user);
 98   }
 99 
100   if (process.env.HOME === undefined) {
101     throw new Error('no bindings to getpwnam, and env[HOME] was undefined');
102   }
103 
104   return path.join(process.env.HOME, pstr);
105 };

138 /**
139  * Return a full bundle name.

143  * @return {String} Full bundle name.

149 /*
150  * Extract bundle name from the full bundle name which includes the version number.
151  *
152  * @param {String} full_bundle_name Bundle name as returned by the get_full_bundle_name function.
153  * @return {String} Bundle name.
154  */
155 exports.get_bundle_name = function(full_bundle_name) {
156   var splitted, bundle_name;
157 
158   splitted = full_bundle_name.split('@');
159 
160   if (splitted.length !== 2) {
161     throw new exports.Errorf('Could not extract bundle name from the provided string: %s', full_bundle_name);
162   }
163 
164   bundle_name = splitted[0];
165   return bundle_name;
166 };

179 /*
180  * Return instance name and path.
181  *
182  * @param {String} full_bundle_name Bundle name which is returned by the get_full_bundle_name function.

184  * @return {Array} First item is the instance name and the second item is instance path.

197 /**
198  * Converts an object to a directory tree, where objects are treated as
199  * directories and anything else is dumped to a regular file as a string.
200  *
201  * @param {String} basedir  The path to the tree's (not-yet-existing) root.
202  * @param {Object} template  The object to use as a template.
203  * @param {Boolean} ignore_existing If true, function won't return callback with an error when encountering
204  *                                  a directory which already exists.
205  * @param {Function} cb     Callback, taking a possible error.
206  */
207 exports.template_to_tree = function(basedir, template, ignore_existing, cb) {
208   fs.mkdir(basedir, 0700, function(err) {
209     if (err) {
210       if (!(ignore_existing && err.errno === constants.EEXIST)) {
211         return cb(err);
212       }
213     }
214 
215     var actions = [];
216 
217     function recurse_action(basedir, key) {
218       var curpath = path.join(basedir, key);
219       return function(callback) {
220         // Recurse on sub-templates
221         if (typeof(template[key]) === 'object') {
222           exports.template_to_tree(curpath, template[key], ignore_existing, callback);
223         }
224         // Render anything else to files
225         else {
226           fs.open(curpath, 'w', 0700, function(err, fd) {
227             if (err) {
228               callback(err);
229             }
230             else {
231               fs.write(fd, template[key].toString(), null, 'ascii', callback);
232               fs.close(fd);
233             }
234           });
235         }
236       };
237     }
238 
239     // Build a list of actions to fill in this level of the tree
240     for (var key in template) {
241       if (template.hasOwnProperty(key)) {
242         actions.push(recurse_action(basedir, key));
243       }
244     }
245 
246     // Execute the action sequence
247     async.parallel(actions, function(err) {
248       cb(err);
249     });
250   });
251 };

253 /**
254  * Generate a random string of upper lower case letters and decimal digits.
255  *
256  * @param {Number} len  The length of the string to return;.
257  * @return {String} Random string.
258  */
259 exports.randstr = function(len) {
260   var chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
261 
262   function getRandomInt(min, max)
263   {
264     return Math.floor(Math.random() * (max - min + 1)) + min;
265   }
266 
267   var r = [];
268   for (var x = 0; x < len; x++) {
269     r.push(chars[getRandomInt(0, chars.length)]);
270   }
271 
272   return r.join('');
273 };

275 /**
276  * Check if a haystack array contains a needle.
277  *
278  * @param  {Object} needle Object to search for.
279  * @param  {Array} haystack Array to search.
280  * @param  {Boolean} strict true to use the strict comparison.
281  * @param {Function} compare_function Optional compare function which is called with the current array
282  *                                    item as the first argument and the needle as the second one and
283  *                                    should return true if the current item matches the needle.
284  *
285  * @return {Boolean} true if the needle is located in the array, false otherwise.
286  */
287 exports.in_array = function(needle, haystack, strict, compare_function) {
288   return (exports.array_find(needle, haystack, strict, compare_function) !== false);
289 };
290 
291 /**
292  * Same as the in_array but return the index instead of true if the item is found in the array.
293  *
294  * @param  {Object} needle Item to search for.
295  * @param  {Array} haystack Array to search in for the item.
296  * @param  {Boolean} strict true to use the strict comparison (order of the item in the array matters).
297  * @param {Function} compare_function Optional compare function which is called with the current array
298  *                                    item as the first argument and the needle as the second one and
299  *                                    should return true if the current item matches the needle.
300  *
301  * @return {Number|Boolean} item index if the needle is located in the array, false otherwise.
302  */
303 exports.array_find = function(needle, haystack, strict, compare_function) {
304   var compare = compare_function || null;
305 
306   function compare_strict(item1, item2) {
307       return (item1 === item2);
308   }
309 
310   function compare_loose(item1, item2) {
311       return (item1 === item2);
312   }
313 
314   if (!compare_function) {
315     if (strict === true) {
316       compare = compare_strict;
317     }
318     else {
319       compare = compare_loose;
320     }
321   }
322 
323   for (var i = 0; i < haystack.length; i++) {
324     if (compare(haystack[i], needle)) {
325       return i;
326     }
327   }
328 
329   return false;
330 };

332 /**
333  * Checks if two arrays contain the same elements.
334  *
335  * @param  {Array} array1 First array.
336  * @param  {Array} array2 Second array.
337  * @param  {Boolean} strict true to use the strict comparison (order of the item in the array matters).
338  *
339  * @return {Boolean} true if the arrays contain same items, false otherwise.
340  */
341 exports.arrays_contains_same_elements = function(array1, array2, strict) {
342   var compare;
343 
344   if (array1.length !== array2.length) {
345     return false;
346   }
347 
348   function compare_strict(index) {
349     return (array1[index] === array2[index]);
350   }
351 
352   function compare_loose(index) {
353     return (exports.in_array(array1[index], array2));
354   }
355 
356   if (strict) {
357     compare = compare_strict;
358   }
359   else {
360     compare = compare_loose;
361   }
362 
363   for (var i = 0; i < array1.length; i++) {
364     if (!compare(i)) {
365       return false;
366     }
367   }
368 
369   return true;
370 };

372 /**
373  * Checks if the array1 is subset of the array2.
374  *
375  * @param  {Array} array1 Smaller array (needle).
376  * @param  {Array} array2 Superset array.
377  * @param  {Boolean} strict true if the array1 must be a proper subset of array2.
378  *
379  * @return {Boolean} true if the array1 if subset of the array2, false otherwise.
380  */
381 exports.array_is_subset_of = function(array1, array2, strict) {
382   if (array1.length > array2.length) {
383     return false;
384   }
385 
386   if (strict) {
387     if (array1.length === array2.length) {
388       return false;
389     }
390   }
391 
392   for (var i = 0; i < array1.length; i++) {
393     if (!exports.in_array(array1[i], array2)) {
394       return false;
395     }
396   }
397 
398   return true;
399 };

401 /**
402  * Return list of items which are in array1 but not in array2.
403  *
404  * @param  {Array} array1 Array to find items in.
405  * @param  {Array} array2 Array to test against.
406  *
407  * @return {Array} List of items which are in array1 but not in array2.
408  */
409 exports.array_difference = function(array1, array2) {
410   var difference = [];
411 
412   for (var i = 0; i < array1.length; i++) {
413     if (!(exports.in_array(array1[i], array2))) {
414       difference.push(array1[i]);
415     }
416   }
417 
418   return difference;
419 };

421 /**
422  * Remove repeated paths.
423  *
424  * @param {Array} paths Array of relative or absolute paths.
425  * @return {Array} Array containing filtered paths sorted in alphabetical order.
426  */
427 exports.filter_repeated_paths = function(paths) {
428   var path;
429   var paths_filtered = [];
430 
431   var compare_function = function(item, needle) {
432     if (item.charAt(item.length - 1) === '/' && needle.indexOf(item) !== -1) {
433       return true;
434     }
435     else if (item === needle) {
436       return true;
437     }
438 
439     return false;
440   };
441 
442   paths.sort();
443   var paths_count = paths.length;
444   for (var i = 0; i < paths_count; i++) {
445     path = paths[i];
446 
447     if (!exports.in_array(path, paths_filtered, null, compare_function)) {
448       paths_filtered.push(path);
449     }
450   }
451 
452   return paths_filtered;
453 };

468 /**
469  * Left pad the given string to the maximum width provided.
470  *
471  * @param  {String} str Input string.
472  * @param  {Number} width Desired length.
473  * @return {String} Left padded string.
474  */
475 exports.lpad = function(str, width) {
476     str = String(str);
477     var n = width - str.length;
478 
479     if (n < 1) {
480       return str;
481 
482     }
483 
484     while (n--) {
485       str = ' ' + str;
486     }
487 
488     return str;
489 };
