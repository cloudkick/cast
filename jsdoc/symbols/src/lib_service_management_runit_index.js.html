 43 /**
 44  * RunitService represents a runit service _control directory (as passed to
 45  * runsv).  These are lazily constructed, so its possible that a RunitService's
 46  * doesn't corresponding directory doesn't actually exist, and you won't know
 47  * until you attempt to read the details, etc.
 48  *
 49  * @param {String} basedir  The directory in which the service resides.
 50  * @param {String} name     The name of the service, corresponds to the name of
 51  *                          the service's _control directory.
 52  * @this
 53  */
 54 function RunitService(path_available, path_enabled, name) {
 55   SupervisedService.call(this, path_available, path_enabled, name);
 56 
 57   this.path_available = path.join(this._base_path_available, this.name);
 58   this.path_enabled = path.join(this._base_path_enabled, this.name);
 59 }
 60 
 61 sys.inherits(RunitService, SupervisedService);

 72 /**
 73  * Retrieves the details of a service. Potential errors and the details object
 74  * are passed to a required callback
 75  *
 76  * @param {Function} cb A required callback taking (err, details).
 77  */
 78 RunitService.prototype.get_details = function(cb) {
 79   var self = this;
 80   var status;
 81   var details = {};
 82   var down_path = path.join(this.path_available, 'down');
 83 
 84   path.exists(down_path, function(exists) {
 85     details.name = self.name;
 86 
 87     self.is_enabled(function(enabled) {
 88       details.enabled = enabled;
 89 
 90       if (enabled) {
 91         self._get_status(function(err, sstatus) {
 92           // If we were unable to retrieve the status, set the status to false
 93           status = err ? false : sstatus;
 94           details = self._format_details(self.name, enabled, status);
 95 
 96           cb(null, details);
 97           return;
 98         });
 99       }
100       else {
101         details = self._format_details(self.name, enabled, null);
102 
103         cb(null, details);
104         return;
105       }
106     });
107   });
108 };

120 /**
121  * Get status information about a service
122  *
123  * @param {Function} cb A callback that takes a possible error and
124  *                      a status object.
125  */
126 RunitService.prototype._get_status = function(cb) {
127   var sstatus = {};
128   var sstatus_path = path.join(this.path_available, 'supervise', 'status');
129   fs.open(sstatus_path, 'r', 0644, function(err, fd) {
130     if (err) {
131       return cb(err, sstatus);
132     }
133     /**
134      * The 'status' file binary format is:
135      *   Start time is stored as:
136      *     Seconds in a uint64_t in indices 0 - 7
137      *     Nanoseconds in an unsigned long in indices 8 - 11
138      *   PID packed in indices 12 - 15
139      *   Paused is 0|1 at index 16
140      *   Want is u|d at index 17
141      *   TERM is 0|1 at index 18
142      *   State is (0|1|2) = (down|run|finish) at index 19
143      */
144     var buf = new Buffer(20);
145     fs.read(fd, buf, 0, buf.length, null, function(err, bytesRead) {
146       if (err) {
147         return cb(err, sstatus);
148       }
149       var time = jspack.Unpack('>L', buf, 4)[0];
150       var pid = jspack.Unpack('<L', buf, 12)[0];
151       var paused = buf[16];
152       var want = buf[17];
153       var term = buf[18];
154       var state = buf[19];
155       sstatus.time = jspack.Unpack('>L>L', buf, 0)[1];
156       sstatus.pid = (pid !== 0) ? pid : -1;
157       sstatus.paused = paused ? true : false;
158       sstatus.want = {
159         'u': 'up',
160         'd': 'down'
161       }[want];
162       sstatus.term = term ? true : false;
163       sstatus.state = {
164         0: 'down',
165         1: 'run',
166         2: 'finish'
167       }[state];
168       return cb(err, sstatus);
169     });
170   });
171 };

173 /**
174  * Format the service details.
175  *
176  * @param {String} name Service name.
177  * @param {Boolean} enabled true if the service is enabled, false otherwise.
178  * @param {Object} status Object as returned by the _get_status method method.
179  * @returned {Object} Object with the following keys: name, time, pid, state.
180  *                    For more info, see {@link SupervisedService.get_details}
181  */
182 RunitService.prototype._format_details = function(name, enabled, status) {
183   var details_formatted, state;
184 
185   details_formatted = {
186     'name': name,
187     'enabled': enabled
188   };
189 
190   if (status) {
191     details_formatted.status = {
192       'time': status.time,
193       'pid': status.pid,
194       'state': (status.state === 'up') ? 'running' : 'down'
195     };
196   }
197   else {
198     details_formatted.status = false;
199   }
200 
201   return details_formatted;
202 };

204 /**
205  * Pipe characters into the runsv control pipe to control the service. For a
206  * list of available control characters and their corresponding signals, see
207  * the runsv man page.
208  *
209  * @param {String} cchar The control character to write to the pipe.
210  * @param {Function} cb A callback called with a possible error.
211  */
212 RunitService.prototype._control = function(cchar, cb) {
213   var self = this;
214   this.is_enabled(function(enabled) {
215     if (!enabled) {
216       return cb(new Error('Service disabled'));
217     }
218     var cpath = path.join(self.path_available, 'supervise', '_control');
219     // Node's syntax is a little annoying on this, we have to specify a mode
220     // in order to provide a callback
221     fs.open(cpath, constants.O_WRONLY, 0700, function(err, fd) {
222       if (err) {
223         return cb(new Error('control file not yet created'));
224       }
225       var buf = new Buffer(cchar, 'ascii');
226       fs.write(fd, buf, 0, buf.length, null, function(err, writen) {
227         return cb(err);
228       });
229     });
230   });
231 };
232 
233 /**
234  * Restart the service. The callback fires when the _control characters have
235  * been written, without regard to what actually happens to the service. This
236  * sends a TERM followed by a CONT, then has runsv start the service and set it
237  * to up.
238  *
239  * @param {Function} cb A callback called with a possible error.
240  */
241 RunitService.prototype.restart = function(cb) {
242   this._control('tcu', cb);
243 };

263 /**
264  * Enable a service by symlinking to it from the enabled directory
265  *
266  * @param {Function} cb A callback with a possible error.
267  */
268 RunitService.prototype.enable = function(cb) {
269   var self = this;
270   self.is_enabled(function(enabled) {
271     if (!enabled) {
272       var target;
273       if (self.path_available[0] === '/') {
274         target = self.path_available;
275       }
276       else {
277         target = path.join(process.cwd(), self.path_available);
278       }
279       return fs.symlink(target, self.path_enabled, cb);
280     }
281     return cb();
282   });
283 };
284 
285 /**
286  * Disable a service by removing the symlink to it from the enabled directory
287  *
288  * @param {Function} cb A callback with a possible error.
289  */
290 RunitService.prototype.disable = function(cb) {
291   var self = this;
292   self.is_enabled(function(enabled) {
293     if (enabled) {
294       return fs.unlink(self.path_enabled, cb);
295     }
296     return cb();
297   });
298 };
299 
300 /**
301  * Disable a service (remove a symlink) and delete the service directory.
302  *
303  * @param {Function} callback A callback with a possible error.
304  */
305 RunitService.prototype.destroy = function(callback) {
306   var self = this;
307 
308   this.disable(function(err) {
309     fsutil.rmtree(self.path_available, callback);
310   });
311 };

328 /**
329  * Retrieve a service within the service directory by name. This takes a
330  * required callback which will receive a possible error and the RunitService
331  * object (which may not represent a valid service, see {@link RunitService}).
332  *
333  * @param {String} name Name of the service.
334  * @param {Function} callback A callback taking an error and a {@link RunitService}.
335  */
336 RunitServiceManager.prototype.get_service = function(name, callback) {
337   var self = this;
338   fs.stat(path.join(self.path_available, name), function(err, stats) {
339     if (err) {
340       return callback(err);
341     }
342     else {
343       return callback(null, new RunitService(self.path_available, self.path_enabled, name));
344     }
345   });
346 };
347 
348 /**
349  * Retrieve a list of objects containing details for all valid services in the
350  * service directory.
351  *
352  * @param {Function} cb A callback taking an error a list of retrieved details.
353  */
354 RunitServiceManager.prototype.list_services_details = function(cb) {
355   var self = this;
356   var services = [];
357 
358   fs.readdir(self.path_available, function(err, files) {
359     // If the services directory can't be read, just return error now
360     if (err) {
361       return cb(err);
362     }
363 
364     async.forEach(files, function(file, callback) {
365       self.get_service(file, function(err, service) {
366         if (err) {
367           return callback(err);
368         }
369 
370         service.get_details(function(err, details) {
371           if (!err) {
372             services.push(details);
373           }
374           return callback(err);
375         });
376 
377       });
378     },
379     function(err) {
380       return cb(err, services);
381     });
382   });
383 };

385 RunitServiceManager.prototype.get_service_template = function(service_name, service_path, entry_file, application_type,
386                                                               callback) {
387   var templates = require('service_management/runit/templates/base');
388 
389   templates.get_application_template(service_name, service_path, entry_file, application_type,
390                                      function(error, template) {
391     if (error) {
392       callback(error);
393       return;
394     }
395 
396     callback(null, template);
397   });
398 };

400 /**
401  * Create a RunitService layout with the given name. This is incomplete.
402  *
403  * @param {String} service_name Service names.
404  * @param {Object} service_template Service template object (see runit/templates/base.js for example).
405  * @param {Function} callback A callback taking a possible error.
406  */
407 RunitServiceManager.prototype.create_service = function(service_name, service_template, callback) {
408   var svcpath = path.join(this.path_available, service_name);
409 
410   fs.stat(svcpath, function(err, stats) {
411     if (!err) {
412       return callback(new Error('Service name already in use'));
413     }
414     else {
415       misc.template_to_tree(svcpath, service_template, true, callback);
416     }
417   });
418 };

420 /*
421  * Return a new RunitServiceManager object.
422  *
423  * @param {String} path_available The directory in which the available services reside.
424  * @param {String} path_enabled The directory in which the enabled services reside.
425  *
426  * @return {RunitServiceManager}
427  */
428 var get_manager = function(path_available, path_enabled) {
429   var _path_available = path_available || path.join(config.get().service_dir_available);
430   var _path_enabled = path_enabled || path.join(config.get().service_dir_enabled);
431 
432   return new RunitServiceManager(_path_available, _path_enabled);
433 };
