 47 /**
 48  * Print a formatted table to standard output.
 49  *
 50  * @param {Array} columns Array of objects. Each object must contains the following properties (optional properties are marked with *):
 51  *                        {String} title - column title
 52  *                        {String} value_property - name of the property in the rows object which holds the value for this column
 53  *                        {Number} padding_left* - left pad the string to the width provided
 54  *                        {Number} padding_right* - right pad the string to the width provided
 55  *                        {Function} format_function* - a function which is applied for each value of this column.
 56  *
 57  * @param {Array} rows Array of objects which hold the values for each column.
 58  * @param {String} no_data_text The text which is printed to standard output if the rows array is empty.
 59  */
 60 exports.print_table = function(columns, rows, no_data_text) {
 61   var string, column_title, value_property, value, padding_left, padding_right;
 62   var _no_data_text = no_data_text || 'No data available';
 63 
 64   if (rows.length === 0) {
 65     sys.print(_no_data_text);
 66   }
 67   else
 68   {
 69     columns.forEach(function(column) {
 70       column_title = column.title;
 71 
 72       if (column.hasOwnProperty('padding_left')) {
 73         padding_left = column.padding_left;
 74       }
 75       else {
 76         padding_left = 0;
 77         column.padding_left = 0;
 78       }
 79 
 80       if (column.hasOwnProperty('padding_right')) {
 81         padding_right = column.padding_right;
 82       }
 83       else {
 84         padding_right = 0;
 85         column.padding_right = 0;
 86       }
 87 
 88       string = misc.lpad(column_title, padding_left);
 89       string = misc.rpad(string, column.padding_right);
 90 
 91       sys.print(string);
 92     });
 93 
 94     sys.print('\n');
 95 
 96     rows.forEach(function(row) {
 97       columns.forEach(function(column) {
 98         column_title = column.title;
 99         value_property = column.value_property || column_title.toLowerCase();
100         value = row[value_property];
101 
102         if (column.hasOwnProperty('format_function')) {
103           value = column.format_function.call(null, value);
104         }
105 
106         string = misc.lpad(value, column.padding_left);
107         string = misc.rpad(string, column.padding_right);
108 
109         sys.print(string);
110       });
111 
112       sys.print('\n');
113     });
114   }
115 
116   sys.print('\n');
117 };

119 /**
120  * Print the provided text to stdout preceeded by the specified number of
121  * spaces and wrapped, at spaces, to a maximum of the specified width.
122  *
123  * @param {String} text   The text to print
124  * @param {Number} width  The maximum line length, defaults to 80
125  * @param {Number} indent How many spaces to use as indent, defaults to 2
126  */
127 exports.print_wrapped = function(text, width, indent) {
128   var chunk, split_index, remaining, indentstr;
129   width = width || 80;
130   indent = indent || 2;
131   indentstr = new Array(indent + 1).join(' ');
132   remaining = text;
133 
134   while (remaining.length > (width - indent)) {
135     chunk = remaining.slice(0, (width - indent));
136     split_index = chunk.lastIndexOf(' ');
137     chunk = chunk.slice(0, split_index);
138     remaining = remaining.substr(split_index + 1);
139     sys.puts(indentstr + chunk);
140   }
141 
142   sys.puts(indentstr + remaining);
143 };

145 /*
146  * Ask user a question and return the input.
147  *
148  * @param {String} question Question which is sent to standard output
149  * @param {Array} valid_options Array of valid options (e.g. ['yes', 'no'])
150  * @param {String} default_option Option which is used as a default if empty line is received
151  * @param {Function} callback Callback which is called with user input
152  */
153 exports.prompt = function(question, valid_options, default_option, callback) {
154   var stdin = process.openStdin();
155   stdin.resume();
156   var options, option, question_mark, data_string;
157 
158   if (valid_options) {
159     if (default_option && !misc.in_array(default_option, valid_options)) {
160       throw new Errorf('Invalid default option: %s', default_option);
161     }
162 
163     option = (default_option) ? sprintf(' [%s]', default_option) : '';
164     options = sprintf(' (%s)%s', valid_options.join(', '), option);
165   }
166   else {
167     options = '';
168   }
169 
170   if (question[question.length - 1] === '?') {
171     question = question.substr(0, (question.length - 1));
172     question_mark = '?';
173   }
174   else {
175     question_mark = '';
176   }
177 
178   sys.print(sprintf('%s%s%s ', question, options, question_mark));
179 
180   function handle_data(data) {
181     data_string = data.toString().trim();
182 
183     if (!data_string && (valid_options && default_option)) {
184       data_string = default_option;
185     }
186 
187     if (data_string) {
188       if (valid_options && !misc.in_array(data_string, valid_options)) {
189         sys.puts(sprintf('Invalid option "%s", valid options are: %s', data_string,
190                          valid_options.join(', ')));
191         return;
192       }
193       // Pause is necessary to get Node to exit without manual intervention
194       stdin.pause();
195       stdin.removeListener('data', handle_data);
196       callback(data_string);
197     }
198   }
199 
200   stdin.on('data', handle_data);
201 };

203 /*
204  * Apply formatting to the input string (replaces formatting tag with the corresponding ANSI escape codes).
205  *
206  * @param {String} string String with the formatting tags (tags can also be nested). For example:
207  *                        [italic]italic text[/italic] [bold][red]bold red text[/red][/bold]
208  *
209  * @return {String} String with formatting applied.
210  */
211 exports.stylize = function(string) {
212   var stylized = string;
213   var style_regex = Object.keys(styles).join('|');
214   var key, regex;
215 
216   var replace_function = function(str, p1, p2, p3) {
217     return sprintf('\033[%sm%s\033[%sm', styles[p2][0], p3, styles[p2][1]);
218   };
219 
220   for (key in styles) {
221     if (styles.hasOwnProperty(key)) {
222       if (stylized.indexOf(sprintf('[%s]', key)) === -1) {
223         continue;
224       }
225 
226       regex = new RegExp(sprintf('(\\[(%s)\\])(.*?)(\\[/(%s)\\])', key, key), 'gi');
227       stylized = stylized.replace(regex, replace_function);
228     }
229   }
230 
231   return stylized;
232 };
233 
234 /*
235  * Apply formatting and call Node puts function for each of the input arguments.
236  *
237  * @param {String}
238 */
239 exports.puts = function() {
240   var term = process.env.TERM;
241 
242   if (!use_ansi_codes || !term || !misc.in_array(term.toLowerCase(), terminal_support_ansi_codes)) {
243     sys.puts.apply(this, arguments);
244     return;
245   }
246 
247   for (var i = 0; i < arguments.length; i++) {
248     sys.puts(exports.stylize(arguments[i]));
249   }
250 };
