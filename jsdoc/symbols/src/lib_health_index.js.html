 18 var CheckStatus = {
 19   SUCCESS: 0,
 20   ERROR: 1,
 21   UNKNOWN: 2
 22 };
 23 
 24 var MAX_RESULTS = 5;
 25 var available_checks = [['http', 'HTTPCheck'], ['tcp', 'TCPCheck']];
 26 
 27 /**
 28  * A class which stores a single check result.
 29  *
 30  * @param {CheckStatus} status The check status.
 31  * @param {String} details The check details.
 32  *
 33  * @constructor
 34  */
 35 function CheckResult(status, details) {
 36   this.date = new Date();
 37 
 38   this.status = status || CheckStatus.UNKNOWN;
 39   this.details = details || '';
 40 }
 41 
 42 /**
 43  * Base check class.
 44  * All the check classes should inherit from it and add their own implementation for the run method.
 45  *
 46  * @param {String} name The check name.
 47  * @param {Array} required_arguments Array of the required arguments.
 48  * @param {Array} optional_arguments Array of the optional arguments.
 49 *  @param {Object} default_values Default values for the optional arguments.
 50  * @param {Object} check_arguments Provided check arguments.
 51 
 52  * @constructor
 53  */
 54 function Check(name, required_arguments, optional_arguments, default_values, check_arguments) {
 55   this.name = name;
 56   this.check_arguments = check_arguments || null;
 57 
 58   this.result_history = [];
 59   this.last_run_date = null;
 60 
 61   if (this.check_arguments) {
 62     this.verify_arguments(required_arguments, this.check_arguments);
 63     this.check_arguments = this.format_arguments(required_arguments,
 64                                                  optional_arguments,
 65                                                  default_values,
 66                                                  this.check_arguments);
 67   }
 68 }
 69 
 70 /**
 71  * Verifies that all the required arguments are provided.
 72  * If some of the required arguments is not provided an exception is thrown.
 73  *
 74  * @param {Array} required_arguments Array of the required arguments.
 75  * @param {Object} check_arguments Check arguments.
 76  */
 77 Check.prototype.verify_arguments = function(required_arguments, check_arguments) {
 78   var missing_arguments = [];
 79 
 80   required_arguments.forEach(function(value, index) {
 81     if (!check_arguments.hasOwnProperty(value)) {
 82       missing_arguments.push(value);
 83     }
 84   });
 85 
 86   if (missing_arguments.length) {
 87     throw new Error('Missing required check argument(s): ' + missing_arguments.join(', '));
 88   }
 89 };
 90 
 91 /**
 92  * Return a new arguments object which only contains the valid arguments and all the
 93  * optional arguments replaced with the default values.
 94  *
 95  * @param {Array} required_arguments Array of the required arguments.
 96  * @param {Array} optional_arguments Array of the optional arguments.
 97  * @param {Object} default_values Default values for the optional arguments.
 98  * @param {Object} check_arguments Check arguments.
 99  *
100  * @return {Object} arguments object.
101  */
102 Check.prototype.format_arguments = function(required_arguments, optional_arguments, default_values, check_arguments) {
103   var i;
104   var item;
105   var formatted_arguments = {};
106 
107   for (i = 0; i < required_arguments.length; i++) {
108     item = required_arguments[i];
109     formatted_arguments[item] = check_arguments[item];
110   }
111 
112   for (i = 0; i < optional_arguments.length; i++) {
113     item = optional_arguments[i];
114 
115     if (check_arguments.hasOwnProperty(item)) {
116       formatted_arguments[item] = check_arguments[item];
117     }
118     else {
119       formatted_arguments[item] = default_values[item];
120     }
121   }
122 
123   return formatted_arguments;
124 };
125 
126 Check.prototype.run = function(callback) {
127   throw new Error('Not implemented');
128 };
129 
130 /**
131  * Add a result to the result cache and (if provided) fire the callback.
132  *
133  * @param {CheckResult} result Check result object.
134  * @param {Function} callback An optional callback which takes the CheckResult object
135  *                            as it's only argument.
136  */
137 Check.prototype.add_result = function(result, callback) {
138   if (this.result_history.length >= MAX_RESULTS) {
139     this.result_history.pop();
140   }
141 
142   this.result_history.unshift(result);
143   this.last_run_date = result.date;
144 
145   if (callback) {
146     callback(result);
147   }
148 };
149 
150 /**
151  * Return the last check result.
152  *
153  * @return {?Object} CheckResult object on success, null otherwise.
154  */
155 Check.prototype.get_last_result = function() {
156   return this.get_result_at_index(0);
157 };
158 
159 /**
160  * Clear the check result history.
161  */
162 Check.prototype.clear_result_history = function() {
163   this.result_history = [];
164 };
165 
166 /**
167  * Return the check result located at the specified index.
168  * If the provided index is invalid, exception is thrown.
169  *
170  * @param {Integer} result_index Index of the check result (the most recent result is located on the
171  * first place - index 0).
172  *
173  * @return {?Object} object on success, null otherwise.
174  */
175 Check.prototype.get_result_at_index = function(result_index) {
176   var result;
177   var index = result_index || 0;
178 
179   if (!this.result_history.length) {
180     return null;
181   }
182 
183   if ((index < 0) || (index > this.result_history.length)) {
184     throw new Error('Invalid result index specified.');
185   }
186 
187   result = this.result_history[index];
188   return result;
189 };
190 
191 exports.Check = Check;
192 exports.CheckStatus = CheckStatus;
193 exports.CheckResult = CheckResult;
194 
195 exports.available_checks = available_checks;
