 18 var fs = require('fs');
 19 var path = require('path');
 20 
 21 var async = require('extern/async');
 22 
 23 var config = require('util/config');
 24 var fsutil = require('util/fs');
 25 var certgen = require('security/certgen.js');
 26 var norris = require('norris');
 27 
 28 var constants = require('util/constants');
 29 
 30 /*
 31  * Function which determines if this is the first start of the agent.
 32  *
 33  * @param {Function} callback Callback which is called with a possible error as the first
 34  *                            argument and true as the second one if this if the first run,
 35  *                            false otherwise.
 36  */
 37 function is_first_run(callback) {
 38   var p = path.join(config.get().data_root, '.cast-first-run');
 39   fs.open(p,
 40           constants.O_WRONLY | constants.O_CREAT | constants.O_EXCL,
 41           0600,
 42     function(err, fd) {
 43       if (err) {
 44         if (err.errno === constants.EEXIST) {
 45           // Ignore EEXIST, this simply means this is not first run
 46           callback(null, false);
 47           return;
 48         }
 49         else if (err.errno === constants.ENOENT) {
 50           // If the data_root directory does not exist, this means that this is a first run
 51           callback(null, true);
 52           return;
 53         }
 54 
 55         callback(err, false);
 56         return;
 57       }
 58 
 59       fs.closeSync(fd);
 60       callback(null, true);
 61   });
 62 }
 63 
 64 /*
 65  * This function preforms the initialization process.
 66  *
 67  * @param {Function} callback Callback which is called with a possible error.
 68  */
 69 exports.initialize = function(callback) {
 70   is_first_run(function(err, first_run) {
 71     if (err) {
 72       callback('Failed to determine if this is the first run: ' + err);
 73       return;
 74     }
 75 
 76     var ops = [];
 77 
 78     if (first_run) {
 79       ops.push(function(callback) {
 80         var paths_to_create;
 81          // Create the neccessary directories
 82         fsutil.ensure_directory(config.get().data_root, function(err) {
 83           if (err) {
 84             callback(err);
 85             return;
 86           }
 87 
 88           paths_to_create = [config.get().data_dir,
 89                              config.get().service_dir_available,
 90                              config.get().bundle_dir,
 91                              config.get().extracted_dir,
 92                              config.get().app_dir];
 93 
 94           async.forEachSeries(paths_to_create,
 95             function(path_, callback) {
 96               fsutil.ensure_directory(path_, callback);
 97             },
 98 
 99             function(err) {
100               if (err) {
101                 callback(err);
102                 return;
103               }
104 
105               callback();
106           });
107         });
108       });
109     }
110 
111     ops.push(function(callback) {
112       path.exists(config.get().ssl_cert, function(exists) {
113         if (!exists) {
114           norris.get(function(facts)  {
115             certgen.selfsigned(facts.hostname,
116               config.get().ssl_key,
117               config.get().ssl_cert,
118               function(err) {
119                 callback(err);
120               });
121           });
122           return;
123         }
124         else {
125           callback();
126         }
127       });
128     });
129 
130     async.series(ops,
131       function(err, results) {
132         callback(err);
133       });
134   });
135 };
