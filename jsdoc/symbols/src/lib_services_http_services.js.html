 18 var clutch = require('extern/clutch');
 19 var log = require('util/log');
 20 var config = require('util/config');

 26 var service_management = require('service_management');
 27 
 28 function list_services(req, res) {
 29   var manager = service_management.get_default_manager().get_manager();
 30 
 31   manager.list_services_details(function(err, services) {
 32     if (err) {
 33       return http.return_error(res, 500, err.message);
 34     }
 35     http.return_json(res, 200, services);
 36   });
 37 }
 38 
 39 function get_service(req, res, service) {
 40   var manager = service_management.get_default_manager().get_manager();
 41 
 42   manager.get_service(service, function(err, svc) {
 43     if (err) {
 44       return http.return_error(res, 404, 'Service not found');
 45     }
 46 
 47     svc.get_details(function(err, details) {
 48       if (err) {
 49         return http.return_error(res, 500, err.message);
 50       }
 51 
 52       http.return_json(res, 200, details);
 53     });
 54   });
 55 }
 56 
 57 function tail_service(req, res, service, bytes_to_read) {
 58   var log_file;
 59 
 60   var manager = service_management.get_default_manager().get_manager();
 61   manager.get_service(service, function(err, svc) {
 62     if (err) {
 63       return http.return_error(res, 404, 'Service not found');
 64     }
 65 
 66     log_file = svc.get_log_path();
 67     tail_file(log_file, bytes_to_read, false, function(error, data, unsubscribe) {
 68       if (error) {
 69         return http.return_error(res, bytes_to_read, error.message);
 70       }
 71 
 72       res.writeHead(200, { 'Content-Type': 'text/plain' });
 73 
 74       res.end(data);
 75     });
 76   });
 77 }
 78 
 79 function tail_follow_service(req, res, service, bytes_to_read) {
 80   var log_file;
 81   var head_written = false;
 82   var listeners_set = false;
 83 
 84   var manager = service_management.get_default_manager().get_manager();
 85   manager.get_service(service, function(err, svc) {
 86     if (err) {
 87       return http.return_error(res, 404, 'Service not found');
 88     }
 89 
 90     res.connection.setTimeout(0);
 91 
 92     log_file = svc.get_log_path();
 93     tail_file(log_file, bytes_to_read, true, function(error, data, unsubscribe) {
 94       if (error) {
 95         return http.return_error(res, bytes_to_read, error.message);
 96       }
 97 
 98       if (!head_written) {
 99         res.writeHead(200, { 'Content-Type': 'text/plain', 'Connection': 'keep-alive' });
100         head_written = true;
101       }
102 
103       res.write(data);
104       res.connection.setKeepAlive(true);
105 
106       if (!listeners_set) {
107         res.on('error', unsubscribe);
108         req.on('error', unsubscribe);
109         req.connection.on('end', unsubscribe);
110 
111         listeners_set = true;
112       }
113     });
114   });
115 }

117 function action_service(req, res, action, service) {
118   var manager = service_management.get_default_manager().get_manager();
119 
120   manager.get_service(service, function(err, svc) {
121     if (err) {
122       return http.return_error(res, 404, 'Service not found');
123     }
124     svc[action](function(err) {
125       if (err) {
126         return http.return_error(res, 500, err.message);
127       }
128       // TODO: This doesn't make much sense
129       http.return_json(res, 200, {
130         service: service,
131         method: action,
132         result: 'success'
133       });
134     });
135   });
136 }

138 function enable_service(req, res, service) {
139   action_service(req, res, 'enable', service);
140 }

142 function disable_service(req, res, service) {
143   action_service(req, res, 'disable', service);

147   action_service(req, res, 'start', service);
148 }
149 
150 function stop_service(req, res, service) {
151   action_service(req, res, 'stop', service);
152 }
153 
154 function restart_service(req, res, service) {
155   action_service(req, res, 'restart', service);
156 }
