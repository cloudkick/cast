 18 var exec = require('child_process').exec;
 19 var sprintf = require('extern/sprintf').sprintf;
 20 var log = require('util/log');
 21 
 22 /**
 23  * Generates a Self signed X509 Certificate.
 24  *
 25  * @param {String} Hostname to generate the certficate for.
 26  * @param {String} Path to write the private key to.
 27  * @param {String} Path to write the certificate to.
 28  * @param {Function} Callback, first parameter is null on success, contains an error otherwise.
 29  */
 30 exports.selfsigned = function(hostname, output_key, output_cert, cb)
 31 {
 32   var cmd = sprintf("openssl req -batch -x509 -nodes -days 1825 -subj '/CN=%s' -sha1 -newkey rsa:2048 -keyout %s -out %s",
 33                      hostname, output_key, output_cert);
 34   exec(cmd, function(err, stdout, stderr) {
 35     if (err) {
 36       log.err('openssl command failed: ' + cmd, err, stdout, stderr);
 37       cb(err);
 38       return;
 39     }
 40     cb(null);
 41   });
 42 };
