 18 /* The principle of this app is as follows.
 19  * as messages come in it adds it to the end of
 20  * the buffer and then copies it to a new buffer shifted over
 21  * chunk.length to make room for the new log lines.
 22  */
 23 var stdin = process.openStdin(),
 24     http = require('http'),
 25     Buffer = require('buffer').Buffer,
 26     total = 512,
 27     buf = new Buffer(total),
 28     offset = 0;
 29 
 30 stdin.setEncoding('utf8');
 31 
 32 // Listen for the data and add it to the "Ring" Buffer
 33 stdin.addListener('data', function(chunk) {
 34   process.stdout.write(chunk);
 35   if (offset + chunk.length > total) {
 36     // Going to buffer overrun.
 37     var old_buf = buf;
 38     buf = new Buffer(total);
 39     old_buf.copy(buf, 0, chunk.length, old_buf.length);
 40     offset -= chunk.length;
 41   }
 42   offset += buf.write(chunk, offset);
 43 });
 44 
 45 http.createServer(function(request, response) {
 46   response.writeHead(200, {
 47     'content-type': 'text/plain',
 48     'stream': 'keep-alive',
 49     'accept': '*/*'});
 50   response.write(buf.slice(0, offset));
 51   stdin.addListener('data', function(chunk) {
 52     response.write(chunk);
 53   });
 54 }).listen(8125);
 55 
 56 console.log('Server running at http://127.0.0.1:8125/');
