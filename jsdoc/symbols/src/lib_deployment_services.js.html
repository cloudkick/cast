 29 /**
 30  * Creates an instance of a service, building out the runit application directory.
 31  * @param {String} instance_name Name of the instance.
 32  * @param {String} instance_path Path to of the instance.
 33  * @param {String} entry_file Path of the file to run inside runit.
 34  * @param {String} application_type Type of the application template to use, must be "shell" or "nodejs" at this time.
 35  * @param {Function} callback Callback on completion, first parameter if present is an error.
 36  */
 37 exports.create_service = function(instance_name, instance_path, entry_file, application_type, callback) {
 38   var manager = service_management.get_default_manager().get_manager();
 39 
 40   manager.get_service_template(instance_name, instance_path, entry_file, application_type, function(err, template) {
 41     if (err) {
 42       callback(err);
 43       return;
 44     }
 45 
 46     manager.create_service(instance_name, template, function(err) {
 47       if (err) {
 48         callback(err);
 49         return;
 50       }
 51 
 52       callback();
 53     });
 54   });
 55 };
