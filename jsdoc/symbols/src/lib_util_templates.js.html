 28 /**
 29  * Return context object which can be used when rendering a template.
 30  *
 31  * @param {Boolean} use_norris_facts If true, the template context will be populated with the norris facts.
 32  * @param {Object} context Other context with which the resulting context will be merged.
 33  * @param {Function} callback Callback which is called with a possible error as the first argument and the context
 34  *                            object as the second one on success.
 35  */
 36 var get_template_context = function(use_norris_facts, context, callback) {
 37   var template_context = {};
 38 
 39   async.series([function(callback) {
 40     if (use_norris_facts) {
 41       norris.get(function(facts) {
 42 
 43         template_context = misc.merge(template_context, {'facts': facts});
 44         callback();
 45       });
 46     }
 47     else {
 48       callback();
 49     }
 50   }],
 51 
 52   function(error) {
 53     if (error) {
 54       return callback(error);
 55     }
 56 
 57     template_context = misc.merge(template_context, context);
 58 
 59     callback(null, template_context);
 60   });
 61 };

 63 /**
 64  * Return object with instance data which can be used when rendering a template.
 65  *
 66  * @param {String} instance_name Instance name.
 67  * @param {Number} instance_number Instance number.
 68  * @param {String} instance_path Instance path.
 69  * @return {Object}
 70  */
 71 var get_instance_template_object = function(instance_name, instance_number, instance_path) {
 72   var instance_object = {
 73     'name': instance_name,
 74     'number': instance_number,
 75     'path': instance_path
 76   };
 77 
 78   return instance_object;
 79 };

 81 /**
 82  * Render a template with the provided context.
 83  *
 84  * @param {String} template_path Full path to a template file.
 85  * @param {Object} context Context which is used when rendering a template.
 86  * @param {Function} callback Callback which is called with a possible error as a first argument and the rendered
 87  *                            template String as the second argument on success.
 88  */
 89 var render_template_as_string = function(template_path, context, callback) {
 90   var template = new T.Template(template_path);
 91 
 92   template.load(function(error, template) {
 93     if (error) {
 94       return callback(error);
 95     }
 96 
 97     template.render(context, function(error, output) {
 98       if (error) {
 99         return callback(error);
100       }
101 
102       callback(null, output.join(''));
103     });
104   });
105 };

107 /**
108  * Return an object which can be used with the template_to_tree function.
109  *
110  * @param {String} file_path Relative path to a file (e.g. root_dir/subdir/file2.html).
111  * @param {Object} context Context which is used when rendering a template.
112  * @param {Function} callback Callback which is called with a possible error as a first argument and the rendered
113  *                            template String as the second argument on success.
114  */
115 var get_template_data_as_directory_tree_object = function(file_path, data) {
116   var directory_tree_object = {};
117   var prev_obj_ref, item;
118 
119   var paths = file_path.split('/');
120   var paths_length = paths.length;
121 
122   if (paths.length === 1) {
123     directory_tree_object[file_path] = data;
124   }
125   else {
126     for (var i = 0; i < (paths_length - 1); i++) {
127       item = paths[i];
128 
129       if (!prev_obj_ref) {
130         directory_tree_object[item] = {};
131         prev_obj_ref = directory_tree_object[item];
132       }
133       else {
134         prev_obj_ref[item] = {};
135         prev_obj_ref = prev_obj_ref[item];
136       }
137     }
138 
139     item = paths[paths_length - 1];
140     prev_obj_ref[item] = data;
141   }
142 
143   return directory_tree_object;
144 };

146 /**
147  * Render and save a list of templates.
148  *
149  * @param {String} templates_path Path to a directory which contains the template files.
150  * @param {Array} templates_list Array of relative paths to the template files.
151  * @param {String} string_path Path to the target directory where the rendered templates will be saved.
152  * @param {Object} context Context which is used when rendering a template.
153  * @param {Function} callback Callback which is called with a possible error as a first argumen.
154  */
155 var render_and_save_templates = function(templates_path, template_list, target_path, context, callback) {
156   var templates_directory_tree_object = {};
157   var directory_tree_object, template_path;
158 
159   async.forEach(template_list, function(template_file, callback) {
160     template_path = path.join(templates_path, template_file);
161 
162     render_template_as_string(template_path, context, function(error, rendered_template) {
163       if (error) {
164         // Template rendering failed, skip this template
165         return callback();
166       }
167 
168       // Otherwise save the rendered template in the target_path
169       directory_tree_object = get_template_data_as_directory_tree_object(template_file, rendered_template);
170       misc.template_to_tree(target_path, directory_tree_object, true, function(error) {
171         return callback(error);
172       });
173     });
174   },
175 
176   function(error) {
177     callback(error);
178   });
179 };
