 18 var path = require('path');
 19 var fs = require('fs');
 20 
 21 /**
 22  * This detects if the entry file is a symlinked path, and tries to
 23  * insert the correct lib into the require.paths for eveyrthing to work correctly.
 24  */
 25 exports.path_setup = function(file_path) {
 26   var stat, root;
 27   var p =  file_path || __filename;
 28 
 29   stat = fs.lstatSync(p);
 30   if (stat.isSymbolicLink()) {
 31     p = fs.readlinkSync(p);
 32   }
 33 
 34   root = path.dirname(p);
 35 
 36   // in-place path
 37   require.paths.unshift(path.join(root, '../lib'));
 38   // installed path
 39   require.paths.unshift(path.join(root, '../lib/cast'));
 40 };
