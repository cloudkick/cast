 18 var path = require('path');
 19 
 20 var config = require('util/config');
 21 var Errorf = require('util/misc').Errorf;
 22 
 23 var available_managers = {
 24   'runit': {
 25       'service': require('service_management/runit').RunitService,
 26       'manager': require('service_management/runit').RunitServiceManager,
 27       'get_manager': require('service_management/runit').get_manager,
 28       'config_defaults': require('service_management/runit').config_defaults
 29   }
 30 };
 31 
 32 /*
 33  * Property holding the service manager object.
 34  */
 35 var default_manager = null;
 36 
 37 /**
 38  * Return service manager object.
 39  *
 40  * @param {String} name Manager name.
 41  * @return {Object} Object with the following keys: service, manager, get_manager, config_defaults
 42  */
 43 exports.get_manager = function(name) {
 44 
 45   if (!available_managers.hasOwnProperty(name)) {
 46     throw new Errorf('%s service manager does not exist.', name);
 47   }
 48 
 49   return available_managers[name];
 50 };
 51 
 52 /**
 53  * Return a default service manager object.
 54  *
 55  * @return {Object} Default manager object with the following keys: service, manager, get_manager, config_defaults
 56  */
 57 exports.get_default_manager = function() {
 58   var service_manager;
 59 
 60   if (!default_manager) {
 61     // Manager should not change during run-time, so caching the value is ok
 62     service_manager = config.get().service_manager;
 63     default_manager = exports.get_manager(service_manager);
 64   }
 65 
 66   return default_manager;
 67 };
