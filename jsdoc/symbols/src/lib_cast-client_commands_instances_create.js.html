 18 var sys = require('sys');
 19 var fs = require('fs');
 20 var path = require('path');
 21 
 22 var sprintf = require('extern/sprintf').sprintf;
 23 var async = require('extern/async');
 24 
 25 var http = require('util/http');
 26 var misc = require('util/misc');
 27 
 28 var enable_commands = require('cast-client/commands/instances/enable');
 29 
 30 var config = {
 31   'short_description': 'Create a new application instance',
 32   'long_description': 'Create a new application instance.',
 33   'required_arguments' : [['appname', 'Application name'], ['version', 'Application version number']],
 34   'optional_arguments': [],
 35   'options': [
 36     {
 37       names: ['--enable'],
 38       dest: 'enable',
 39       action: 'store_true',
 40       desc: 'Enable and start the instance after it has been created'
 41     }
 42   ]
 43 };
 44 
 45 function handle_command(args) {
 46   var instance_number;
 47 
 48   var bundle_name = misc.get_valid_bundle_name(args.appname);
 49   var bundle_name_full = misc.get_full_bundle_name(bundle_name, args.version);
 50 
 51   http.get_response(undefined, sprintf('/instances/%s/', bundle_name_full), 'POST', true,
 52                     function(error, response) {
 53     if (error) {
 54       return sys.puts(sprintf('Error: %s', error.message));
 55     }
 56 
 57     if (response.status_code !== 200) {
 58       return sys.puts(sprintf('HTTP Error, status code: %d, returned body: %s', response.status_code,
 59                                                                                 response.body.message));
 60     }
 61 
 62     instance_number = response.body.instance_number;
 63 
 64     sys.puts(sprintf('New instance for application %s has been successfully created.', args.appname));
 65     sys.puts(sprintf('Instance number: %s', instance_number));
 66 
 67     if (args.enable) {
 68       enable_commands.handle_command({appname: args.appname, version: args.version,
 69                                        number: instance_number});
 70     }
 71   });
 72 }
 73 
 74 exports.config = config;
 75 exports.handle_command = handle_command;
