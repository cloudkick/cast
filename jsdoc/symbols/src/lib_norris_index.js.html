 18 /**
 19  * Norris Engine gives you Facts about the currently running machine.
 20  */
 21 
 22 var config = require('util/config');
 23 var misc = require('util/misc');
 24 var log = require('util/log');
 25 var async = require('extern/async');
 26 var defaults = {};
 27 var factcache = {};
 28 var expiry = new Date(0);
 29 var requests = [];
 30 var facttypes = ['arch', 'hostname', 'kernel', 'gnutar', 'runit_services_path', 'node_binary', 'node_version'];
 31 
 32 
 33 /**
 34  * Expire Norris fact cache
 35  */
 36 exports.expire = function() {
 37   expiry = Date.now();
 38 };
 39 
 40 
 41 /**
 42  * Retrieve Norris facts
 43  *
 44  * @param {Function} cb A callback taking the facts object as its only argument.
 45  */
 46 exports.get = function(cb) {
 47   var now = Date.now();
 48 
 49   // If the cache is expired rebuild it
 50   if (now >= expiry) {
 51     requests.push(cb);
 52 
 53     // If a refresh is already in progress, just subscribe for the results
 54     if (requests.length > 1) {
 55       return;
 56     }
 57 
 58     // Otherwise we need to actually rebuild the cache ourselves
 59     factcache = misc.merge({}, defaults);
 60 
 61     async.forEach(facttypes, function(item, callback) {
 62       var ft = require('./facts/' + item);
 63       ft.get(function(fact) {
 64         factcache = misc.merge(factcache, fact);
 65         callback();
 66       });
 67     },
 68     function(err) {
 69       // Have incoming requests hit the cache directly at this point
 70       expiry = new Date(now + (config.get().norris_ttl * 1000));
 71       var subscribers = requests;
 72       var facts = factcache;
 73       requests = [];
 74 
 75       // Notify subscribers
 76       async.forEach(subscribers, function(subscriber, callback) {
 77         subscriber(facts);
 78         callback();
 79       }, function() {});
 80     });
 81   }
 82 
 83   // Otherwise just return the cache
 84   else {
 85     process.nextTick(function() {
 86       cb(factcache);
 87     });
 88   }
 89 };
