 51 /**
 52  * Recursively create a directory.
 53  *
 54  * @param {String} ensure  The path to recursively create.
 55  * @param {number} chmod  File permissions to use when creating directories.
 56  * @param {Function} callback     Callback, taking a possible error.
 57  */
 58 exports.mkdir = function(ensure, chmod, callback)
 59 {
 60   if (ensure.charAt(0) !== '/') {
 61     ensure = path.join(process.cwd(), ensure);
 62   }
 63 
 64   var dirs = ensure.split('/');
 65   var walker = [];
 66 
 67   if (arguments.length < 3) {
 68     callback = chmod;
 69     chmod = 0755;
 70   }
 71 
 72   // gobble the "/" first
 73   walker.push(dirs.shift());
 74 
 75   (function S(d) {
 76 
 77     if (d === undefined) {
 78       return callback();
 79     }
 80 
 81     walker.push(d);
 82     var dir = walker.join('/');
 83 
 84     fs.stat(dir, function(er, s) {
 85 
 86       if (er) {
 87         fs.mkdir(dir, chmod, function(er, s) {
 88           if (er && er.message.indexOf('EEXIST') === 0) {
 89             // When multiple concurrent actors are trying to ensure the same directories,
 90             // it can sometimes happen that something doesn't exist when you do the stat,
 91             // and then DOES exist when you try to mkdir.  In this case, just go back to
 92             // the stat to make sure it's a dir and not a file.
 93             S('');
 94             return;
 95           }
 96           else if (er) {
 97             callback(new Error('Failed to make ' + dir + ' while ensuring ' + ensure + '\n' + er.message));
 98             return;
 99           }
100           S(dirs.shift());
101         });
102       }
103       else {
104         if (s.isDirectory()) {
105           S(dirs.shift());
106         }
107         else {
108           callback(new Error('Failed to mkdir ' + dir + ': File exists'));
109         }
110       }
111     });
112   })(dirs.shift());
113 };

116 function rmtree_(p, callback_, error_context) {
117   if (!p) {
118     return callback_(new Error('Trying to rm nothing?'));
119   }
120 
121   if (error_context === undefined) {
122     error_context = {};
123     error_context.has_error = false;
124   }
125 
126   function callback(err) {
127     if (err) {
128       error_context.has_error = true;
129     }
130     callback_(err);
131   }
132 
133   fs.lstat(p, function(er, s) {
134     if (error_context.has_error) {
135       return;
136     }
137 
138     if (er) {
139       return callback(new Error('Failed to lstat: ' + er));
140     }
141 
142     if (s.isDirectory()) {
143       fs.readdir(p, function(er, files) {
144         if (error_context.has_error) {
145           return;
146         }
147 
148         if (er) {
149           return callback(er);
150         }
151 
152         var count = files.length;
153         var n = 0;
154         function dirdone(err) {
155           if (error_context.has_error) {
156             return;
157           }
158           if (err) {
159             /* @FIXME: Function does not return on error. */
160             callback(err);
161           }
162           else {
163             n++;
164             if (n >= count) {
165               fs.rmdir(p, function(err) {
166                 if (error_context.has_error) {
167                   return;
168                 }
169                 callback(err);
170               });
171             }
172           }
173         }
174 
175         if (count === 0) {
176           dirdone();
177         }
178         else {
179           files.forEach(function(file) {
180             if (error_context.has_error) {
181               return;
182             }
183             rmtree_(path.join(p, file), dirdone, error_context);
184           });
185         }
186       });
187     }
188     else {
189       fs.unlink(p, function(er) {
190         if (error_context.has_error) {
191           return;
192         }
193         return callback(er);
194       });
195     }
196   });
197 }

199 /**
200  * Deletes an entire tree off the filesystem, first removing all files,
201  * and then deleting the directories.
202  *
203  * @param {String} p  The path to recursively delete.
204  * @param {Function} callback_     Callback on finish, taking a possible error.
205  * @return {Undefined} No return value, uses callack.
206  */
207 exports.rmtree = function(p, callback_) {
208   return rmtree_(p, callback_);
209 };

211 var copy_tree_ = function(source_path, destination_path, callback_, error_context) {
212    if (error_context === undefined) {
213     error_context = {};
214     error_context.has_error = false;
215   }
216 
217   function callback(err) {
218     if (err) {
219       error_context.has_error = true;
220     }
221 
222     callback_(err);
223   }
224 
225   fs.lstat(source_path, function(err, stats) {
226     if (error_context.has_error) {
227       return;
228     }
229 
230     if (err) {
231       return callback(new Error('Failed to lstat: ' + err));
232     }
233 
234     if (stats.isDirectory()) {
235       fs.mkdir(destination_path, 0755, function(err) {
236         if (err && err.errno !== process.EEXIST) {
237           return callback(err);
238         }
239 
240         fs.readdir(source_path, function(err, files) {
241           if (error_context.has_error) {
242             return;
243           }
244 
245           if (err) {
246             return callback(err);
247           }
248 
249           var count = files.length;
250           var n = 0;
251           function dirdone(err) {
252             if (error_context.has_error) {
253               return;
254             }
255 
256             if (err) {
257               callback(err);
258             }
259             else {
260               n++;
261               if (n >= count) {
262                 callback(err);
263               }
264             }
265           }
266 
267           if (count === 0) {
268             dirdone();
269           }
270           else {
271             files.forEach(function(file) {
272               if (error_context.has_error) {
273                 return;
274               }
275 
276               copy_tree_(path.join(source_path, file), path.join(destination_path, file), dirdone, error_context);
277             });
278           }
279         });
280       });
281     }
282     else {
283       exports.copy_file(source_path, destination_path, function(err) {
284         if (error_context.has_error) {
285           return;
286         }
287 
288         return callback(err);
289       });
290     }
291   });
292 };

294 /**
295  * Recursivly copy a directory.
296  *
297  * @param {String} source_path Source path.
298  * @param {String} target_path Target path.
299  * @param {Function} callback Callback which is called with a possible error.
300  */
301 exports.copy_tree = function(source_path, destination_path, callback) {
302   source_path = (source_path.charAt(source_path.length - 1) === '/') ? source_path : source_path + '/';
303   destination_path = (destination_path.charAt(destination_path.length - 1) === '/') ? destination_path :
304                       destination_path + '/';
305 
306   if (destination_path.indexOf(source_path) === 0) {
307     return callback(new Error('Destination path is inside a source path'));
308   }
309 
310   fs.stat(source_path, function(err, stats) {
311     if (err) {
312       return callback(err);
313     }
314 
315     if (!stats.isDirectory()) {
316       return callback(new Error('Source path must be a directory'));
317     }
318 
319     copy_tree_(source_path, destination_path, callback);
320   });
321 };

323 /**
324  * Copy a file.
325  *
326  * @param {String} source_path Source path.
327  * @param {String} destination_path Destination path.
328  * @param {Function} callback Callback which is called with a possible error.
329  */
330 exports.copy_file = function(source_path, destination_path, callback) {
331   var config = require('util/config');
332 
333   fs.stat(source_path, function(err, stats) {
334     if (err) {
335       return callback(err);
336     }
337 
338     if (stats.isDirectory()) {
339       return callback(new Error('Source path must be a file'));
340     }
341 
342     var reader = fs.createReadStream(source_path, { 'bufferSize': config.get().fileread_buffer_size });
343     var writer = fs.createWriteStream(destination_path);
344 
345     sys.pump(reader, writer, callback);
346   });
347 };

349 /**
350  * Return an array of files and directories matching a provided pattern.
351  *
352  * @param {String} directory_path Full path to a directory.
353  * @param {String} match_pattern Regular expression match pattern.
354  * @param {Boolean} exclude_directories If true, directories will not be included in the result.
355  * @param {Function} callback Callback which is called with a possible error as the first argument and
356  *                            array of files matching a pattern as the second one on success.
357  */
358 exports.get_matching_files = function(directory_path, match_pattern, exclude_directories, callback) {
359   var filter_regex;
360   var files = [];
361 
362   try {
363     filter_regex = new RegExp(match_pattern);
364   }
365   catch (err) {
366     callback(err);
367     return;
368   }
369 
370   var filter_file = function(file, callback) {
371     var file_path = path.join(directory_path, file);
372 
373     fs.lstat(file_path, function(err, stats) {
374       if (err) {
375         callback(false);
376         return;
377       }
378 
379       if (exclude_directories && stats.isDirectory()) {
380         callback(false);
381         return;
382       }
383 
384       if (!file.match(filter_regex)) {
385         callback(false);
386         return;
387       }
388 
389       callback(true);
390       return;
391     });
392   };
393 
394   fs.readdir(directory_path, function(err, files) {
395     if (err) {
396       callback(err);
397       return;
398     }
399 
400     async.filter(files, filter_file, function(results) {
401       callback(null, results);
402     });
403   });
404 };

406 /**
407  * Return object with all the directory names as keys which can be used with the template_to_tree function.
408  *
409  * @param {String} source_path Full path to a directory.
410  * @param {Array} ignored_paths Which paths relative to the source_path to skip (path name must contain a trailing
411  *                               slash).
412  * @param {Function} callback Callback which is called with a possible error as the first argument and
413  *                            template object as the second one on success.
414  */
415 exports.get_directory_tree_as_template_object = function(source_path, ignored_paths, callback) {
416   var directory_tree_object = {};
417 
418   function walk_directory(directory_path, ref, callback_) {
419     fs.readdir(directory_path, function(err, files) {
420       if (err) {
421         callback_(err);
422         return;
423       }
424 
425       async.forEach(files, function(file, callback) {
426         var current_path = path.join(directory_path, file);
427         var relative_path = current_path.replace(source_path + '/', '');
428         var curr_ref;
429 
430         fs.stat(current_path, function(err, stats) {
431           if (err) {
432             callback(err);
433             return;
434           }
435 
436           if (!stats.isDirectory()) {
437             callback();
438             return;
439           }
440 
441           // Skip ignored path
442           if (misc.in_array(relative_path + '/', ignored_paths)) {
443             return callback();
444           }
445 
446           ref[file] = {};
447           curr_ref = ref[file];
448 
449           walk_directory(current_path, curr_ref, callback);
450         });
451       },
452 
453       function(err) {
454         if (err) {
455           callback_(err);
456           return;
457         }
458 
459         callback_();
460       });
461     });
462   }
463 
464   walk_directory(source_path, directory_tree_object, function(err) {
465     if (err) {
466       callback(err);
467       return;
468     }
469 
470     callback(null, directory_tree_object);
471   });
472 };

474 /**
475  * Create a hardlink in the target path to a file in the source path for all the files in the source path file tree.
476  *
477  * @param {String} source_path Source path.
478  * @param {String} target_path Target path.
479  * @param {Array} ignored_paths Which paths relative to the source_path to skip (if a path is a directory, it must
480  *                              include a trailing slash).
481  * @param {Function} callback Callback which is called with a possible error as a first argument.
482  */
483 exports.hard_link_files = function(source_path, target_path, ignored_paths, callback) {
484   function walk_directory(directory_path, callback_) {
485     fs.readdir(directory_path, function(err, files) {
486       if (err) {
487         return callback_(err);
488       }
489 
490       async.forEach(files, function(file, callback) {
491         var source_file_path = path.join(directory_path, file);
492 
493         var current_directory = directory_path.replace(source_path, '');
494         var current_path = path.join(current_directory, file).substr(1);
495         var target_file_path = path.join(target_path, current_directory, file);
496 
497         fs.stat(source_file_path, function(err, stats) {
498           if (err) {
499             return callback(err);
500           }
501 
502           if (stats.isDirectory()) {
503             current_path = current_path + '/';
504           }
505 
506           // Skip ignored paths
507           if (misc.in_array(current_path, ignored_paths)) {
508             return callback();
509           }
510 
511           if (stats.isDirectory()) {
512             return walk_directory(source_file_path, callback);
513           }
514 
515           fs.link(source_file_path, target_file_path, function(err) {
516             if (err) {
517               return callback(err);
518             }
519 
520             callback();
521           });
522         });
523       },
524 
525       function(err) {
526         if (err) {
527           return callback_(err);
528         }
529 
530         callback_();
531       });
532     });
533   }
534 
535   walk_directory(source_path, function(err) {
536     callback(err);
537   });
538 };

540 /**
541  * Create a symbolic links for the specified paths.
542  *
543  * @param {Array} paths Array of tuples where the first item is a source path and the second one is a destination path.
544  * @param {Boolean} create_missing_paths true to create the missing paths.
545  * @param {Function} callback Callback which is called with a possible error.
546  */
547 exports.symbolic_link_files = function(paths, create_missing_paths, callback) {
548   async.forEach(paths, function(file_path, callback) {
549     var destination_directory;
550 
551     var source_path = file_path[0];
552     var destination_path = file_path[1];
553 
554     // Remove trailing slash (if it exists)
555     if (destination_path.charAt(destination_path.length - 1) === '/') {
556       destination_path = destination_path.substr(0, (destination_path.length - 1));
557     }
558 
559     destination_directory = path.dirname(destination_path);
560 
561     path.exists(destination_directory, function(exists) {
562       if (!exists) {
563         if (!create_missing_paths) {
564           callback(new Error('Cannot create a symbolic link, because a destination directory does not exist'));
565         }
566         else {
567           exports.mkdir(destination_directory, 0755, function(err) {
568             if (err) {
569               return callback(err);
570             }
571 
572             fs.symlink(source_path, destination_path, callback);
573           });
574         }
575       }
576       else {
577         fs.symlink(source_path, destination_path, callback);
578       }
579     });
580   },
581 
582   function(err) {
583     if (err) {
584       return callback(err);
585     }
586 
587     callback();
588   });
589 };

591 /**
592  * Change the owner for the specified paths.
593  *
594  * @oaram {Array} Array of triples where the first item is the full path to a file / directory,
595  *                second one is a user id and the third one is a group id
596  * @param {Function} callback Callback which is called with a possible error.
597  *
598  */
599 exports.chown_paths = function(paths, callback) {
600   var file_path, uid, gid;
601 
602   async.forEach(paths, function(item, callback) {
603     file_path = item[0];
604     uid = item[1];
605     gid = item[2];
606 
607     fs.chown(file_path, uid, gid, callback);
608   },
609 
610   function(error) {
611     callback(error);
612   });
613 };

615 /**
616  * Make sure a directory exists, create it (non recursively) if not. The
617  * callback takes an error which will occur if creation fails, the path
618  * already exists and is not a directory, or stat-ing the file fails for
619  * whatever reason.
620  *
621  * @param {String} p  The path to the directory to ensure.
622  * @param {Function} callback The callback which takes a possible error.
623  */
624 exports.ensure_directory = function(p, callback) {
625   path.exists(p, function(exists) {
626     if (exists) {
627       fs.stat(p, function(err, stats) {
628         if (err) {
629           return callback(err);
630         }
631         else if (!stats.isDirectory()) {
632           return callback(new Error('Path exists and is not a directory: ' + p));
633         }
634         else {
635           return callback();
636         }
637       });
638     }
639     else {
640       fs.mkdir(p, 0755, callback);
641     }
642   });
643 };

645 /**
646  * Reads a JSON format file off of disk.
647  *
648  * @param {String} path The path to read from.
649  * @param {Function} callback The callback which takes a possible error, second parameter is the json object.
650  */
651 exports.json_file = function(path, callback)
652 {
653   var read_stream = fs.createReadStream(path);
654   var data_buffer = [];
655 
656   function end_callback()
657   {
658     var data = data_buffer.join('');
659     var obj = null;
660     try {
661       obj = JSON.parse(data);
662     }
663     catch (err) {
664       callback(new Error('File ' + path + ' contains invalid JSON: ' + err), null);
665       return;
666     }
667     callback(null, obj);
668     return;
669   }
670 
671   read_stream.on('data', function(chunk) {
672     data_buffer.push(chunk);
673   });
674 
675   read_stream.on('error', function(error) {
676     read_stream.removeAllListeners('data');
677     read_stream.removeAllListeners('end');
678     read_stream.removeAllListeners('error');
679     callback(error, null);
680   });
681 
682   read_stream.on('end', end_callback);
683 };
