 18 var assert = require('assert');
 19 var lock_managers = {};
 20 
 21 /**
 22  * Manages multiple named locks.
 23  */
 24 function LockManager() {
 25   this.queues = {};
 26   this.locks = {};
 27 }
 28 
 29 /**
 30  * Asynchronously get a lock. The callback won't fire until the lock is
 31  * achieved. The callback is passed a 'release' callback which will release
 32  * the lock when called.
 33  *
 34  * @param {String} name The name of the lock.
 35  * @param {Function} cb A callback which takes as a single argument another
 36  * 'release' callback which will release the lock when called.
 37  */
 38 LockManager.prototype.with_lock = function(lockname, cb) {
 39   var self = this;
 40   function release() {
 41     if (self.queues[lockname].length === 0) {
 42       self.locks[lockname] = false;
 43       return;
 44     }
 45     else {
 46       process.nextTick(function() {
 47         var ncb = self.queues[lockname].shift();
 48         ncb(release);
 49       });
 50     }
 51   }
 52 
 53   if (!this.locks[lockname]) {
 54     this.locks[lockname] = true;
 55     if (!this.queues[lockname]) {
 56       this.queues[lockname] = [];
 57     }
 58     process.nextTick(function() {
 59       cb(release);
 60     });
 61   }
 62   else {
 63     this.queues[lockname].push(cb);
 64   }
 65 };
 66 
 67 /**
 68  * Get or instantiate a lock manager by name.
 69  */
 70 exports.get_lock_manager = function(manager) {
 71   if (!lock_managers[manager]) {
 72     lock_managers[manager] = new LockManager();
 73   }
 74   return lock_managers[manager];
 75 };
