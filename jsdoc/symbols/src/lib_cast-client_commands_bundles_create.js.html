 18 var sys = require('sys');
 19 var fs = require('fs');
 20 var path = require('path');
 21 
 22 var sprintf = require('extern/sprintf').sprintf;
 23 var async = require('extern/async');
 24 
 25 var tarball = require('util/tarball');
 26 var misc = require('util/misc');
 27 var lock = require('util/lock');
 28 var spinner = require('util/spinner');
 29 var dotfiles = require('util/client_dotfiles');
 30 var fsutil = require('util/fs');
 31 
 32 var upload_command = require('cast-client/commands/bundles/upload');
 33 
 34 var manifest = require('manifest/index');
 35 var manifest_constants = require('manifest/constants');
 36 
 37 /** Configuration options for create subcommand */
 38 exports.config = {
 39   'short_description': 'Create an save the application bundle',
 40   'long_description': 'Create an application bundle. If you don\'t specify the app or ' +
 41                       'bundle path, the current working directory is used.',
 42   'required_arguments' : [['version', 'Application version number']],
 43   'optional_arguments': [['apppath', 'Path to folder containing the cast.json file and your application files']],
 44   'options': [
 45     {
 46       names: ['--upload'],
 47       dest: 'upload',
 48       action: 'store_true',
 49       desc: 'Upload a bundle to the server after it is created'
 50     }
 51   ]
 52 };
 53 
 54 /**
 55  * Handler for creating a bundle.
 56  * @param {Object} args Command line arguments.
 57  * @param {String} args.version Application version number to create.
 58  * @param {String} [args.apppath=process.cwd()] Application path to operate on.
 59  */
 60 exports.handle_command = function(args)
 61 {
 62   var application_path = args.apppath || process.cwd();
 63   var manifest_file = path.join(application_path, manifest_constants.MANIFEST_FILENAME);
 64   var dot_cast_project;
 65   var bundle_dir_path;
 66   var bundle_name;
 67   var tarball_name;
 68 
 69   async.waterfall([
 70     // Check that user provided a valid version
 71     function(callback) {
 72       if (!misc.is_valid_bundle_version(args.version)) {
 73         return callback(new Error('Provided version string is invalid'));
 74       }
 75 
 76       callback();
 77     },
 78 
 79     // Check that the manifest file exists
 80     function(callback) {
 81       path.exists(manifest_file, function(exists) {
 82         if (!exists) {
 83           var msg = sprintf('Failed to find a manifest file %s in %s',
 84                       manifest_constants.MANIFEST_FILENAME,
 85                       application_path);
 86           return callback(new Error(msg));
 87         }
 88         return callback();
 89       });
 90     },
 91 
 92     // Validate the manifest
 93     function(callback) {
 94       manifest.validate_manifest(manifest_file, function(err, manifest_object) {
 95         if (err) {
 96           return callback(new Error('Manifest validation failed: ' + err.message));
 97         }
 98 
 99         bundle_name = misc.get_valid_bundle_name(manifest_object.name);
100         tarball_name = sprintf('%s.tar.gz', misc.get_full_bundle_name(bundle_name, args.version));
101         return callback();
102       });
103     },
104 
105     // Make sure the dot cast project directory exists
106     function(callback) {
107       dotfiles.ensure_dot_cast_project(application_path, callback);
108     },
109 
110     // Make sure the temp directory exists within it
111     function(dot_cast_project_path, callback) {
112       dot_cast_project = dot_cast_project_path;
113       bundle_dir_path = path.join(dot_cast_project_path, 'tmp');
114       fsutil.ensure_directory(bundle_dir_path, callback);
115     },
116 
117     // Get the lock
118     function(callback) {
119       var lock_path = path.join(bundle_dir_path, sprintf('%s.lock', tarball_name));
120       lock.with_lock(lock_path, function(err, release) {
121         if (err) {
122           err = new Error('Failed to acquire a lock, probably someone is already creating a bundle with the same name');
123         }
124         return callback(err, release);
125       });
126     },
127 
128     // Create the tarball
129     function(release, callback) {
130       var progress_spinner = spinner.spinner('Manifest validation succeeded, creating a bundle ');
131       var timeout_id = setInterval(function() {
132         progress_spinner.tick();
133       }, 100);
134       progress_spinner.start();
135 
136       tarball.create_tarball(application_path, bundle_dir_path, tarball_name, {
137         delete_if_exists: true,
138         exclude_pattern: path.basename(dot_cast_project)
139       },
140       function(err) {
141         release();
142         clearInterval(timeout_id);
143         progress_spinner.end();
144         return callback(err);
145       });
146     }
147   ],
148   function(err) {
149     if (err) {
150       sys.puts('Error: ' + err.message);
151     }
152     else if (args.upload) {
153       upload_command.handle_command({version: args.version});
154     }
155     else {
156       sys.puts('Bundle created');
157     }
158   });
159 };
