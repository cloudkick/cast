 94 /**
 95  * Connect to the agent endpoint.
 96  */
 97 Agent.prototype.connect_to_endpoint = function() {
 98   var self = this;
 99 
100   // @TODO: Add SSL support
101   this.connection = net.createConnection(endpoint_port, endpoint_host);
102 
103   var handle_error = function(err) {
104     log.info(sprintf('Connection to %s:%s failed, reason: %s', endpoint_host, endpoint_port, err.message));
105     self.clear_connect_timeout();
106 
107     self.increase_backoff_delay();
108     self.reconnect_to_endpoint();
109   };
110 
111   this.connection.on('error', handle_error);
112 
113   // Set connection timeout handler
114   this.connect_timeout_id = setTimeout(function() {
115     self.handle_connect_timeout();
116   }, constants.CONNECT_TIMEOUT);
117 
118   this.connection.on('connect', function() {
119     self.connection.setKeepAlive(true);
120     self.connection.setNoDelay(true);
121 
122     log.info(sprintf('Connected to the endpoint %s:%s', endpoint_host,
123                                                         endpoint_port));
124     // Clear the connect timeout, reset the connection delay timer and clear the restarting flag
125     self.clear_connect_timeout();
126     self.reset_connect_delay();
127     self.restarting = false;
128 
129     // Send the hello command
130     self.send_cmd_hello();
131 
132     // Set the ping interval
133     self.ping_interval_id = setInterval(function() {
134       self.send_cmd_ping();
135     }, constants.PING_INTERVAL);
136 
137     var remove_all_listeners = function() {
138       self.connection.removeAllListeners('data');
139       self.connection.removeAllListeners('end');
140       self.connection.removeAllListeners('close');
141     };
142 
143     var handle_data = function(data) {
144       self.handle_data(data.toString());
145     };
146 
147     var handle_end = function(err) {
148       self.connection.end();
149     };
150 
151     var handle_close = function() {
152       log.info('Endpoint closed the connection');
153 
154       // Remove all the listeners, reset ping counters and interval
155       self.connected = false;
156       remove_all_listeners();
157       self.reset_ping_counters_and_clear_interval();
158 
159       if (!self.restarting) {
160         // Increase backoff time delay and try to reconnect to the endpoint
161         self.increase_backoff_delay();
162         self.reconnect_to_endpoint();
163       }
164     };
165 
166     self.connection.on('data', handle_data);
167     self.connection.on('end', handle_end);
168     self.connection.on('close', handle_close);
169   });
170 };

172 /**
173  * Reconnect to the endpoint. If there is an active connection, destroy it.
174  *
175  * @param {boolean} True to reconnect immediately, otherwise wait connect_delay seconds.
176  */
177 Agent.prototype.reconnect_to_endpoint = function(now) {
178   var connect_delay;
179   var self = this;
180 
181   if (now === true) {
182     connect_delay = 1000;
183   }
184   else {
185     connect_delay = this.connect_delay;
186   }
187 
188   if (this.connection) {
189     // If there is an active connection, kill it
190     this.connection.destroy();
191   }
192 
193   log.info(sprintf('Reconnecting in %s seconds...', (Math.round(connect_delay / 1000))));
194 
195   this.connect_timeout_id = setTimeout(function() {
196     self.connect_to_endpoint();
197   }, connect_delay);
198 };

200 /**
201  * Log the command.
202  *
203  * @param  {String} command Command name.
204  * @param  {String} type Command type ('incoming' or 'outgoing')
205  * @return {Array} command_arguments Command arguments
206  */
207 Agent.prototype.log_command = function(command, type, command_arguments) {
208   var command_data, time, i, argument, value, log_string;
209 
210   if (!misc.in_array(type, ['incoming', 'outgoing'])) {
211     throw new Errorf('Invalid command type: %s (valid types are incoming)' +
212                     ' and outgoing', type);
213   }
214 
215   command_data = (type === 'incoming') ? constants.INCOMING_COMMAND_MAPPING[command] :
216                                         constants.OUTGOING_COMMAND_MAPPING[command];
217 
218   time = new Date();
219 
220   var ret = [];
221   for (i = 0; i < command_arguments.length; i++) {
222     argument = command_data.args[i];
223 
224     if (misc.in_array(argument, command_data.dont_log)) {
225       // This argument is not logged
226       continue;
227     }
228 
229     value = command_arguments[i];
230 
231     ret.push(sprintf('%s = %s', argument, value));
232   }
233 
234   ret = ret.join(', ');
235   log_string = sprintf('%s %s %s (%s): %s', ((type === 'incoming') ? '<-' : '->'), type, command, time,
236                        ret);
237   log.info(log_string);
238 };

240 /**
241  * Increase the connection backoff time delay.
242  */
243 Agent.prototype.increase_backoff_delay = function() {
244   this.connect_delay += Math.ceil(Math.random() * 2000);
245 };

247 /**
248  * If not connected, increase the backoff time delay and try to re-connect.
249  */
250 Agent.prototype.handle_connect_timeout = function() {
251   if (!this.connected) {
252     log.info(sprintf('Connection to %s:%s failed, reason: timeout', endpoint_host, endpoint_port));
253 
254     if (this.connection) {
255       this.connection.destroy();
256     }
257 
258     this.increase_backoff_delay();
259     this.reconnect_to_endpoint();
260   }
261 };

263 /*
264  * Clear the connection timeout handler and set the connected property to true.
265  *
266  * This function is called after the connection is successfully established or if the connection fails
267  * because of a different reason then timeout (connection refused, etc.)
268  */
269 Agent.prototype.clear_connect_timeout = function() {
270   if (!this.connected && this.connect_timeout_id) {
271     this.connected = true;
272     clearTimeout(this.connect_timeout_id);
273   }
274 };

296 /*
297  * Send command to the endpoint.
298  *
299  * @param {String} command Command name.
300  * @param {Array} command_arguments Command arguments.
301  */
302 Agent.prototype.send_command = function(command, command_arguments) {
303   var command_data, arguments_count, sprintf_arguments, command_string;
304 
305   if (!misc.in_array(command, Object.keys(constants.OUTGOING_COMMAND_MAPPING))) {
306     log.err(sprintf('Invalid command: %s', command));
307     return;
308   }
309 
310   command_data = constants.OUTGOING_COMMAND_MAPPING[command];
311 
312   arguments_count = command_arguments.length;
313   if (arguments_count !== command_data.arg_count) {
314     log.err(sprintf('Invalid number of arguments (%s) for command %s', arguments_count, command));
315     return;
316   }
317 
318   sprintf_arguments = command_arguments.slice();
319   sprintf_arguments.unshift(command_data.command);
320   command_string = sprintf.apply(null, sprintf_arguments);
321   this.connection.write(command_string + '\n');
322 
323   this.log_command(command, 'outgoing', command_arguments);
324 };

326 /*
327  * Handler for incoming data.
328  *
329  * @param {String} data Incoming data (protocol is line orinted, so each command / message is on a
330  *                      separate line).
331  */
332 Agent.prototype.handle_data = function(data) {
333   var i, line, next_line, line_split, command;
334   var command_data, command_arguments;
335   var data_split = data.split('\n');
336   var data_split_count = data_split.length;
337 
338   for (i = 0; i < data_split_count; i++) {
339     line = misc.trim(data_split[i]);
340 
341     if (line === '') {
342       continue;
343     }
344 
345     line_split = line.split(' ');
346     command = line_split[0];
347 
348     if (!misc.in_array(command, Object.keys(constants.INCOMING_COMMAND_MAPPING))) {
349       log.err(sprintf('Received invalid line: %s', line));
350       continue;
351     }
352 
353     // Special case for a run_check command - payload is on a separate line (separated by a \r\n)
354     if (command === 'run_check') {
355       // Next line is a check payload
356       next_line = data_split[i + 1];
357 
358       if ((i + 1) >= (data_split_count - 1) || next_line.charAt(0) !== '{') {
359         log.err('Missing payload for the run_check command');
360         continue;
361       }
362 
363       next_line = misc.trim(next_line);
364       line = sprintf('%s %s', line, next_line);
365       i++;
366     }
367 
368     command_data = constants.INCOMING_COMMAND_MAPPING[command];
369 
370     if (command_data.args_split) {
371       // Only split commands where args_split property equals true (some commands receive a single string
372       // argument which can contain spaces and commands like this are not split).
373       line_split = line.split(' ');
374       command_arguments = line_split.splice(1);
375     }
376     else {
377       command_arguments = [ misc.trim(line.replace(command, '')) ];
378     }
379 
380     this.handle_command(command, command_arguments);
381   }
382 };

384 /*
385  * Run appropriate function for the incoming command.
386  *
387  * @param {String} command Command name.
388  * @param {Array} command_arguments Command arguments.
389  */
390 Agent.prototype.handle_command = function(command, command_arguments) {
391   var argument_count, command_data;
392 
393   argument_count = command_arguments.length;
394   command_data = constants.INCOMING_COMMAND_MAPPING[command];
395 
396   if (argument_count !== command_data.arg_count) {
397     log.err('Invalid number for arguments (%s) for command %s', argument_count, command);
398     return;
399   }
400 
401   Agent.prototype.log_command(command, 'incoming', command_arguments);
402   command_data.handler.apply(this, command_arguments);
403 };

417 /*
418  * Send ping command to the endpoint.
419  */
420 Agent.prototype.send_cmd_ping = function() {
421   var current_time = misc.get_unix_timestamp();
422   var time_diff = current_time - this.last_ping_time;
423 
424   if ((this.ping_sent_count - this.pong_got_count) >= constants.MAX_MISSED_PONGS) {
425     log.info(sprintf('Failed to receive %s ping replies, restarting...',
426              constants.MAX_MISSED_PONGS));
427     this.reconnect_to_endpoint();
428     return;
429   }
430 
431   this.send_command('ping', [ misc.get_unix_timestamp() ]);
432   this.last_ping_time = current_time;
433   this.ping_sent_count++;
434 };

460 /*
461  * Run a check.
462  *
463  * @param {String} token Check token.
464  * @param {String} check_name Check name.
465  * @param {Number} payload_len Payload length.
466  * @param {String} payload Check payload (JSON encoded string)
467  */
468 Agent.prototype.handle_cmd_run_check = function(token, check_name, payload_len, payload) {
469   var self = this;
470   var check_arguments, parsed_payload, check_worker;
471 
472   try {
473     parsed_payload = JSON.parse(payload);
474   }
475   catch (err) {
476     log.err(sprintf('Failed to parse "%s" check payload: %s', check_name, err.toString()));
477     return;
478   }
479 
480   // @TODO: Should probably limit the maximum number of active workers
481   check_worker = worker.getWorker(constants.WORKER_CHECK_SCRIPT, {'timeout': 500});
482   check_arguments = { 'check_name': check_name, 'payload': parsed_payload };
483 
484   check_worker.postMessage(check_arguments);
485 
486   var terminate_worker = function() {
487     if (!check_worker.isTerminated()) {
488       check_worker.terminate();
489     }
490   };
491 
492   var on_result = function(result) {
493     // @TODO: send check result
494     terminate_worker();
495     // self.send_command('result', []);
496   };
497 
498   var on_error = function(err) {
499     // @TODO: send error
500     terminate_worker();
501     // self.send_command('result', []);
502   };
503 
504   check_worker.on('message', on_result);
505   check_worker.on('error', on_error);
506 };
