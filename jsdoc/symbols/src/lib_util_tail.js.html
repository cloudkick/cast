 27 /**
 28  * Return bytes_to_read bytes of data from the end of the file and optionally calls the provided callback with the new data
 29  * as the file is modified and new data is available.
 30  *
 31  * @param  {String} file_path Absolute path to a file.
 32  * @param  {Number} bytes_to_read How many bytes from the end of the file to read.
 33  * @param  {Boolean} follow If true, the provided callback will be continuously called (until un-subscribed) with the new data
 34  *                          as the file is modified and new data is available (aka tail -F).
 35  * @param  {Function} callback Callback which is called with an error as the first argument, data with as the second one
 36  *                              and the un-subscribe function as the third one.
 37  */
 38 var tail_file = function(file_path, bytes_to_read, follow, callback) {
 39   var read_file, unsubscribe;
 40   var stat;
 41 
 42   var watch_file_listener = function(curr, prev) {
 43     var start, end;
 44 
 45     var inode_curr = curr.ino;
 46     var inode_prev = prev.ino;
 47 
 48     var size_curr = curr.size;
 49     var size_prev = prev.size;
 50     var size_diff = size_curr - size_prev;
 51 
 52     var mtime_curr = curr.mtime.valueOf();
 53     var mtime_prev = prev.mtime.valueOf();
 54 
 55     if ((inode_curr !== inode_prev) || (size_diff < 0) || (size_diff === 0 && mtime_curr !== mtime_prev)) {
 56       // Log file was rotated or truncated
 57       start = 0;
 58       end = (bytes_to_read > size_curr) ? size_curr : bytes_to_read;
 59     }
 60     else if (size_diff === 0) {
 61       // No change in the file size (probably file ownership or permissions were changed), ignore this event
 62       return;
 63     }
 64     else {
 65       start = size_prev;
 66       end = size_curr;
 67     }
 68 
 69     read_file(start, end);
 70   };
 71 
 72   if (follow) {
 73     stat = fs.watchFile(file_path, watch_file_listener);
 74   }
 75 
 76   unsubscribe = function() {
 77     stat.removeListener('change', watch_file_listener);
 78 
 79     if (Object.keys(stat._events).length === 0) {
 80       // No more listeners left, unwatch the file
 81       fs.unwatchFile(file_path);
 82     }
 83   };
 84 
 85   /**
 86    * Read a file.
 87    *
 88    * @param {Number} start Start offset.
 89    * @param {Number} end End offset.
 90    */
 91   read_file = function(start, end) {
 92     var file_stream = fs.createReadStream(file_path, {start: start, end: end});
 93 
 94     file_stream.on('data', function(data) {
 95       callback(null, data, unsubscribe);
 96     });
 97 
 98     file_stream.on('error', function(error) {
 99       if (follow) {
100         unsubscribe();
101       }
102       callback(error, null, null);
103     });
104   };
105 
106   fs.stat(file_path, function(error, stats) {
107     var start, end;
108 
109     if (error) {
110       return callback(error, null, null);
111     }
112 
113     start = (bytes_to_read >= stats.size) ? 0 : (stats.size - bytes_to_read);
114     end = stats.size;
115 
116     read_file(start, end);
117   });
118 };
119 
120 exports.tail_file = tail_file;
