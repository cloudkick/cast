 18 var fs = require('fs');
 19 var sys = require('sys');
 20 
 21 var async = require('extern/async');
 22 var sprintf = require('extern/sprintf').sprintf;
 23 var Errorf = require('util/misc').Errorf;
 24 
 25 var dotfiles = require('util/client_dotfiles');
 26 
 27 exports.config = {
 28   short_description: 'Set a default remote',
 29   long_description: 'Set a default remote which will be used with all the commands',
 30   required_arguments: [
 31     ['name', 'Remote name']
 32   ],
 33   optional_arguments: []
 34 };
 35 
 36 exports.handle_command = function(args) {
 37   async.waterfall([
 38     // Get or create the current remotes object
 39     function(callback) {
 40       dotfiles.get_remotes(callback);
 41     },
 42 
 43     function(remotes, callback) {
 44       var remote;
 45 
 46       if (Object.keys(remotes).length === 0) {
 47         callback(new Error('No remotes exist yet. You can add one by calling "remotes add" command'));
 48         return;
 49       }
 50 
 51       // Check if remote with this exists
 52       if (!remotes.hasOwnProperty(args.name)) {
 53         callback(new Errorf('Remote with name "%s" does not exist', args.name));
 54         return;
 55       }
 56 
 57       for (var remote_name in remotes) {
 58         if (remotes.hasOwnProperty(remote_name)) {
 59           remote = remotes[remote_name];
 60 
 61           if (remote_name === args.name) {
 62             remote.is_default = true;
 63           }
 64           else {
 65             remote.is_default = false;
 66           }
 67         }
 68       }
 69 
 70       callback(null, remotes);
 71     },
 72 
 73     // Save the remotes
 74     function(remotes, callback) {
 75       var fstream = fs.createWriteStream(dotfiles.dot_cast_remotes_path);
 76 
 77       fstream.write(JSON.stringify(remotes, null, 4));
 78       fstream.end();
 79 
 80       fstream.on('close', function() {
 81         return callback();
 82       });
 83 
 84       fstream.on('error', function(err) {
 85         fstream.removeAllListeners('end');
 86         return callback(new Errorf('Error writing remotes file: %s', err));
 87       });
 88     }
 89   ],
 90 
 91   function(err) {
 92     if (err) {
 93       sys.puts('Error: ' + err.message);
 94     }
 95     else {
 96       sys.puts(sprintf('Remote %s set as default', args.name));
 97     }
 98   });
 99 };
