 18 var sys = require('sys');
 19 var fs = require('fs');
 20 var crypto = require('crypto');
 21 var ps = require('util/pubsub');
 22 var config = require('util/config');
 23 var clutch = require('extern/clutch');
 24 var http = require('util/http');
 25 var apps = ['bundles', 'services', 'facts', 'health', 'instances'];
 26 
 27 function index(req, res) {
 28     res.writeHead(200, {'Content-Type': 'text-plain'});
 29     apps.forEach(function(v) {
 30       res.write(v + '\n');
 31     });
 32     res.end();
 33 }
 34 
 35 /* Used by test cases that just want an instance of the server, not yet running */
 36 exports._serverOnly = function() {
 37   var r = ['GET /$', index];
 38   var urls = [];
 39   apps.forEach(function(v) {
 40     urls.push(['* /' + v, require('services/http/' + v).urls]);
 41   });
 42 
 43   urls.push(r);
 44 
 45   var routes = clutch.route404(urls);
 46 
 47   var server = http.server(routes);
 48 
 49   return server;
 50 };
 51 
 52 exports.load = function() {
 53   var conf = config.get();
 54   var server = exports._serverOnly();
 55   ps.ensure(ps.AGENT_STATE_START, function() {
 56     /*if (conf.ssl_enabled) {
 57       var key = fs.readFileSync(conf.ssl_key).toString();
 58       var cert = fs.readFileSync(conf.ssl_cert).toString();
 59       var sslcontext = crypto.createCredentials({method: 'TLSv1_server_method', key: key, cert: cert});
 60       sslcontext.context.setCiphers(conf.ssl_ciphers);
 61       server.setSecure(sslcontext);
 62     }*/
 63     server.listen(conf.port, '0.0.0.0');
 64   });
 65 
 66   ps.on(ps.AGENT_STATE_STOP, function() {
 67     server.close();
 68   });
 69 };
