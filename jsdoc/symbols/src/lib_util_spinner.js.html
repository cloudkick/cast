 18 var sys = require('sys');
 19 var sprintf = require('../extern/sprintf').sprintf;
 20 var dots = ['|', '/', '-', '\\'];
 21 var stdio = process.binding('stdio');
 22 
 23 /**
 24  * Display a prompt followed by a little spinning character. Call tick
 25  * repeatedly to update.
 26  *
 27  * @param {String} prompter The prompt string.
 28  */
 29 function Spinner(prompter) {
 30   this.ticks = 0;
 31   this.promptstr = prompter;
 32 }
 33 
 34 /**
 35  * Start the spinner.
 36  */
 37 Spinner.prototype.start = function() {
 38   process.stdout.write('\n' + this.promptstr + '  ');
 39   this.tick();
 40 };
 41 
 42 /**
 43  * Update the spinner with the given character, internal use only.
 44  *
 45  * @param {String} c  The character to set.
 46  */
 47 Spinner.prototype.update = function(c) {
 48   process.stdout.write('\b' + c);
 49 };
 50 
 51 /**
 52  * Call this repeatedly to spin the spinner.
 53  */
 54 Spinner.prototype.tick = function() {
 55   this.ticks++;
 56   var c = dots[this.ticks % dots.length];
 57   this.update(c);
 58 };
 59 
 60 /**
 61  * End the spinner.
 62  */
 63 Spinner.prototype.end = function() {
 64   process.stdout.write('\n');
 65 };
 66 
 67 /**
 68  * Show a prompt followed by an updating percentage value.
 69  *
 70  * @param {String} prompter The prompt string.
 71  * @param {Number} max  The value to use for 100%.
 72  */
 73 function PercentSpinner(prompter, max) {
 74   this.max = max;
 75   this.current = 0;
 76   this.promptstr = prompter;
 77 }
 78 
 79 /**
 80  * Start the spinner.
 81  */
 82 PercentSpinner.prototype.start = function() {
 83   process.stdout.write('\n' + this.promptstr + '       ');
 84   this.tick(0);
 85 };
 86 
 87 /**
 88  * Set the given percentage string, internal use only.
 89  *
 90  * @param {String} c  The percentage string to set.
 91  */
 92 PercentSpinner.prototype.update = function(c) {
 93   process.stdout.write('\b\b\b\b\b' + c);
 94 };
 95 
 96 /**
 97  * Update the spinner. Call this repeatedly with new values as they become
 98  * available.
 99  *
100  * @param {Number} value  The current value to set the spinner to.
101  */
102 PercentSpinner.prototype.tick = function(value)
103 {
104   this.current = value;
105   var p = (this.current / this.max) * 100;
106   this.update(sprintf('%4d%%', p));
107 };
108 
109 /**
110  * End the spinner.
111  */
112 PercentSpinner.prototype.end = function() {
113   process.stdout.write('\n');
114 };
115 
116 /**
117  * Show prompt, and on the next line an updating percentage value and bar.
118  *
119  * @param {String} prompter The prompt string.
120  * @param {Number} max  The value to use for 100%.
121  */
122 function PercentBarSpinner(prompter, max) {
123   this.current = 0;
124   this.max = max;
125   this.promptstr = prompter;
126 }
127 
128 /**
129  * Start the spinner.
130  */
131 PercentBarSpinner.prototype.start = function() {
132   process.stdout.write(this.promptstr + '\n');
133   this.tick(0);
134 };
135 
136 /**
137  * Rewrite the current line, internal use only.
138  *
139  * @param {String} line The line to update to.
140  */
141 PercentBarSpinner.prototype.update = function(line) {
142   process.stdout.write('\r' + line);
143 };
144 
145 /**
146  * Update the spinner value.
147  *
148  * @param {Number} value  The new value to use.
149  */
150 PercentBarSpinner.prototype.tick = function(value) {
151   this.current = value;
152   var bar = this.barstr();
153   var val = this.percent();
154   this.update(sprintf(' %4d%%', val) + ' [' + this.barstr() + '] ');
155 };
156 
157 /**
158  * End the spinner.
159  */
160 PercentBarSpinner.prototype.end = function() {
161   process.stdout.write('\n');
162 };
163 
164 /**
165  * Get the width to use for the inside of the percentage bar.
166  *
167  * @return The width in columns.
168  */
169 PercentBarSpinner.prototype.width = function() {
170   return stdio.getColumns() - 10;
171 };
172 
173 /**
174  * Get the current percentage value.
175  *
176  * @return The percentage value.
177  */
178 PercentBarSpinner.prototype.percent = function() {
179   return (this.current / this.max) * 100;
180 };
181 
182 /**
183  * Get the string to use for the inside of the percentage bar.
184  *
185  * @return The percentage bar string.
186  */
187 PercentBarSpinner.prototype.barstr = function() {
188   var width = this.width();
189   var p = Math.floor((this.current / this.max) * width);
190   if (p >= width) {
191     return new Array(width + 1).join('=');
192   }
193   else {
194     var offset = p === 0 ? -1 : 0;
195     return new Array(p).join('=') + '>' + new Array(width - p + offset + 1).join(' ');
196   }
197 };
198 
199 exports.spinner = function(prompter) {
200   return new Spinner(prompter);
201 };
202 
203 exports.percent = function(prompter, max) {
204   return new PercentSpinner(prompter, max);
205 };
206 
207 exports.percentbar = function(prompter, max) {
208   return new PercentBarSpinner(prompter, max);
209 };
