 18 var async = require('extern/async');
 19 
 20 var misc = require('util/misc');
 21 
 22 var norris = require('norris');
 23 
 24 /*
 25  * Checks if the applications meets the requirements (based on norris facts).
 26  *
 27  * @param {Function} callback Callback which is called with a possible error as the first argument and true as the
 28  *                            second one if the requirements are met.
 29  */
 30 exports.meets_requirements = function(requirements, callback) {
 31   norris.get(function(facts) {
 32 
 33     if (Object.keys(facts).length === 0) {
 34       callback(new Error('No facts available'));
 35     }
 36 
 37     async.forEach(Object.keys(requirements), function(key, callback) {
 38       var requirement, value, comparator, error_message, comparator_function;
 39 
 40       if (!misc.in_array(key, Object.keys(facts))) {
 41         callback(new Error('Fact for ' + key + ' is not available.'));
 42         return;
 43       }
 44 
 45       requirement = requirements[key];
 46       value = requirement[0];
 47       comparator_function = requirement[1];
 48       error_message = requirement[2];
 49 
 50       if (!comparator_function(value, facts[key])) {
 51         callback(new Error(error_message));
 52         return;
 53       }
 54 
 55       callback();
 56     },
 57 
 58     function(err) {
 59       if (err) {
 60         callback(new Error('Requirements not met: ' + err.message));
 61         return;
 62       }
 63 
 64       callback(null, true);
 65     });
 66   });
 67 };

 69 /* General requirements comparator functions */
 70 
 71 /*
 72  * Return true if ver_b <= ver_a, false otherwise.
 73  *
 74  * @param {String|Object} ver_a Version string (e.g. 0.1.1) or Object with keys: major, minor, patch
 75  * @param {String|Object} ver_b Version string (e.g. 0.1.1) or Object with keys: major, minor, patch
 76  *
 77  */
 78 exports.compare_versions = function(ver_a, ver_b) {
 79   function get_version_as_object(version) {
 80     var splitted;
 81 
 82     if (typeof(version) === 'string') {
 83       splitted = version.split('.');
 84 
 85       version = {};
 86       version.major = splitted[0];
 87       version.minor = splitted[1];
 88       version.patch = splitted[2];
 89     }
 90 
 91     return version;
 92   }
 93 
 94   ver_a = get_version_as_object(ver_a);
 95   ver_b = get_version_as_object(ver_b);
 96 
 97   if (ver_b.major === ver_a.major) {
 98     if (ver_b.minor === ver_a.minor) {
 99       if (ver_b.patch < ver_a.patch) {
100         return false;
101       }
102       else {
103         return true;
104       }
105     }
106     else if (ver_b.minor < ver_a.minor) {
107       return false;
108     }
109     else {
110       return true;
111     }
112   }
113   else if (ver_b.major < ver_a.major) {
114     return false;
115   }
116   else {
117     return true;
118   }
119 };

121 /*
122  * Return true if the fact is defined (not null, false or undefined), false otherwise.
123  *
124  * @param {Object} expected Expected value (in this case we don't care about this variable so it is ignored)
125  * @param {String|Object} value The value.
126  *
127  */
128 exports.is_defined = function(expected, value) {
129   if (value === null || value === undefined || value === false) {
130     return false;
131   }
132 
133   return true;
134 };
