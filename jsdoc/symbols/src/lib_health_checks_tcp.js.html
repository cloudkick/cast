 18 var sys = require('sys');
 19 var net = require('net');
 20 
 21 var sprintf = require('extern/sprintf').sprintf;
 22 var log = require('util/log');
 23 
 24 var Check = require('health').Check;
 25 var CheckResult = require('health').CheckResult;
 26 var CheckStatus = require('health').CheckStatus;
 27 
 28 var config = {
 29   'name': 'TCP check',
 30   'required_arguments': ['ip_address', 'port', 'type'],
 31   'optional_arguments': ['match_value', 'command', 'connect_timeout', 'idle_timeout'],
 32   'default_values': {
 33     'connect_timeout': 5000,
 34     'idle_timeout': 3000
 35   },
 36   'types': {
 37     CONNECTION_CHECK: 0,
 38     RESPONSE_REGEX_MATCH: 1
 39   }
 40 };
 41 
 42 function TCPCheck(check_arguments) {
 43   Check.call(this, config.name, config.required_arguments,
 44              config.optional_arguments, config.default_values, check_arguments);
 45 }
 46 
 47 // Inherit from the base Check class
 48 sys.inherits(TCPCheck, Check);
 49 
 50 TCPCheck.prototype.run = function(callback) {
 51   var connect_timeout_id;
 52   var got_response = false;
 53 
 54   log.info(sprintf('Running TCP check (IP: %s, port: %s)',
 55                    this.check_arguments.ip_address, this.check_arguments.port));
 56 
 57   var self = this;
 58   var result = new CheckResult();
 59 
 60   var connection = net.createConnection(self.check_arguments.port, self.check_arguments.ip_address);
 61 
 62   var connect_timeout = function() {
 63     // Node doesn't allow you to set connection timeout, so the underlying file description
 64     // is manually destroyed if the connection cannot be established after connect_timeout
 65     // milliseconds.
 66     if (!got_response) {
 67       got_response = true;
 68 
 69       result.status = CheckStatus.ERROR;
 70       result.details = sprintf('Check failed - connection timed out after %s seconds',
 71                                (self.check_arguments.connect_timeout / 1000));
 72       self.add_result(result, callback);
 73 
 74       connection.destroy();
 75     }
 76   };
 77 
 78   var clear_connect_timeout = function() {
 79     if (!got_response && connect_timeout_id) {
 80       got_response = true;
 81       clearTimeout(connect_timeout_id);
 82     }
 83   };
 84 
 85   connect_timeout_id = setTimeout(connect_timeout, self.check_arguments.connect_timeout);
 86 
 87   connection.on('connect', function() {
 88     clear_connect_timeout();
 89     connection.setTimeout(self.check_arguments.idle_timeout);
 90 
 91     if (self.check_arguments.type === config.types.CONNECTION_CHECK) {
 92       result.status = CheckStatus.SUCCESS;
 93       result.details = sprintf('Successfully established connection to IP %s port %s',
 94                                self.check_arguments.ip_address,
 95                                self.check_arguments.port);
 96 
 97       self.add_result(result, callback);
 98       connection.end();
 99     }
100     else if (self.check_arguments.type === config.types.RESPONSE_REGEX_MATCH) {
101       var data_buffer = [];
102 
103       if (self.check_arguments.command) {
104         connection.end(self.check_arguments.command);
105       }
106 
107       connection.on('data', function(chunk) {
108         data_buffer.push(chunk);
109       });
110 
111       connection.on('end', function() {
112         var body = data_buffer.join('');
113 
114         if (body.match(self.check_arguments.match_value)) {
115           result.status = CheckStatus.SUCCESS;
116           result.details = sprintf('The response body matched the regular expression: %s',
117                                   self.check_arguments.match_value.toString());
118         }
119         else {
120           result.status = CheckStatus.ERROR;
121           result.details = sprintf('The response body didn\'t match the regular expression: %s',
122                                    self.check_arguments.match_value.toString());
123         }
124 
125         self.add_result(result, callback);
126         connection.end();
127       });
128     }
129   });
130 
131   connection.on('timeout', function() {
132     connection.end();
133   });
134 
135   connection.on('error', function(exception) {
136     clear_connect_timeout();
137 
138     if (self.last_run_date !== result.date) {
139       result.status = CheckStatus.ERROR;
140       result.details = sprintf('Check failed - returned exception: %s', exception);
141 
142       self.add_result(result, callback);
143     }
144   });
145 };
146 
147 TCPCheck.prototype.format_arguments = function(required_arguments, optional_arguments,
148                                                default_values, check_arguments)
149 {
150   var formatted_arguments = Check.prototype.format_arguments.call(this, required_arguments,
151                                                                   optional_arguments, default_values,
152                                                                   check_arguments);
153 
154   if (formatted_arguments.type !== config.types.CONNECTION_CHECK &&
155       formatted_arguments.type !== config.types.RESPONSE_REGEX_MATCH) {
156     throw new Error(sprintf('Invalid check type: %s', formatted_arguments.type));
157   }
158 
159   if (formatted_arguments.type === config.types.RESPONSE_REGEX_MATCH) {
160     formatted_arguments.match_value = new RegExp(formatted_arguments.match_value);
161   }
162 
163   return formatted_arguments;
164 };
165 
166 exports.config = config;
167 exports.TCPCheck = TCPCheck;
