 18 var exec = require('child_process').exec;
 19 
 20 var async = require('extern/async');
 21 var sprintf = require('extern/sprintf').sprintf;
 22 
 23 var trim = require('util/misc').trim;
 24 
 25 exports.get = function(done) {
 26   var check_version = function(tar_binary, callback) {
 27     exec(sprintf('%s --version', tar_binary), function(err, stdout, stderr) {
 28       if (err) {
 29         callback(false);
 30         return;
 31       }
 32 
 33       if (stdout.toLowerCase().indexOf('gnu tar') === -1) {
 34         callback(false);
 35         return;
 36       }
 37 
 38       callback(true);
 39     });
 40   };
 41 
 42   exec('which gnutar gtar tar', function(err, stdout, stderr) {
 43     var o = trim(stdout).split(/\r\n|\r|\n/);
 44 
 45     async.filterSeries(o, check_version, function(results) {
 46       var gnutar_binary;
 47 
 48       if (!results) {
 49         gnutar_binary = null;
 50       }
 51       else {
 52         gnutar_binary = results[0];
 53       }
 54 
 55       done({gnutar: gnutar_binary});
 56     });
 57   });
 58 };
