 30 /*
 31  * Return an array of available instances for the given bundle.
 32  *
 33  * @param {String} bundle_name Bundle name as returned by the get_valid_bundle_name function
 34  * @param {String} bundle_version Bundle version string or 'all' to retrieve instances for all the
 35  *                                bundle versions
 36  * @param {Function} callback Callback which is called with a possible error as the first
 37  *                            argument and an array of available instances sorted by instance
 38  *                            number as the second one (each item is a tuple of instance
 39  *                            name and instance number)
 40  */
 41 exports.get_available_instances = function(bundle_name, bundle_version, callback)
 42 {
 43   var sorted, match_regex;
 44   var instance_application_path = path.join(config.get().app_dir, bundle_name);
 45 
 46   if (bundle_version === 'all') {
 47     match_regex = /.*?-[0-9]+/;
 48   }
 49   else {
 50     match_regex = sprintf('.*?@%s-[0-9]+', misc.escape_regexp_string(bundle_version));
 51   }
 52 
 53   fsutil.get_matching_files(instance_application_path, match_regex, false, function(error, files) {
 54     if (error) {
 55       return callback(error);
 56     }
 57 
 58     async.map(files, function(file, callback) {
 59       var splitted = file.split('-');
 60 
 61       callback(null, [splitted[0], parseInt(splitted[1], 10)]);
 62     },
 63 
 64     function(error, results) {
 65       if (error) {
 66         return callback(error);
 67       }
 68 
 69       results.sort(function(a, b) {
 70         if (a[1] === b[1]) {
 71           return (a[0].localeCompare(b[0]) < 0) ? -1 : 1;
 72         }
 73         else {
 74           return a[1] < b[1] ? -1 : 1;
 75         }
 76       });
 77 
 78       callback(null, results);
 79     });
 80   });
 81 };

 83 /*
 84  * This function cleans up everything related to a specific instance and should only be called
 85  * if the instance creation fails.
 86  *
 87  * @param {String} instance_name Instance name as returned by misc.get_valid_instance_name function
 88  * @param {string} instance_path Instance path
 89  * @param {string} instance_data_path Instance data path
 90  * @param {Function} callback Callback which is called when the cleanup process has finished
 91  */
 92 exports.cleanup = function(instance_name, instance_path, instance_data_path, callback) {
 93   var manager = service_management.get_default_manager().get_manager();
 94   var ops = [
 95     async.apply(flowctrl.call_ignoring_error, fsutil.rmtree, null, instance_path),
 96     async.apply(flowctrl.call_ignoring_error, fsutil.rmtree, null, instance_data_path)
 97   ];
 98 
 99   manager.get_service(instance_name, function (err, service) {
100     if (!err) {
101       ops.push(async.apply(flowctrl.call_ignoring_error, service.destroy, service));
102     }
103 
104     async.parallel(ops,
105 
106     function(err) {
107       callback();
108     });
109   });
110 };

112 /**
113  * Check if an instance exists (a.k.a. instance application directory exists).
114  *
115  * @param {String} bundle_name_full Bundle name as returned by the get_full_bundle_name function.
116  * @param {int} instance_number Instance number.
117  * @param {Function} callback Callback which is called a with possible error as the first argument and true
118  *                            as the second one if the instance exists, false otherwise.
119  */
120 exports.instance_exists = function(bundle_name_full, instance_number, callback) {
121   var instance_path;
122 
123   try {
124     instance_path = misc.get_instance_name_and_path(bundle_name_full, instance_number)[1];
125   }
126   catch (err) {
127     callback(err);
128     return;
129   }
130 
131   path.exists(instance_path, function(exists) {
132     callback(null, exists);
133   });
134 };
