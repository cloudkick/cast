 18 var sprintf = require('extern/sprintf').sprintf;
 19 
 20 var misc = require('util/misc');
 21 var config = require('util/config').get().runit;
 22 
 23 var manifest_constants = require('manifest/constants');
 24 
 25 var BASE_TEMPLATE = {
 26   run: '#!/bin/bash\n# Stub run file\nexit 0',
 27   finish: '#!/bin/bash\n# Stubbed out finish file\nexit 0;',
 28   down: "# Down file so it doesn't start up automagically",
 29   log: {
 30     'run': sprintf('#!/bin/bash\nexec chpst -u%s svlogd -tt ./%s', config.svlogd_daemon_user, config.log_directory),
 31     'config': sprintf('s%s\nn%s', config.max_log_size, config.max_log_num),
 32     main: {}
 33   }
 34 };
 35 
 36 var get_application_template = function(service_name, service_path, entry_file, application_type, callback)
 37 {
 38   var get_runit_template = null;
 39   if (!misc.in_array(application_type, manifest_constants.APPLICATION_TYPES)) {
 40     return callback(new Error(sprintf('Invalid application type %s', application_type)));
 41   }
 42 
 43   try {
 44     get_runit_template = require(sprintf('service_management/runit/templates/%s', application_type)).get_template;
 45   }
 46   catch (error) {
 47     return callback(new Error(sprintf('Cannot load template module for application type %s', application_type)));
 48   }
 49 
 50   get_runit_template(service_name, service_path, entry_file, function(error, template) {
 51     if (error) {
 52       return callback(error);
 53     }
 54 
 55     template = misc.merge(BASE_TEMPLATE, template);
 56 
 57     callback(null, template);
 58   });
 59 };
 60 
 61 exports.get_application_template = get_application_template;
