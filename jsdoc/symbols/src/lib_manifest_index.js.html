 18 var path = require('path');
 19 var fs = require('fs');
 20 
 21 var async = require('extern/async');
 22 var sprintf = require('extern/sprintf').sprintf;
 23 
 24 var misc = require('util/misc');
 25 var ufs = require('util/fs');
 26 var constants = require('manifest/constants');
 27 var validators = require('manifest/validators');
 28 
 29 /**
 30  * Custom error class containing information about a invalid field
 31  *
 32  * @param {String} field Field name.
 33  * @param {String} message Field error message.
 34  *
 35  * @constructor
 36  */
 37 function InvalidFieldError(field, message) {
 38   this.name = 'InvalidFieldError';
 39   this.field = field;
 40   this.field_message = message;
 41   this.message = sprintf('Field %s is not valid (%s)', this.field, this.field_message);
 42 }
 43 
 44 /**
 45  * Validate that the manifest file is valid.
 46  *
 47  * @param {String} manifest_path Absolute path to the manifest file.
 48  * @param {Function} callback Callback which is called with the error as the first argument if the validation fails,
 49  *                            with manifest_object as the second argument otherwise.
 50  */
 51 exports.validate_manifest = function(manifest_path, callback) {
 52   exports.get_manifest_data_as_object(manifest_path, true, function(error, manifest_object) {
 53     if (error) {
 54       return callback(error, null);
 55     }
 56 
 57     var required_fields = [];
 58     var optional_fields = [];
 59     var manifest_fields = [];
 60 
 61     var field;
 62     for (field in manifest_object) {
 63       if (manifest_object.hasOwnProperty(field)) {
 64         manifest_fields.push(field);
 65       }
 66     }
 67 
 68     for (field in constants.COMMON_FIELDS) {
 69       if (constants.COMMON_FIELDS[field].required) {
 70         required_fields.push(field);
 71       }
 72       else {
 73         optional_fields.push(field);
 74       }
 75     }
 76 
 77     var application_type = manifest_object.type;
 78     for (field in constants.APPLICATION_FIELDS[application_type]) {
 79       if (constants.APPLICATION_FIELDS[application_type][field].required) {
 80         required_fields.push(field);
 81       }
 82       else {
 83         optional_fields.push(field);
 84       }
 85     }
 86 
 87     var all_fields = required_fields.concat(optional_fields);
 88     var missing_required_fields = misc.array_difference(required_fields, manifest_fields);
 89 
 90     // Check that the manifest contains all the required fields
 91     if (missing_required_fields.length > 0) {
 92       return callback(new Error(sprintf('Manifest file is missing required fields: %s',
 93                                         missing_required_fields.join(', '))));
 94     }
 95 
 96     // Check that the manifest contains valid optional fields (if any)
 97     if (!misc.array_is_subset_of(manifest_fields, all_fields)) {
 98       var invalid_fields = misc.array_difference(manifest_fields, all_fields);
 99       return callback(new Error(sprintf('Manifest contains invalid fields: %s', invalid_fields)));
100     }
101 
102     // Field validation
103     var type, validator;
104     var all_fields_object = misc.merge(constants.COMMON_FIELDS, constants.APPLICATION_FIELDS[application_type]);
105 
106     // Currently the options object which is passed to each validator function only holds information about
107     // the absolute path to the directory where the manifest file is located.
108     var options = { 'manifest_path': path.dirname(manifest_path) };
109 
110     async.forEach(manifest_fields, function(field, callback) {
111       var validator_name;
112       type = all_fields_object[field].type;
113       validator = all_fields_object[field].validator;
114 
115       if (validator) {
116         // Field has a custom validator specified, use this one
117         validator_name = validator;
118       }
119       else {
120         // Field has no validator specified, use a generic type validator
121         if (type === 'string') {
122           validator_name = 'valid_string';
123         }
124         else if (type === 'number') {
125           validator_name = 'valid_number';
126         }
127       }
128 
129       // Validate the field
130       if (type === 'array') {
131         validators.validate_array(manifest_object[field], validator_name, options, function(error) {
132           if (error) {
133             return callback(new InvalidFieldError(field, error.message));
134           }
135 
136           callback();
137         });
138       }
139       else {
140         validators.validate_value(manifest_object[field], validator_name, options, function(error) {
141           if (error) {
142             return callback(new InvalidFieldError(field, error.message));
143           }
144 
145           callback();
146         });
147       }
148     },
149 
150     function(error) {
151       if (error) {
152         return callback(error);
153       }
154 
155       // If we came so far, we have a valid manifest file
156       callback(null, manifest_object);
157     });
158   });
159 };
160 
161 /**
162  * Return manifest data as a Javascript object.
163  *
164  * @param {String} manifest_path Absolute path to the manifest file.
165  * @param {Function} callback Callback which is called with the error as the first argument if there is an error and
166  *                            with manifest_object as the second argument otherwise.
167  */
168 exports.get_manifest_data_as_object = function(manifest_path, merge_with_defaults, callback) {
169   path.exists(manifest_path, function(exists) {
170     if (!exists) {
171       return callback(new Error('Invalid path to the manifest file: ' + manifest_path), null);
172     }
173 
174     ufs.json_file(manifest_path, function(err, obj) {
175       var default_values, manifest_object;
176 
177       if (err) {
178         callback(err);
179         return;
180       }
181 
182       if (merge_with_defaults) {
183         // Merge the manifest file values with the default values for this application type
184         default_values = constants.DEFAULT_VALUES[obj.type];
185         obj = misc.merge(default_values, obj);
186       }
187 
188       callback(null, obj);
189     });
190   });
191 };
