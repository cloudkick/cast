 18 var sys = require('sys');
 19 var sprintf = require('extern/sprintf').sprintf;
 20 var log = require('util/log');
 21 var version = require('util/version');
 22 var ps = require('util/pubsub');
 23 var config = require('util/config');
 24 var getopt = require('util/getopt');
 25 var req = require('util/requirements');
 26 var init = require('cast-agent/init');
 27 
 28 /*
 29  * Object containing services which are started on startup
 30  */
 31 var services = {
 32   'http': 'services/http',
 33   'health': 'services/health',
 34   'agent': 'services/agent'
 35 };
 36 
 37 var requirements = {
 38   'node_version': ['0.3.0', req.compare_versions, 'Minimum node version must be 0.3.0'],
 39   'runit_services_path': [null, req.is_defined, 'runsvdir must be running before the agent can be started']
 40 };
 41 
 42 /**
 43  * Entry point to the Cast Agent Application
 44  */
 45 exports.run = function() {
 46   if (process.env.hasOwnProperty('CAST_DEBUG')) {
 47     log.set_loglevel('debug');
 48   }
 49   else {
 50     log.set_loglevel('info');
 51   }
 52 
 53   var called_stop = false;
 54 
 55   var p = getopt.parser();
 56   p.banner = 'Usage: cast-agent [options]';
 57   p.parse(process.argv);
 58 
 59   ps.once(ps.AGENT_STATE_STOP, function() {
 60     called_stop = true;
 61   });
 62 
 63   ps.once(ps.AGENT_STATE_EXIT, function(args) {
 64     if (args && (args.why !== undefined && args.value !== undefined)) {
 65       log.info('Agent exited, stage: %s, reason: %s', args.why, args.value);
 66     }
 67 
 68     if (called_stop === false) {
 69       ps.emit(ps.AGENT_STATE_STOP);
 70       called_stop = true;
 71     }
 72   });
 73 
 74   ps.once(ps.AGENT_CONFIG_DONE, function() {
 75 
 76     process.addListener('SIGINT', function() {
 77       log.debug('Caught SIGINT, exiting....');
 78       ps.emit(ps.AGENT_STATE_EXIT, {'why': 'signal', 'value': 'SIGINT'});
 79       process.exit();
 80     });
 81 
 82     req.meets_requirements(requirements, function(err, meet_requirements) {
 83       if (err || !meet_requirements) {
 84         ps.emit(ps.AGENT_STATE_EXIT, {'why': 'requirements', 'value': err});
 85         return;
 86       }
 87 
 88       init.initialize(function(err) {
 89         var service, service_module;
 90 
 91         if (err) {
 92           ps.emit(ps.AGENT_STATE_EXIT, {'why': 'intialization', 'value': err});
 93           return;
 94         }
 95 
 96         // Run all the activated services
 97         for (service in services) {
 98           if (services.hasOwnProperty(service)) {
 99             service_module = services[service];
100             require(service_module).load();
101           }
102         }
103 
104         ps.emit(ps.AGENT_STATE_START);
105       });
106     });
107   });
108 
109   config.setup(function(err) {
110     if (err) {
111       ps.emit(ps.AGENT_STATE_EXIT, {'why': 'config', 'value': err});
112     }
113     else {
114       ps.emit(ps.AGENT_CONFIG_DONE);
115     }
116   });
117 };
