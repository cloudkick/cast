 18 var norris = require('norris');
 19 var http = require('util/http');
 20 var clutch = require('extern/clutch');
 21 
 22 function list_facts(req, res) {
 23   norris.get(function(facts) {
 24     http.return_json(res, 200, facts);
 25   });
 26 }
 27 
 28 exports.urls = clutch.route([
 29   ['GET /$', list_facts]
 30 ]);
