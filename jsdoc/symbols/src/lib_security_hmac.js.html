 18 var config = require('util/config');
 19 var crypto = require('crypto');
 20 var misc = require('util/misc');
 21 
 22 var signature_header = 'x-cast-signature';
 23 var nonce_header = 'x-cast-nonce';
 24 var signature_encoding = 'base64';
 25 
 26 function hmac_genreate(input)
 27 {
 28   var h = crypto.createHmac(config.get().hmac_algorithm, config.get().secret);
 29 
 30   input.forEach(function(item) {
 31     h.update(item);
 32   });
 33 
 34   return h.digest(signature_encoding);
 35 }
 36 
 37 /* To prevent timing attacks, you must not just do a plain
 38  * string comparision of an HMAC.  The following function is constant time,
 39  * if the input HMAC is of the correct length, and assuming v8
 40  * doesn't optimize it all out.
 41  */
 42 function hmac_validate(a, b)
 43 {
 44   if (a.length !== b.length) {
 45     return false;
 46   }
 47 
 48   var rv = 0;
 49 
 50   for (var i = 0; i < a.length; i++) {
 51     rv = rv | (a[i] ^ b[i]);
 52   }
 53 
 54   if (rv === 0) {
 55     return true;
 56   }
 57 
 58   return false;
 59 }
 60 
 61 /**
 62  * Signs a request with the default HMAC algorithm and secret.
 63  *
 64  * @param {String} method  HTTP Method to sign.
 65  * @param {String} url  HTTP url to sign.
 66  * @param {Object} headers  Headers to set, modified by this function.
 67  */
 68 exports.create_http_hmac = function(method, url, headers)
 69 {
 70   var inputs = [method, url];
 71 
 72   if (!headers[nonce_header]) {
 73     headers[nonce_header] = misc.randstr(32);
 74   }
 75 
 76   inputs.push(headers[nonce_header]);
 77 
 78   headers[signature_header] = hmac_genreate(inputs);
 79 };
 80 
 81 
 82 exports.validate_http_hmac = function(method, url, headers)
 83 {
 84   var inputs = [method, url];
 85 
 86   if (!headers[nonce_header]) {
 87     headers[nonce_header] = misc.randstr(32);
 88   }
 89 
 90   if (!headers[signature_header]) {
 91     headers[signature_header] = '';
 92   }
 93 
 94   inputs.push(headers[nonce_header]);
 95 
 96   var expected = hmac_genreate(inputs);
 97 
 98   return hmac_validate(expected, headers[signature_header]);
 99 };
