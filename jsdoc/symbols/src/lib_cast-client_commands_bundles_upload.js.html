 18 var sys = require('sys');
 19 var fs = require('fs');
 20 var path = require('path');
 21 var http = require('util/http');
 22 var crypto = require('crypto');
 23 var pumpfile = require('util/http_pumpfile');
 24 var client_config = require('util/config');
 25 var dotfiles = require('util/client_dotfiles');
 26 var misc = require('util/misc');
 27 var spinner = require('util/spinner');
 28 var manifest = require('manifest/index');
 29 var MANIFEST_FILENAME = require('manifest/constants').MANIFEST_FILENAME;
 30 var async = require('extern/async');
 31 
 32 /** Configuration options for upload subcommand */
 33 exports.config = {
 34   short_description: 'Upload an application bundle.',
 35   long_description: 'Upload an application bundle to a remote server.',
 36   required_arguments: [],
 37   optional_arguments: [['version', 'The bundle version to upload, defaults to the newest available']]
 38 };
 39 
 40 /**
 41  * Handler for uploading a bundle.
 42  * @param {Object} args Command line arguments.
 43  * @param {String} args.version Application version number to create.
 44  */
 45 exports.handle_command = function(args)
 46 {
 47   var manifest_path = path.join(process.cwd(), MANIFEST_FILENAME);
 48   var bundledir = path.join(dotfiles.dot_cast_project_path(process.cwd()), 'tmp');
 49   var version = args.version;
 50   var bundlename;
 51   var fpath;
 52   var sha1;
 53   var size = 0;
 54 
 55   async.series([
 56     // Validate the manifest and get the bundle name
 57     function(callback) {
 58       manifest.validate_manifest(manifest_path, function(err, app_manifest) {
 59         if (!err) {
 60           bundlename = misc.get_valid_bundle_name(app_manifest.name);
 61         }
 62         return callback(err);
 63       });
 64     },
 65 
 66     function(callback) {
 67       // If no version was specified then look up the most recently modified bundle
 68       if (!version) {
 69         fs.readdir(bundledir, function(err, files) {
 70           var msg;
 71 
 72           if (err) {
 73             msg = 'Unable to read ' + bundledir + ', have you created a bundle yet?';
 74             return callback(new Error(msg));
 75           }
 76 
 77           var newestfile;
 78           var newestmtime;
 79 
 80           async.forEach(files, function(file, callback) {
 81             fs.stat(path.join(bundledir, file), function(err, stats) {
 82               if (!err && stats.isFile() && (!newestmtime || stats.mtime > newestmtime)) {
 83                 newestmtime = stats.mtime;
 84                 newestfile = file;
 85               }
 86               return callback();
 87             });
 88           },
 89           function(err) {
 90             if (!err && !newestfile) {
 91               msg = 'No bundles found in ' + bundledir + ' have you created one yet?';
 92               err = new Error(msg);
 93             }
 94             else {
 95               fpath = path.join(bundledir, newestfile);
 96             }
 97             return callback(err);
 98           });
 99         });
100       }
101 
102       // If a version was specified, make sure it exists
103       else {
104         var full_bundlename = misc.get_full_bundle_name(bundlename, version);
105         fpath = path.join(bundledir, full_bundlename + '.tar.gz');
106         fs.stat(fpath, function(err, stats) {
107           if (err || !stats.isFile()) {
108             err = new Error('Specified version does not exist.');
109           }
110           return callback(err);
111         });
112       }
113     },
114 
115     // Calculate the SHA1
116     function(callback) {
117       sha1 = crypto.createHash('sha1');
118       var fstream = fs.createReadStream(fpath, {'bufferSize': client_config.get().fileread_buffer_size});
119       var errstate = false;
120 
121       fstream.on('data', function(data) {
122         sha1.update(data);
123         size += data.length;
124       });
125 
126       fstream.on('error', function(err) {
127         errstate = true;
128         return callback(new Error('Error reading bundle file.'));
129       });
130 
131       fstream.on('end', function() {
132         if (!errstate) {
133           return callback();
134         }
135       });
136     },
137 
138     function(callback) {
139       var remotepath = path.join('/', 'bundles', bundlename, path.basename(fpath));
140       var pbar = spinner.percentbar('Uploading ' + path.basename(fpath), size);
141       var completed = false;
142 
143       pbar.start();
144 
145       var client = http.client();
146 
147       client.on('error', function(err) {
148         completed = true;
149         return callback(new Error('A connection error occurred'));
150       });
151 
152       var bytes = 0;
153 
154       var request = client.request('PUT', remotepath, {
155         'host': 'localhost',
156         'content-length': size,
157         'transfer-encoding': 'chunked',
158         'x-content-sha1': sha1.digest('base64')
159       });
160 
161       pumpfile.pumpfileout(fpath, request,
162         function(bytes) {
163           if (!completed) {
164             pbar.tick(bytes);
165           }
166         },
167         function(err) {
168           if (!completed) {
169             completed = true;
170 
171             if (err) {
172               return callback(new Error('Error uploading bundle file'));
173             }
174             pbar.end();
175             sys.puts('Waiting for response...');
176             request.on('response', function(response) {
177               if (response.statusCode === 204) {
178                 return callback();
179               }
180               var chunks = [];
181 
182               response.on('data', function(data) {
183                 chunks.push(data);
184               });
185 
186               response.on('end', function() {
187                 try {
188                   var err = JSON.parse(chunks.join(''));
189                   return callback(new Error(err.message));
190                 }
191                 catch (err2) {
192                   return callback(new Error('Unexpected response from agent'));
193                 }
194               });
195             });
196           }
197         }
198       );
199     }
200   ],
201   function(err) {
202     if (err) {
203       sys.puts('Error: ' + err.message);
204     }
205     else {
206       sys.puts('Upload Successful');
207     }
208   });
209 };
